export const AutoCompleteDemo004 = (() => {
  const mockVal = (str, repeat = 1) => ({
      value: str.repeat(repeat),
  });
  const Demo4 = function () {
      const [options, setOptions] = createSignal([]);
      const getPanelValue = (searchText) => !searchText ? [] : [mockVal(searchText), mockVal(searchText, 2), mockVal(searchText, 3)];
      return (<AutoComplete options={options()} style={{ width: '200px' }} showSearch={{

---

```tsx
import { AutoComplete } from '@ant-design-solid/core'
import { createSignal } from 'solid-js'

const mockVal = (str: string, repeat = 1) => ({
  value: str.repeat(repeat),
})

const Demo4 = function () {
  const [options, setOptions] = createSignal<{ value: string }[]>([])

  const getPanelValue = (searchText: string) =>
    !searchText ? [] : [mockVal(searchText), mockVal(searchText, 2), mockVal(searchText, 3)]

  return (
    <AutoComplete
      options={options()}
      style={{ width: '200px' }}
      showSearch={{
        onSearch: (text) => setOptions(getPanelValue(text)),
      }}
      placeholder="input here"
    />
  )
}

export default Demo4
```

## API

### AutoComplete

Solid APIs use `class` naming. Deprecated React-compatible `popupClassName` and `dropdownClassName` are accepted for migration only.

| Property                   | Description                                                                                 | Type                                                                                      | Default         |
| -------------------------- | ------------------------------------------------------------------------------------------- | ----------------------------------------------------------------------------------------- | --------------- |
| `allowClear`               | Show clear button. Object form customizes the clear icon.                                   | `boolean \| { clearIcon?: JSX.Element }`                                                  | `false`         |
| `backfill`                 | Backfill the input with the active option when using keyboard navigation.                   | `boolean`                                                                                 | `false`         |
| `children`                 | Customize the input element. Supports native input elements and `Input.TextArea`.           | `JSX.Element`                                                                             | `<input />`     |
| `classNames`               | Customize class for each semantic structure. Supports object or function.                   | `AutoCompleteSemanticClassNames \| ((info: { props }) => AutoCompleteSemanticClassNames)` | `-`             |
| `dataSource`               | Deprecated data source of options. Use `options` instead.                                   | `(string \| { value: string; text?: JSX.Element; label?: JSX.Element })[]`                | `-`             |
| `defaultActiveFirstOption` | Whether active first option by default.                                                     | `boolean`                                                                                 | `true`          |
| `defaultOpen`              | Initial open state of popup.                                                                | `boolean`                                                                                 | `false`         |
| `defaultValue`             | Initial input value.                                                                        | `string`                                                                                  | `-`             |
| `disabled`                 | Whether disabled.                                                                           | `boolean`                                                                                 | `false`         |
| `dropdownClass`            | Deprecated popup class alias. Use `classNames.popup`.                                       | `string`                                                                                  | `-`             |
| `dropdownClassName`        | Deprecated React-compatible popup class alias. Use `classNames.popup` or `dropdownClass`.   | `string`                                                                                  | `-`             |
| `dropdownMatchSelectWidth` | Deprecated width alias. Use `popupMatchSelectWidth`.                                        | `boolean \| number`                                                                       | `true`          |
| `dropdownRender`           | Deprecated render alias. Use `popupRender`.                                                 | `(originNode: JSX.Element) => JSX.Element`                                                | `-`             |
| `dropdownStyle`            | Deprecated popup style alias. Use `styles.popup`.                                           | `JSX.CSSProperties`                                                                       | `-`             |
| `filterOption`             | If true, filter options by input. Function receives `inputValue` and `option`.              | `boolean \| ((inputValue: string, option: AutoCompleteOption) => boolean)`                | `true`          |
| `getPopupContainer`        | Parent node used to mount the popup portal.                                                 | `(triggerNode?: HTMLElement) => HTMLElement \| ShadowRoot`                                | `document.body` |
| `notFoundContent`          | Content shown when no options match.                                                        | `JSX.Element`                                                                             | `-`             |
| `open`                     | Controlled popup open state.                                                                | `boolean`                                                                                 | `-`             |
| `options`                  | Select options.                                                                             | `AutoCompleteOption[]`                                                                    | `-`             |
| `placeholder`              | Input placeholder text.                                                                     | `string`                                                                                  | `-`             |
| `popupClass`               | Popup class alias for Solid projects. Prefer `classNames.popup` for semantic styling.       | `string`                                                                                  | `-`             |
| `popupClassName`           | Deprecated React-compatible popup class alias. Use `classNames.popup` or `popupClass`.      | `string`                                                                                  | `-`             |
| `popupMatchSelectWidth`    | Determine whether popup and input are the same width. Number sets a minimum width.          | `boolean \| number`                                                                       | `true`          |
| `popupRender`              | Customize popup content.                                                                    | `(originNode: JSX.Element) => JSX.Element`                                                | `-`             |
| `popupStyle`               | Popup inline style. Prefer `styles.popup` for semantic styling.                             | `JSX.CSSProperties`                                                                       | `-`             |
| `showSearch`               | Search configuration.                                                                       | `true \| AutoCompleteShowSearch`                                                          | `true`          |
| `size`                     | Input size.                                                                                 | `'small' \| 'middle' \| 'large'`                                                          | `-`             |
| `status`                   | Validation status.                                                                          | `'error' \| 'warning'`                                                                    | `-`             |
| `styles`                   | Customize inline style for each semantic structure. Supports object or function.            | `AutoCompleteSemanticStyles \| ((info: { props }) => AutoCompleteSemanticStyles)`         | `-`             |
| `value`                    | Controlled input value.                                                                     | `string`                                                                                  | `-`             |
| `variant`                  | Variants of input.                                                                          | `'outlined' \| 'borderless' \| 'filled' \| 'underlined'`                                  | `'outlined'`    |
| `virtual`                  | Virtual scroll compatibility flag. Current implementation renders the option list directly. | `boolean`                                                                                 | `true`          |
| `onBlur`                   | Called when leaving the input.                                                              | `(event: FocusEvent) => void`                                                             | `-`             |
| `onChange`                 | Called when selecting an option or changing input value.                                    | `(value: string) => void`                                                                 | `-`             |
| `onClear`                  | Called when clear.                                                                          | `() => void`                                                                              | `-`             |
| `onDropdownVisibleChange`  | Deprecated open callback alias. Use `onOpenChange`.                                         | `(open: boolean) => void`                                                                 | `-`             |
| `onFocus`                  | Called when entering the input.                                                             | `(event: FocusEvent) => void`                                                             | `-`             |
| `onInputKeyDown`           | Called when key pressed.                                                                    | `(event: KeyboardEvent) => void`                                                          | `-`             |
| `onOpenChange`             | Called when popup open state changes.                                                       | `(open: boolean) => void`                                                                 | `-`             |
| `onPopupScroll`            | Called when popup scrolls.                                                                  | `(event: UIEvent) => void`                                                                | `-`             |
| `onSelect`                 | Called when an option is selected.                                                          | `(value: string, option: AutoCompleteOption) => void`                                     | `-`             |

### showSearch

| Property       | Description                     | Type                                                                       | Default |
| -------------- | ------------------------------- | -------------------------------------------------------------------------- | ------- |
| `filterOption` | Filter options by input.        | `boolean \| ((inputValue: string, option: AutoCompleteOption) => boolean)` | `true`  |
| `onSearch`     | Called when searching items.    | `(value: string) => void`                                                  | `-`     |
| `searchIcon`   | Search icon compatibility slot. | `JSX.Element`                                                              | `-`     |

### Methods

| Name      | Description   |
| --------- | ------------- |
| `blur()`  | Remove focus. |
| `focus()` | Get focus.    |

### Semantic DOM

| Key        | Description               |
| ---------- | ------------------------- |
| `root`     | Root wrapper.             |
| `selector` | Input selector wrapper.   |
| `input`    | Native input or textarea. |
| `clear`    | Clear button.             |
| `popup`    | Popup root.               |
| `list`     | Popup list container.     |
| `option`   | Option item.              |
| `empty`    | Empty content.            |

### AutoCompleteOption

| Property   | Description                                             | Type                | Default |
| ---------- | ------------------------------------------------------- | ------------------- | ------- |
| `label`    | Option display label. Falls back to value when omitted. | `JSX.Element`       | `-`     |
| `value`    | Option value inserted into the input.                   | `string`            | `-`     |
| `disabled` | Disables selecting this option.                         | `boolean`           | `false` |
| `class`    | Option class.                                           | `string`            | `-`     |
| `style`    | Option style.                                           | `JSX.CSSProperties` | `-`     |
| `title`    | Option title attribute.                                 | `string`            | `-`     |
