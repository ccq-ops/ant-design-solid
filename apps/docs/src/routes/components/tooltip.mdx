### Custom Container And Ref

:::preview

<TooltipDemo010 />

export const TooltipDemo010 = (() => {
  const Demo10 = function () {
      let container;
      let tooltip;
      return (<Space>
        <div ref={container}/>
        <Tooltip ref={(ref) => {
              tooltip = ref;
          }} title="Mounted into a custom container" getPopupContainer={() => container ?? document.body}>
          <Button>Custom container</Button>
        </Tooltip>
        <Button onClick={() => tooltip?.forceAlign()}>Align</Button>
      </Space>);
  };

return Demo10
})()

---

```tsx
import { Button, Space, Tooltip, type TooltipRef } from '@ant-design-solid/core'

const Demo10 = function () {
  let container: HTMLDivElement | undefined
  let tooltip: TooltipRef | undefined

  return (
    <Space>
      <div ref={container} />
      <Tooltip
        ref={(ref) => {
          tooltip = ref
        }}
        title="Mounted into a custom container"
        getPopupContainer={() => container ?? document.body}
      >
        <Button>Custom container</Button>
      </Tooltip>
      <Button onClick={() => tooltip?.forceAlign()}>Align</Button>
    </Space>
  )
}

export default Demo10
```

## API

| Property               | Description                                                             | Type                                                                                                                                                             | Default   |
| ---------------------- | ----------------------------------------------------------------------- | ---------------------------------------------------------------------------------------------------------------------------------------------------------------- | --------- |
| `ref`                  | Imperative Tooltip ref.                                                 | `TooltipRef \| { current?: TooltipRef } \| ((ref: TooltipRef) => void)`                                                                                          | `-`       |
| `title`                | Tooltip content.                                                        | `JSX.Element \| () => JSX.Element`                                                                                                                               | `-`       |
| `overlay`              | Tooltip content alias.                                                  | `JSX.Element \| () => JSX.Element`                                                                                                                               | `-`       |
| `color`                | Overlay background color. Text color adapts for hex colors.             | `string`                                                                                                                                                         | `-`       |
| `placement`            | Overlay placement relative to the trigger.                              | `'top' \| 'left' \| 'right' \| 'bottom' \| 'topLeft' \| 'topRight' \| 'bottomLeft' \| 'bottomRight' \| 'leftTop' \| 'leftBottom' \| 'rightTop' \| 'rightBottom'` | `'top'`   |
| `builtinPlacements`    | Per-placement alignment config overrides.                               | `Partial<Record<TooltipPlacement, TooltipAlignConfig>>`                                                                                                          | `-`       |
| `trigger`              | Interaction that opens the overlay. Arrays combine multiple triggers.   | `'hover' \| 'click' \| 'focus' \| 'contextMenu' \| TooltipTrigger[]`                                                                                             | `'hover'` |
| `open`                 | Controlled overlay open state.                                          | `boolean`                                                                                                                                                        | `-`       |
| `defaultOpen`          | Initial overlay open state for uncontrolled usage.                      | `boolean`                                                                                                                                                        | `false`   |
| `onOpenChange`         | Called when overlay open state changes.                                 | `(open: boolean) => void`                                                                                                                                        | `-`       |
| `afterOpenChange`      | Called after the open state changes.                                    | `(open: boolean) => void`                                                                                                                                        | `-`       |
| `mouseEnterDelay`      | Delay before opening on hover, in seconds.                              | `number`                                                                                                                                                         | `0.1`     |
| `mouseLeaveDelay`      | Delay before closing after hover leaves, in seconds.                    | `number`                                                                                                                                                         | `0.1`     |
| `align`                | Additional positioning offset/alignment options.                        | `TooltipAlignConfig`                                                                                                                                             | `-`       |
| `arrow`                | Shows the arrow and optionally points it at the trigger center.         | `boolean \| { pointAtCenter?: boolean }`                                                                                                                         | `true`    |
| `autoAdjustOverflow`   | Flips placement when the overlay would leave the viewport.              | `boolean \| { adjustX?: boolean; adjustY?: boolean }`                                                                                                            | `true`    |
| `fresh`                | Updates hidden cached content while the tooltip is closed.              | `boolean`                                                                                                                                                        | `false`   |
| `destroyOnHidden`      | Unmounts the overlay DOM when closed.                                   | `boolean`                                                                                                                                                        | `false`   |
| `destroyTooltipOnHide` | Legacy alias for `destroyOnHidden`.                                     | `boolean \| { keepParent?: boolean }`                                                                                                                            | `false`   |
| `rootClass`            | Additional class for the overlay root.                                  | `string`                                                                                                                                                         | `-`       |
| `rootClassName`        | Legacy React-style alias for `rootClass`. Prefer `rootClass`.           | `string`                                                                                                                                                         | `-`       |
| `openClass`            | Additional class for the trigger while open.                            | `string`                                                                                                                                                         | `-`       |
| `openClassName`        | Legacy React-style alias for `openClass`. Prefer `openClass`.           | `string`                                                                                                                                                         | `-`       |
| `classNames`           | Semantic class names for `root`, `container`, and `arrow`.              | `Partial<Record<'root' \| 'container' \| 'arrow', string>> \| (info) => Partial<Record<'root' \| 'container' \| 'arrow', string>>`                               | `-`       |
| `styles`               | Semantic styles for `root`, `container`, and `arrow`.                   | `Partial<Record<'root' \| 'container' \| 'arrow', JSX.CSSProperties>> \| (info) => Partial<Record<'root' \| 'container' \| 'arrow', JSX.CSSProperties>>`         | `-`       |
| `overlayClass`         | Legacy additional class for the overlay root. Prefer `classNames.root`. | `string`                                                                                                                                                         | `-`       |
| `overlayClassName`     | Legacy React-style alias for `overlayClass`. Prefer `classNames.root`.  | `string`                                                                                                                                                         | `-`       |
| `overlayStyle`         | Legacy inline style for the overlay root. Prefer `styles.root`.         | `JSX.CSSProperties`                                                                                                                                              | `-`       |
| `overlayInnerStyle`    | Legacy inline style for the inner content. Prefer `styles.container`.   | `JSX.CSSProperties`                                                                                                                                              | `-`       |
| `zIndex`               | Overrides overlay z-index.                                              | `number`                                                                                                                                                         | `-`       |
| `getPopupContainer`    | Returns the element used to mount the overlay portal.                   | `(triggerNode?: HTMLElement) => HTMLElement`                                                                                                                     | `-`       |

### TooltipRef

| Property        | Description                            | Type             |
| --------------- | -------------------------------------- | ---------------- |
| `forceAlign`    | Recalculates popup position.           | `() => void`     |
| `nativeElement` | Wrapped trigger element.               | `HTMLElement`    |
| `popupElement`  | Popup root element when it is mounted. | `HTMLDivElement` |

### Semantic DOM

| Slot        | Description                  |
| ----------- | ---------------------------- |
| `root`      | Popup root element.          |
| `container` | Popup inner content element. |
| `arrow`     | Popup arrow element.         |

### TooltipAlignConfig

| Property          | Description                               | Type                                       |
| ----------------- | ----------------------------------------- | ------------------------------------------ |
| `points`          | Alignment points for compatibility.       | `[string, string]`                         |
| `offset`          | Extra x/y offset applied after placement. | `[number \| string, number \| string]`     |
| `targetOffset`    | Target offset for compatibility.          | `[number \| string, number \| string]`     |
| `overflow`        | Overflow adjustment flags.                | `{ adjustX?: boolean; adjustY?: boolean }` |
| `useCssRight`     | Compatibility flag.                       | `boolean`                                  |
| `useCssBottom`    | Compatibility flag.                       | `boolean`                                  |
| `useCssTransform` | Compatibility flag.                       | `boolean`                                  |

### ConfigProvider

| Property             | Description                                                  | Type                                                                 | Default |
| -------------------- | ------------------------------------------------------------ | -------------------------------------------------------------------- | ------- |
| `tooltip.unique`     | Keeps only one visible tooltip under the provider at a time. | `boolean`                                                            | `false` |
| `tooltip.arrow`      | Default arrow setting for tooltips under the provider.       | `boolean \| { pointAtCenter?: boolean }`                             | `true`  |
| `tooltip.trigger`    | Default trigger for tooltips under the provider.             | `'hover' \| 'click' \| 'focus' \| 'contextMenu' \| TooltipTrigger[]` | `-`     |
| `tooltip.class`      | Default root class for tooltips under the provider.          | `string`                                                             | `-`     |
| `tooltip.style`      | Default root style for tooltips under the provider.          | `JSX.CSSProperties`                                                  | `-`     |
| `tooltip.classNames` | Default semantic class names.                                | `TooltipSemanticClassNames \| (info) => TooltipSemanticClassNames`   | `-`     |
| `tooltip.styles`     | Default semantic styles.                                     | `TooltipSemanticStyles \| (info) => TooltipSemanticStyles`           | `-`     |
