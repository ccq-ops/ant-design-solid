### Async loading

:::preview

<TreeDemo006 />

export const TreeDemo006 = (() => {
  const initialData = [
      { title: 'Load more', key: 'load-more', isLeaf: false },
      { title: 'Static node', key: 'static-node' },
  ];
  const Demo6 = function () {
      const [treeData, setTreeData] = createSignal(initialData);
      const loadData = async (node) => {
          if (node.key !== 'load-more')
              return;
          await new Promise((resolve) => setTimeout(resolve, 600));
          setTreeData((nodes) => nodes.map((item) => item.key === node.key
              ? {
                  ...item,
                  children: [
                      { title: 'Loaded child 1', key: 'loaded-child-1' },
                      { title: 'Loaded child 2', key: 'loaded-child-2' },
                  ],
              }
              : item));
      };
      return <Tree loadData={loadData} treeData={treeData()}/>;
  };

return Demo6
})()

---

```tsx
import { createSignal } from 'solid-js'
import { Tree } from '@ant-design-solid/core'

const initialData = [
  { title: 'Load more', key: 'load-more', isLeaf: false },
  { title: 'Static node', key: 'static-node' },
]

const Demo6 = function () {
  const [treeData, setTreeData] = createSignal(initialData)

  const loadData = async (node) => {
    if (node.key !== 'load-more') return
    await new Promise((resolve) => setTimeout(resolve, 600))
    setTreeData((nodes) =>
      nodes.map((item) =>
        item.key === node.key
          ? {
              ...item,
              children: [
                { title: 'Loaded child 1', key: 'loaded-child-1' },
                { title: 'Loaded child 2', key: 'loaded-child-2' },
              ],
            }
          : item,
      ),
    )
  }

  return <Tree loadData={loadData} treeData={treeData()} />
}

export default Demo6
```

## API

### Tree

| Property              | Description                                           | Type                                                                       | Default |
| --------------------- | ----------------------------------------------------- | -------------------------------------------------------------------------- | ------- |
| `allowDrop`           | Controls whether a dragged node may drop on a target. | `(info: { dropNode: TreeDataNode; dropPosition: number }) => boolean`      | `-`     |
| `autoExpandParent`    | Expands ancestors for controlled `expandedKeys`.      | `boolean`                                                                  | `false` |
| `blockNode`           | Makes node content fill the row.                      | `boolean`                                                                  | `false` |
| `checkable`           | Shows checkboxes for checkable nodes.                 | `boolean`                                                                  | `false` |
| `checkedKeys`         | Controlled checked node keys.                         | `TreeKey[] \| { checked: TreeKey[]; halfChecked: TreeKey[] }`              | `-`     |
| `checkStrictly`       | Disables parent-child checkbox conduction.            | `boolean`                                                                  | `false` |
| `classNames`          | Semantic class names for tree slots.                  | `TreeSemanticClassNames`                                                   | `-`     |
| `defaultCheckedKeys`  | Initial checked node keys.                            | `TreeKey[]`                                                                | `-`     |
| `defaultExpandAll`    | Expands all nodes initially.                          | `boolean`                                                                  | `false` |
| `defaultExpandedKeys` | Initial expanded node keys.                           | `TreeKey[]`                                                                | `-`     |
| `defaultExpandParent` | Expands ancestors of `defaultExpandedKeys`.           | `boolean`                                                                  | `true`  |
| `defaultSelectedKeys` | Initial selected node keys.                           | `TreeKey[]`                                                                | `-`     |
| `disabled`            | Disables the entire tree.                             | `boolean`                                                                  | `false` |
| `draggable`           | Enables node dragging or custom drag behavior.        | `boolean \| ((node: TreeDataNode) => boolean) \| TreeDraggableConfig`      | `false` |
| `expandedKeys`        | Controlled expanded node keys.                        | `TreeKey[]`                                                                | `-`     |
| `fieldNames`          | Maps custom field names for title, key, and children. | `{ title?: string; key?: string; children?: string }`                      | `-`     |
| `filterTreeNode`      | Marks matching nodes as filtered.                     | `(node: TreeDataNode) => boolean`                                          | `-`     |
| `height`              | Fixed tree height with scroll container.              | `number`                                                                   | `-`     |
| `icon`                | Custom node icon when `showIcon` is enabled.          | `TreeIcon`                                                                 | `-`     |
| `loadData`            | Loads children data when an async node expands.       | `(node: TreeDataNode) => Promise<unknown> \| unknown`                      | `-`     |
| `loadedKeys`          | Controlled loaded node keys.                          | `TreeKey[]`                                                                | `-`     |
| `motion`              | Motion One animation props for tree nodes.            | `MotionComponentProps \| false`                                            | default |
| `multiple`            | Allows selecting multiple nodes.                      | `boolean`                                                                  | `false` |
| `rootStyle`           | Additional style for the root tree element.           | `JSX.CSSProperties`                                                        | `-`     |
| `selectable`          | Whether nodes can be selected.                        | `boolean`                                                                  | `true`  |
| `selectedKeys`        | Controlled selected node keys.                        | `TreeKey[]`                                                                | `-`     |
| `showIcon`            | Shows node icons.                                     | `boolean`                                                                  | `false` |
| `showLine`            | Shows line styling and optional leaf icon.            | `boolean \| { showLeafIcon?: boolean \| TreeIcon }`                        | `false` |
| `styles`              | Semantic inline styles for tree slots.                | `TreeSemanticStyles`                                                       | `-`     |
| `switcherIcon`        | Custom expand/collapse switcher icon.                 | `TreeIcon`                                                                 | `-`     |
| `switcherLoadingIcon` | Custom loading icon while `loadData` is pending.      | `JSX.Element`                                                              | `-`     |
| `titleRender`         | Custom node title renderer.                           | `(node: TreeDataNode) => JSX.Element`                                      | `-`     |
| `virtual`             | Enables virtual rendering when `height` is set.       | `boolean`                                                                  | `true`  |
| `onCheck`             | Called when node checked state changes.               | `(checkedKeys: TreeKey[] \| TreeCheckedKeys, info: TreeCheckInfo) => void` | `-`     |
| `onDoubleClick`       | Called when a node is double-clicked.                 | `(event: MouseEvent, node: TreeDataNode) => void`                          | `-`     |
| `onDragEnd`           | Called when node dragging ends.                       | `(info: TreeDragInfo) => void`                                             | `-`     |
| `onDragEnter`         | Called when dragged node enters a target.             | `(info: TreeDragEnterInfo) => void`                                        | `-`     |
| `onDragLeave`         | Called when dragged node leaves a target.             | `(info: TreeDragInfo) => void`                                             | `-`     |
| `onDragOver`          | Called when dragged node moves over a target.         | `(info: TreeDragInfo) => void`                                             | `-`     |
| `onDragStart`         | Called when node dragging starts.                     | `(info: TreeDragInfo) => void`                                             | `-`     |
| `onDrop`              | Called when a node is dropped.                        | `(info: TreeDropInfo) => void`                                             | `-`     |
| `onExpand`            | Called when node expansion changes.                   | `(expandedKeys: TreeKey[], info: TreeExpandInfo) => void`                  | `-`     |
| `onLoad`              | Called after `loadData` completes.                    | `(loadedKeys: TreeKey[], info: TreeLoadInfo) => void`                      | `-`     |
| `onRightClick`        | Called when a node receives a context menu event.     | `(info: { event: MouseEvent; node: TreeDataNode }) => void`                | `-`     |
| `onSelect`            | Called when node selection changes.                   | `(selectedKeys: TreeKey[], info: TreeSelectInfo) => void`                  | `-`     |

### DirectoryTree

DirectoryTree supports all Tree props and adds directory-style defaults.

| Property       | Description                                | Type                                | Default   |
| -------------- | ------------------------------------------ | ----------------------------------- | --------- |
| `expandAction` | Controls whether clicking expands folders. | `false \| 'click' \| 'doubleClick'` | `'click'` |
| `multiple`     | Enables ctrl/meta and shift multi-select.  | `boolean`                           | `false`   |

### TreeNode

TreeNode is available for JSX children usage.

| Property   | Description              | Type          | Default |
| ---------- | ------------------------ | ------------- | ------- |
| `title`    | Node title content.      | `JSX.Element` | `-`     |
| `key`      | Unique node key.         | `TreeKey`     | `-`     |
| `children` | Child TreeNode elements. | `JSX.Element` | `-`     |

### TreeDataNode

| Property          | Description                                   | Type             | Default |
| ----------------- | --------------------------------------------- | ---------------- | ------- |
| `title`           | Node title content.                           | `JSX.Element`    | `-`     |
| `key`             | Unique node key.                              | `TreeKey`        | `-`     |
| `children`        | Child tree nodes.                             | `TreeDataNode[]` | `-`     |
| `checkable`       | Overrides whether this node shows a checkbox. | `boolean`        | `-`     |
| `disabled`        | Disables the node.                            | `boolean`        | `false` |
| `disableCheckbox` | Disables this node checkbox.                  | `boolean`        | `false` |
| `icon`            | Node-level icon.                              | `TreeIcon`       | `-`     |
| `isLeaf`          | Marks the node as a leaf or async parent.     | `boolean`        | `-`     |
| `selectable`      | Overrides whether this node can be selected.  | `boolean`        | `-`     |

### Ref

| Method     | Description             | Type                                     |
| ---------- | ----------------------- | ---------------------------------------- |
| `scrollTo` | Scrolls to a tree node. | `(options: TreeScrollToOptions) => void` |
