<CheckboxDemo004 />

export const CheckboxDemo004 = (() => {
  const plainOptions = ['Apple', 'Pear', 'Orange'];
  const Demo4 = function () {
      const [checkedList, setCheckedList] = createSignal(['Apple', 'Orange']);
      const checkAll = createMemo(() => plainOptions.length === checkedList().length);
      const indeterminate = createMemo(() => checkedList().length > 0 && checkedList().length < plainOptions.length);
      return (<Space direction="vertical">
        <Checkbox indeterminate={indeterminate()} checked={checkAll()} onChange={(event) => setCheckedList(event.currentTarget.checked ? plainOptions : [])}>
          Check all
        </Checkbox>
        <Checkbox.Group options={plainOptions} value={checkedList()} onChange={(next) => setCheckedList(next)}/>
      </Space>);
  };

return Demo4
})()

```tsx
import { createMemo, createSignal } from 'solid-js'
import { Checkbox, Space } from '@ant-design-solid/core'

const plainOptions = ['Apple', 'Pear', 'Orange']

const Demo4 = function () {
  const [checkedList, setCheckedList] = createSignal<string[]>(['Apple', 'Orange'])
  const checkAll = createMemo(() => plainOptions.length === checkedList().length)
  const indeterminate = createMemo(
    () => checkedList().length > 0 && checkedList().length < plainOptions.length,
  )

  return (
    <Space direction="vertical">
      <Checkbox
        indeterminate={indeterminate()}
        checked={checkAll()}
        onChange={(event) => setCheckedList(event.currentTarget.checked ? plainOptions : [])}
      >
        Check all
      </Checkbox>
      <Checkbox.Group
        options={plainOptions}
        value={checkedList()}
        onChange={(next) => setCheckedList(next as string[])}
      />
    </Space>
  )
}

export default Demo4
```

## API

### Checkbox

| Property         | Description                                                | Type                                            | Default |
| ---------------- | ---------------------------------------------------------- | ----------------------------------------------- | ------- |
| `checked`        | Controlled checked state.                                  | `boolean`                                       | `-`     |
| `defaultChecked` | Initial checked state for uncontrolled usage.              | `boolean`                                       | `false` |
| `disabled`       | Disables the checkbox.                                     | `boolean`                                       | `false` |
| `indeterminate`  | Displays the mixed selection state.                        | `boolean`                                       | `false` |
| `skipGroup`      | Ignores parent `Checkbox.Group` control.                   | `boolean`                                       | `false` |
| `value`          | Value used when the checkbox is inside `Checkbox.Group`.   | `string \| number \| boolean`                   | `-`     |
| `class`          | Custom root class. Solid replacement for antd `className`. | `string`                                        | `-`     |
| `style`          | Custom root style.                                         | `JSX.CSSProperties`                             | `-`     |
| `classNames`     | Semantic classes for `root`, `icon`, and `label`.          | `Record<string, string> \| Function`            | `-`     |
| `styles`         | Semantic styles for `root`, `icon`, and `label`.           | `Record<string, JSX.CSSProperties> \| Function` | `-`     |
| `prefixCls`      | Custom CSS class prefix.                                   | `string`                                        | `-`     |
| `children`       | Checkbox label content.                                    | `JSX.Element`                                   | `-`     |
| `onChange`       | Called when the native checkbox change event fires.        | `JSX.EventHandler<HTMLInputElement, Event>`     | `-`     |

### Checkbox.Group

| Property       | Description                                       | Type                                    | Default |
| -------------- | ------------------------------------------------- | --------------------------------------- | ------- |
| `value`        | Controlled checked values.                        | `(string \| number \| boolean)[]`       | `-`     |
| `defaultValue` | Initial checked values for uncontrolled usage.    | `(string \| number \| boolean)[]`       | `-`     |
| `options`      | Options rendered as checkboxes.                   | `OptionInput[]`                         | `-`     |
| `disabled`     | Disables all checkboxes in the group.             | `boolean`                               | `false` |
| `name`         | `name` applied to child checkbox inputs.          | `string`                                | `-`     |
| `prefixCls`    | Custom checkbox CSS class prefix.                 | `string`                                | `-`     |
| `children`     | Custom checkbox children controlled by the group. | `JSX.Element`                           | `-`     |
| `onChange`     | Called with the next checked value array.         | `(checkedValue: OptionValue[]) => void` | `-`     |

### OptionInput

```ts
type OptionValue = string | number | boolean

type OptionInput =
  | OptionValue
  | {
      label: JSX.Element
      value: OptionValue
      disabled?: boolean
      class?: string
      style?: JSX.CSSProperties
      title?: string
      id?: string
      required?: boolean
      onChange?: JSX.EventHandler<HTMLInputElement, Event>
    }
```

### Methods

| Name            | Description                        |
| --------------- | ---------------------------------- |
| `focus()`       | Focuses the checkbox input.        |
| `blur()`        | Blurs the checkbox input.          |
| `nativeElement` | Returns the native checkbox input. |
