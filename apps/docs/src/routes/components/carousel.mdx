# Carousel

Display a group of content in a carousel with dots, arrows, and autoplay.

### Basic

:::preview

<CarouselDemo001 />

export const CarouselDemo001 = (() => {
  const contentStyle = {
      margin: 0,
      height: '160px',
      color: '#fff',
      'line-height': '160px',
      'text-align': 'center',

        </div>
        <div>
          <h3 style={contentStyle}>4</h3>
        </div>
      </Carousel>);
  };

  return Demo1
})()

        <Radio.Button value="bottom">Bottom</Radio.Button>
        <Radio.Button value="start">Start</Radio.Button>

      </Radio.Group>
      <Carousel dotPlacement={dotPlacement()}>
        <div>
          <h3 style={contentStyle}>1</h3>
        </div>
        <div>
          <h3 style={contentStyle}>2</h3>
        </div>
        <div>
          <h3 style={contentStyle}>3</h3>
        </div>
        <div>
          <h3 style={contentStyle}>4</h3>
        </div>

export const CarouselDemo008 = (() => {
  const contentStyle = {
      margin: 0,

          <Button onClick={() => carousel?.goTo(1)}>Go to 2</Button>
          <Button onClick={() => carousel?.next()}>Next</Button>
          <span>Current: {current() + 1}</span>
        </Space>
      </Space>);
  };

  return Demo8

        <Button onClick={() => carousel?.next()}>Next</Button>
        <span>Current: {current() + 1}</span>

    </Space>
  )
}

## API

### Carousel

| Property           | Description                                                                  | Type                                                                         | Default     |
| ------------------ | ---------------------------------------------------------------------------- | ---------------------------------------------------------------------------- | ----------- |
| `adaptiveHeight`   | Adjust the slide height automatically.                                       | `boolean`                                                                    | `false`     |
| `afterChange`      | Callback function called after the current index changes.                    | `(current: number) => void`                                                  | `-`         |
| `appendDots`       | Custom dots container renderer.                                              | `(dots: JSX.Element) => JSX.Element`                                         | `-`         |
| `arrows`           | Whether to show switch arrows.                                               | `boolean`                                                                    | `false`     |
| `autoplay`         | Whether to scroll automatically, or display dot progress with object config. | `boolean \| { dotDuration?: boolean }`                                       | `false`     |
| `autoplaySpeed`    | Delay between each auto scroll in milliseconds.                              | `number`                                                                     | `3000`      |
| `beforeChange`     | Callback function called before the current index changes.                   | `(current: number, next: number) => void`                                    | `-`         |
| `customPaging`     | Custom dot content renderer.                                                 | `(index: number) => JSX.Element`                                             | `-`         |
| `dotPlacement`     | The position of the dots.                                                    | `'top' \| 'bottom' \| 'start' \| 'end'`                                      | `'bottom'`  |
| `dotPosition`      | Deprecated dot position alias.                                               | `'top' \| 'bottom' \| 'left' \| 'right' \| 'start' \| 'end'`                 | `'bottom'`  |
| `dots`             | Whether to show dots or dot configuration.                                   | `boolean \| { class?: string }`                                              | `true`      |
| `draggable`        | Enable scrollable via dragging on desktop.                                   | `boolean`                                                                    | `false`     |
| `easing`           | Transition interpolation function name.                                      | `string`                                                                     | `'linear'`  |
| `effect`           | Transition effect.                                                           | `'scrollx' \| 'fade'`                                                        | `'scrollx'` |
| `fade`             | Whether to use fade transition.                                              | `boolean`                                                                    | `false`     |
| `infinite`         | Infinitely wrap around contents.                                             | `boolean`                                                                    | `true`      |
| `initialSlide`     | Initial active slide index.                                                  | `number`                                                                     | `0`         |
| `nextArrow`        | Custom next arrow content.                                                   | `JSX.Element`                                                                | `-`         |
| `pauseOnDotsHover` | Pause autoplay while hovering dots.                                          | `boolean`                                                                    | `false`     |
| `pauseOnFocus`     | Pause autoplay while focused.                                                | `boolean`                                                                    | `false`     |
| `pauseOnHover`     | Pause autoplay while hovering carousel.                                      | `boolean`                                                                    | `false`     |
| `prefixCls`        | Custom CSS class prefix.                                                     | `string`                                                                     | `-`         |
| `prevArrow`        | Custom previous arrow content.                                               | `JSX.Element`                                                                | `-`         |
| `responsive`       | Responsive breakpoint settings.                                              | `{ breakpoint: number; settings: Partial<CarouselSettings> \| 'unslick' }[]` | `-`         |
| `rootClass`        | Custom class on the root element.                                            | `string`                                                                     | `-`         |
| `rtl`              | Enable right-to-left mode.                                                   | `boolean`                                                                    | `false`     |
| `slickGoTo`        | Programmatically switch to the given slide index.                            | `number`                                                                     | `-`         |
| `slidesToScroll`   | Number of slides to scroll at a time.                                        | `number`                                                                     | `1`         |
| `slidesToShow`     | Number of slides to show at a time.                                          | `number`                                                                     | `1`         |
| `speed`            | Animation speed in milliseconds.                                             | `number`                                                                     | `500`       |
| `vertical`         | Enable vertical sliding.                                                     | `boolean`                                                                    | `false`     |
| `waitForAnimate`   | Whether to wait for the animation when switching.                            | `boolean`                                                                    | `false`     |

Solid style differences: use `class` instead of `className`; use `rootClass` instead of antd React's `rootClassName`; `dots` object accepts `{ class?: string }`.

### CarouselRef

| Property   | Description                        | Type                                                 | Default |
| ---------- | ---------------------------------- | ---------------------------------------------------- | ------- |
| `goTo`     | Switches to the given slide index. | `(slide: number, dontAnimate?: boolean) => void`     | `-`     |
| `next`     | Switches to the next slide.        | `() => void`                                         | `-`     |
| `prev`     | Switches to the previous slide.    | `() => void`                                         | `-`     |
| `autoPlay` | Resumes autoplay.                  | `(playType?: 'update' \| 'leave' \| 'blur') => void` | `-`     |

### Design Token

| Token            | Description                    | Type               | Default |
| ---------------- | ------------------------------ | ------------------ | ------- |
| `arrowOffset`    | Arrow offset to Carousel edge. | `number`           | `8`     |
| `arrowSize`      | Size of arrows.                | `number`           | `16`    |
| `dotActiveWidth` | Width of active indicator.     | `number \| string` | `24`    |
| `dotGap`         | Gap between indicators.        | `number`           | `4`     |
| `dotHeight`      | Height of indicator.           | `number \| string` | `3`     |
| `dotOffset`      | Dot offset to Carousel edge.   | `number`           | `12`    |
| `dotWidth`       | Width of indicator.            | `number \| string` | `16`    |
