### Popup inheritance

:::preview

<WatermarkDemo006 />

export const WatermarkDemo006 = (() => {
  const Demo6 = function () {
      const [open, setOpen] = createSignal(false);
      return (<Watermark content="Confidential">
        <div class="h-[180px] bg-[var(--docs-surface-solid)] p-6 text-[var(--docs-text)]">
          <Button type="primary" onClick={() => setOpen(true)}>
            Open modal
          </Button>
          <Modal open={open()} title="Inherited watermark" onOk={() => setOpen(false)} onCancel={() => setOpen(false)}>
            Popup components can inherit the surrounding watermark.
          </Modal>
        </div>
      </Watermark>);
  };

return Demo6
})()

---

```tsx
import { createSignal } from 'solid-js'
import { Button, Modal, Watermark } from '@solid-ant-design/core'

const Demo6 = function () {
  const [open, setOpen] = createSignal(false)

  return (
    <Watermark content="Confidential">
      <div class="h-[180px] bg-[var(--docs-surface-solid)] p-6 text-[var(--docs-text)]">
        <Button type="primary" onClick={() => setOpen(true)}>
          Open modal
        </Button>
        <Modal
          open={open()}
          title="Inherited watermark"
          onOk={() => setOpen(false)}
          onCancel={() => setOpen(false)}
        >
          Popup components can inherit the surrounding watermark.
        </Modal>
      </div>
    </Watermark>
  )
}

export default Demo6
```

export const WatermarkDemo007 = (() => {
  const Demo7 = function () {
      const [open, setOpen] = createSignal(false);

          <Modal open={open()} title="No inherited watermark" onOk={() => setOpen(false)} onCancel={() => setOpen(false)}>
            This modal is not covered by the parent watermark.
          </Modal>
        </div>
      </Watermark>);
  };

return Demo7
})()

## API

| Property        | Description                                                          | Type                 | Default                    |
| --------------- | -------------------------------------------------------------------- | -------------------- | -------------------------- |
| `width`         | Watermark tile width. Text watermarks use estimated content width.   | `number`             | `120` for image watermark  |
| `height`        | Watermark tile height. Text watermarks use estimated content height. | `number`             | `64` for image watermark   |
| `rotate`        | Rotation angle in degrees.                                           | `number`             | `-22`                      |
| `zIndex`        | Watermark overlay z-index.                                           | `number`             | `999`                      |
| `image`         | Image URL. Has higher priority than text content.                    | `string`             | `-`                        |
| `content`       | Watermark text content or multiple text lines.                       | `string \| string[]` | `-`                        |
| `font`          | Text font configuration.                                             | [Font](#font)        | [Font](#font)              |
| `gap`           | Horizontal and vertical gap between repeated tiles.                  | `[number, number]`   | `[100, 100]`               |
| `offset`        | Offset from the upper-left corner of the container.                  | `[number, number]`   | `[gap[0] / 2, gap[1] / 2]` |
| `inherit`       | Pass the watermark to popup components such as Modal and Drawer.     | `boolean`            | `true`                     |
| `onRemove`      | Callback when the watermark overlay is removed by DOM mutation.      | `() => void`         | `-`                        |
| `prefixCls`     | Custom CSS class prefix.                                             | `string`             | `-`                        |
| `class`         | Additional root class.                                               | `string`             | `-`                        |
| `rootClassName` | Additional root class alias for antd compatibility.                  | `string`             | `-`                        |
| `children`      | Content covered by the watermark.                                    | `JSX.Element`        | `-`                        |

### Font

| Property     | Description               | Type                                                    | Default               |
| ------------ | ------------------------- | ------------------------------------------------------- | --------------------- |
| `color`      | Font color.               | `string`                                                | `rgba(0, 0, 0, 0.15)` |
| `fontSize`   | Font size.                | `number \| string`                                      | `16`                  |
| `fontWeight` | Font weight.              | `'normal' \| 'lighter' \| 'bold' \| 'bolder' \| number` | `normal`              |
| `fontFamily` | Font family.              | `string`                                                | `sans-serif`          |
| `fontStyle`  | Font style.               | `'none' \| 'normal' \| 'italic' \| 'oblique'`           | `normal`              |
| `textAlign`  | Text alignment direction. | `'left' \| 'right' \| 'center' \| 'start' \| 'end'`     | `center`              |
