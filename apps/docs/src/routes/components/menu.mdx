# Menu

Menu displays a list of navigation actions with optional groups and submenus.

export const MenuDemo011 = (() => {
  function currentTheme() {
      return document.documentElement.dataset.theme === 'dark' ? 'dark' : 'light';
  }
  const Demo11 = function () {
      const [menuTheme, setMenuTheme] = createSignal(currentTheme());
      onMount(() => {
          const observer = new MutationObserver(() => setMenuTheme(currentTheme()));

import { Menu } from '@ant-design-solid/core'
import { createSignal, onCleanup, onMount } from 'solid-js'

function currentTheme() {
  return document.documentElement.dataset.theme === 'dark' ? 'dark' : 'light'
}

const Demo11 = function () {
  const [menuTheme, setMenuTheme] = createSignal(currentTheme())

  onMount(() => {
    const observer = new MutationObserver(() => setMenuTheme(currentTheme()))
    observer.observe(document.documentElement, {
      attributes: true,
      attributeFilter: ['data-theme'],
    })
    onCleanup(() => observer.disconnect())
  })

  return (
    <Menu
      mode="horizontal"
      theme={menuTheme()}
      triggerSubMenuAction="click"
      overflowedIndicator="More"
      popupRender={(node) => (
        <div class="docs-border docs-surface-solid rounded border p-1 shadow">{node}</div>
      )}
      items={[
        { key: 'mail', label: 'Mail' },
        {
          key: 'settings',
          label: 'Settings',
          popupClass: 'min-w-44',
          popupOffset: [0, 8],
          children: [
            { key: 'profile', label: 'Profile' },
            { key: 'security', label: 'Security' },
          ],
        },
      ]}
    />
  )
}

export default Demo11

## API

### Menu

| Property               | Description                                         | Type                                                | Default      |
| ---------------------- | --------------------------------------------------- | --------------------------------------------------- | ------------ |
| `items`                | Menu items, submenus, groups, dividers, or `null`.  | `MenuItem[]`                                        | `-`          |
| `children`             | Compound `Menu.Item` / `Menu.SubMenu` entries.      | `JSX.Element`                                       | `-`          |
| `mode`                 | Menu layout mode.                                   | `'vertical' \| 'horizontal' \| 'inline'`            | `'vertical'` |
| `selectedKeys`         | Controlled selected item keys.                      | `string[]`                                          | `-`          |
| `defaultSelectedKeys`  | Initial selected item keys for uncontrolled usage.  | `string[]`                                          | `-`          |
| `openKeys`             | Controlled open submenu keys.                       | `string[]`                                          | `-`          |
| `defaultOpenKeys`      | Initial open submenu keys for uncontrolled usage.   | `string[]`                                          | `-`          |
| `inlineCollapsed`      | Collapses inline menus.                             | `boolean`                                           | `false`      |
| `inlineIndent`         | Inline nested item indent in pixels.                | `number`                                            | `24`         |
| `multiple`             | Enables multiple selected keys.                     | `boolean`                                           | `false`      |
| `selectable`           | Allows item selection.                              | `boolean`                                           | `true`       |
| `expandIcon`           | Custom submenu expand icon.                         | `JSX.Element \| (props) => JSX.Element`             | `-`          |
| `forceSubMenuRender`   | Keeps submenu lists mounted while hidden.           | `boolean`                                           | `false`      |
| `overflowedIndicator`  | Horizontal overflow indicator hook.                 | `JSX.Element`                                       | `-`          |
| `popupRender`          | Custom popup submenu renderer.                      | `(node, props) => JSX.Element`                      | `-`          |
| `subMenuOpenDelay`     | Hover open delay in seconds.                        | `number`                                            | `0`          |
| `subMenuCloseDelay`    | Hover close delay in seconds.                       | `number`                                            | `0.1`        |
| `theme`                | Menu color theme class.                             | `'light' \| 'dark'`                                 | `'light'`    |
| `tooltip`              | Collapsed item title behavior; `false` disables.    | `false \| { title?, class?, style? }`               | `-`          |
| `triggerSubMenuAction` | Submenu trigger interaction.                        | `'hover' \| 'click'`                                | `'hover'`    |
| `classNames`           | Semantic class hooks.                               | `Record<MenuSemanticName, string> \| (info) => ...` | `-`          |
| `styles`               | Semantic style hooks.                               | `Record<MenuSemanticName, CSSProperties> \| ...`    | `-`          |
| `zIndex`               | Overrides popup submenu z-index.                    | `number`                                            | `-`          |
| `getPopupContainer`    | Returns popup submenu portal container.             | `(triggerNode?: HTMLElement) => HTMLElement`        | `-`          |
| `onClick`              | Called when an enabled menu item is clicked.        | `(info: MenuClickInfo) => void`                     | `-`          |
| `onSelect`             | Called when an item is selected.                    | `(info: MenuSelectInfo) => void`                    | `-`          |
| `onDeselect`           | Called when an item is deselected in multiple mode. | `(info: MenuSelectInfo) => void`                    | `-`          |
| `onOpenChange`         | Called when open submenu keys change.               | `(openKeys: string[]) => void`                      | `-`          |

### MenuItemType

| Property   | Description                                  | Type            | Default |
| ---------- | -------------------------------------------- | --------------- | ------- |
| `key`      | Stable menu item key.                        | `string`        | `-`     |
| `label`    | Menu label content.                          | `JSX.Element`   | `-`     |
| `icon`     | Optional leading icon.                       | `JSX.Element`   | `-`     |
| `disabled` | Disables item interaction.                   | `boolean`       | `false` |
| `danger`   | Applies danger styling.                      | `boolean`       | `false` |
| `extra`    | Extra content rendered at the item end.      | `JSX.Element`   | `-`     |
| `title`    | Native title used in collapsed inline menus. | `string`        | `-`     |
| `type`     | Item marker.                                 | `'item'`        | `-`     |
| `class`    | Custom item class.                           | `string`        | `-`     |
| `style`    | Custom item style.                           | `CSSProperties` | `-`     |

### SubMenuType

| Property       | Description                                                     | Type                           | Default |
| -------------- | --------------------------------------------------------------- | ------------------------------ | ------- |
| `key`          | Stable submenu key.                                             | `string`                       | `-`     |
| `label`        | Submenu title content.                                          | `JSX.Element`                  | `-`     |
| `children`     | Nested menu entries.                                            | `MenuItem[]`                   | `-`     |
| `icon`         | Optional leading icon.                                          | `JSX.Element`                  | `-`     |
| `disabled`     | Disables submenu interaction.                                   | `boolean`                      | `false` |
