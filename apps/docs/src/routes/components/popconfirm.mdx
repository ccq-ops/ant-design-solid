# Popconfirm

Popconfirm asks users to confirm a potentially risky action.

### Basic

:::preview

<PopconfirmDemo001 />

export const PopconfirmDemo001 = (() => {
  const Demo1 = function () {
      return (<Popconfirm title="Delete?" description="This action cannot be undone." onConfirm={() => {
              message.success('Deleted');
          }}>
        <Button danger>Delete</Button>
      </Popconfirm>);
  };

return Demo1
})()

### Controlled

:::preview

<PopconfirmDemo006 />

export const PopconfirmDemo006 = (() => {
  const Demo6 = function () {
      const [open, setOpen] = createSignal(false);
      return (<Space>
        <Popconfirm open={open()} onOpenChange={setOpen} title="Controlled popup" onConfirm={() => {
              setOpen(false);
          }} onCancel={() => setOpen(false)}>
          <Button>Controlled</Button>
        </Popconfirm>
        <Button onClick={() => setOpen(true)}>Open from outside</Button>
      </Space>);
  };

return Demo6
})()

---

```tsx
import { createSignal } from 'solid-js'
import { Button, Popconfirm, Space } from '@solid-ant-design/core'

const Demo6 = function () {

  return (
    <Space>
      <Popconfirm
        open={open()}
        onOpenChange={setOpen}
        title="Controlled popup"
        onConfirm={() => {
          setOpen(false)
        }}
        onCancel={() => setOpen(false)}
      >
        <Button>Controlled</Button>
      </Popconfirm>
      <Button onClick={() => setOpen(true)}>Open from outside</Button>
    </Space>
  )
}

export default Demo6
```

### Async confirm

:::preview

<PopconfirmDemo007 />

export const PopconfirmDemo007 = (() => {
  const Demo7 = function () {
      return (<Popconfirm title="Save changes?" description="The popup closes after the promise resolves." onConfirm={() => new Promise((resolve) => {
              window.setTimeout(() => {
                  message.success('Saved');
                  resolve();
              }, 600);
          })}>
        <Button>Async save</Button>
      </Popconfirm>);
  };

return Demo7
})()

          }, 600)

      <Button>Async save</Button>
    </Popconfirm>
  )
}

export default Demo7

## API

| Property             | Description                                                       | Type                                                                                                                                                           | Default                       |
| -------------------- | ----------------------------------------------------------------- | -------------------------------------------------------------------------------------------------------------------------------------------------------------- | ----------------------------- |
| `title`              | Main confirmation title.                                          | `JSX.Element \| () => JSX.Element`                                                                                                                             | `-`                           |
| `description`        | Additional explanatory content.                                   | `JSX.Element \| () => JSX.Element`                                                                                                                             | `-`                           |
| `open`               | Controlled popup open state.                                      | `boolean`                                                                                                                                                      | `-`                           |
| `defaultOpen`        | Initial popup open state.                                         | `boolean`                                                                                                                                                      | `false`                       |
| `disabled`           | Disables opening the popup.                                       | `boolean`                                                                                                                                                      | `false`                       |
| `placement`          | Popup placement relative to the trigger.                          | `top` \| `left` \| `right` \| `bottom` \| `topLeft` \| `topRight` \| `bottomLeft` \| `bottomRight` \| `leftTop` \| `leftBottom` \| `rightTop` \| `rightBottom` | `top`                         |
| `trigger`            | Popup trigger mode.                                               | `hover` \| `focus` \| `click` \| `contextMenu` \| `Array<string>`                                                                                              | `click`                       |
| `okText`             | OK button text.                                                   | `JSX.Element`                                                                                                                                                  | `OK`                          |
| `okType`             | OK button type.                                                   | `ButtonType`                                                                                                                                                   | `primary`                     |
| `okButtonProps`      | OK button props.                                                  | `ButtonProps`                                                                                                                                                  | `-`                           |
| `cancelText`         | Cancel button text.                                               | `JSX.Element`                                                                                                                                                  | `Cancel`                      |
| `cancelButtonProps`  | Cancel button props.                                              | `ButtonProps`                                                                                                                                                  | `-`                           |
| `showCancel`         | Whether to show the cancel button.                                | `boolean`                                                                                                                                                      | `true`                        |
| `icon`               | Custom confirmation icon.                                         | `JSX.Element`                                                                                                                                                  | `<ExclamationCircleFilled />` |
| `onConfirm`          | Called when OK is clicked. Rejected promises keep the popup open. | `(event?: MouseEvent) => void \| Promise<void>`                                                                                                                | `-`                           |
| `onCancel`           | Called when Cancel is clicked.                                    | `(event?: MouseEvent) => void`                                                                                                                                 | `-`                           |
| `onOpenChange`       | Called whenever the popup requests an open-state change.          | `(open: boolean) => void`                                                                                                                                      | `-`                           |
| `afterOpenChange`    | Called after popup open state changes.                            | `(open: boolean) => void`                                                                                                                                      | `-`                           |
| `onPopupClick`       | Called when the popup content is clicked.                         | `(event: MouseEvent) => void`                                                                                                                                  | `-`                           |
| `mouseEnterDelay`    | Delay in seconds before opening on mouse enter.                   | `number`                                                                                                                                                       | `0.1`                         |
| `mouseLeaveDelay`    | Delay in seconds before closing on mouse leave.                   | `number`                                                                                                                                                       | `0.1`                         |
| `align`              | Fine-grained popup alignment config.                              | `TooltipAlignConfig`                                                                                                                                           | `-`                           |
| `arrow`              | Show arrow or point arrow at target center.                       | `boolean \| { pointAtCenter?: boolean }`                                                                                                                       | `true`                        |
| `autoAdjustOverflow` | Whether to adjust placement when popup is off screen.             | `boolean \| TooltipOverflowConfig`                                                                                                                             | `true`                        |
| `fresh`              | Keep popup content updating while hidden.                         | `boolean`                                                                                                                                                      | `false`                       |
| `destroyOnHidden`    | Destroy popup DOM when hidden.                                    | `boolean`                                                                                                                                                      | `true`                        |
| `rootClass`          | Class on popup root.                                              | `string`                                                                                                                                                       | `-`                           |
| `openClass`          | Class on trigger while popup is open.                             | `string`                                                                                                                                                       | `-`                           |
| `overlayClass`       | Class on popup root.                                              | `string`                                                                                                                                                       | `-`                           |
| `overlayStyle`       | Style on popup root.                                              | `JSX.CSSProperties`                                                                                                                                            | `-`                           |
| `overlayInnerStyle`  | Style on popup content container.                                 | `JSX.CSSProperties`                                                                                                                                            | `-`                           |
| `classNames`         | Semantic DOM class names.                                         | `Partial<Record<'root' \| 'container' \| 'arrow' \| 'icon' \| 'title' \| 'description' \| 'buttons', string>> \| ((info) => object)`                           | `-`                           |
| `styles`             | Semantic DOM styles.                                              | `Partial<Record<'root' \| 'container' \| 'arrow' \| 'icon' \| 'title' \| 'description' \| 'buttons', JSX.CSSProperties>> \| ((info) => object)`                | `-`                           |
| `color`              | Popup background color.                                           | `string`                                                                                                                                                       | `-`                           |
| `zIndex`             | Overrides popup z-index.                                          | `number`                                                                                                                                                       | `-`                           |
| `getPopupContainer`  | Returns the element used to mount the popup portal.               | `(triggerNode?: HTMLElement) => HTMLElement`                                                                                                                   | `-`                           |
| `class`              | Class on trigger wrapper.                                         | `string`                                                                                                                                                       | `-`                           |
| `children`           | Trigger element.                                                  | `JSX.Element`                                                                                                                                                  | `-`                           |
