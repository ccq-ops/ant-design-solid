export const ImageDemo008 = (() => {
  const Demo = function () {
      const [open, setOpen] = createSignal(false);

})()

        src="https://dummyimage.com/120x80/e6f4ff/1677ff.png&text=Image"
        preview={{ open: open(), onOpenChange: setOpen }}
        alt="Controlled"
      />
    </Space>
  )
}

export default Demo

## API

### Image

| Property            | Description                                        | Type                                                                      | Default |
| ------------------- | -------------------------------------------------- | ------------------------------------------------------------------------- | ------- |
| `src`               | Image source URL.                                  | `string`                                                                  | `-`     |
| `alt`               | Image alternate text.                              | `string`                                                                  | `-`     |
| `width`             | Image width.                                       | `number \| string`                                                        | `-`     |
| `height`            | Image height.                                      | `number \| string`                                                        | `-`     |
| `fallback`          | Fallback image URL used after load failure.        | `string`                                                                  | `-`     |
| `placeholder`       | Placeholder shown before image load completes.     | `JSX.Element \| boolean \| { progress?: boolean \| ImageProgressConfig }` | `-`     |
| `preview`           | Preview behavior configuration.                    | `boolean \| PreviewType`                                                  | `true`  |
| `class`             | Root class name.                                   | `string`                                                                  | `-`     |
| `rootClass`         | Extra root class name.                             | `string`                                                                  | `-`     |
| `classNames`        | Semantic DOM class names.                          | `ImageSemanticClassNames \| ((info) => ImageSemanticClassNames)`          | `-`     |
| `styles`            | Semantic DOM styles.                               | `ImageSemanticStyles \| ((info) => ImageSemanticStyles)`                  | `-`     |
| `wrapperStyle`      | Deprecated alias for `styles.root`.                | `JSX.CSSProperties`                                                       | `-`     |
| `zIndex`            | Compatibility shortcut for `preview.zIndex`.       | `number`                                                                  | `-`     |
| `getPopupContainer` | Compatibility shortcut for `preview.getContainer`. | `(triggerNode?: HTMLElement) => HTMLElement`                              | `-`     |
| `onLoad`            | Called when the image loads.                       | `JSX.EventHandler<HTMLImageElement, Event>`                               | `-`     |
| `onError`           | Called when the image fails to load.               | `JSX.EventHandler<HTMLImageElement, Event>`                               | `-`     |

Native image attributes such as `loading`, `crossOrigin`, `decoding`, `srcSet`, `sizes`, `referrerPolicy`, and `fetchPriority` are passed to the underlying `img`.

### PreviewType

| Property          | Description                                      | Type                                                                                    | Default                      |
| ----------------- | ------------------------------------------------ | --------------------------------------------------------------------------------------- | ---------------------------- |
| `src`             | Preview image source.                            | `string`                                                                                | Image `src`                  |
| `alt`             | Preview image alt text.                          | `string`                                                                                | Image `alt`                  |
| `open`            | Controlled preview open state.                   | `boolean`                                                                               | `-`                          |
| `defaultOpen`     | Initial uncontrolled open state.                 | `boolean`                                                                               | `false`                      |
| `onOpenChange`    | Called when preview open state changes.          | `(open: boolean, prevOpen: boolean) => void`                                            | `-`                          |
| `getContainer`    | Preview portal container.                        | `(triggerNode?: HTMLElement) => HTMLElement`                                            | `document.body`              |
| `zIndex`          | Preview z-index.                                 | `number`                                                                                | token `zIndexPopupBase + 80` |
| `mask`            | Preview mask content or visibility.              | `boolean \| JSX.Element`                                                                | `true`                       |
| `scaleStep`       | Zoom step.                                       | `number`                                                                                | `0.5`                        |
| `minScale`        | Minimum zoom scale.                              | `number`                                                                                | `1`                          |
| `maxScale`        | Maximum zoom scale.                              | `number`                                                                                | `50`                         |
| `closeIcon`       | Custom close icon.                               | `JSX.Element`                                                                           | `<CloseOutlined />`          |
| `icons`           | Custom operation icons.                          | `Partial<Record<ImageTransformAction, JSX.Element>>`                                    | `-`                          |
| `cover`           | Cover displayed over the thumbnail.              | `JSX.Element \| { coverNode?: JSX.Element; placement?: 'top' \| 'bottom' \| 'center' }` | `-`                          |
| `imageRender`     | Custom preview image render.                     | `(originNode, info) => JSX.Element`                                                     | `-`                          |
| `actionsRender`   | Custom toolbar render.                           | `(originNode, info) => JSX.Element`                                                     | `-`                          |
| `onTransform`     | Called after zoom, rotate, flip, move, or reset. | `(info: ImageTransformInfo) => void`                                                    | `-`                          |
| `visible`         | Deprecated alias for `open`.                     | `boolean`                                                                               | `-`                          |
| `onVisibleChange` | Deprecated alias for `onOpenChange`.             | `(visible: boolean, prevVisible: boolean) => void`                                      | `-`                          |
| `toolbarRender`   | Deprecated alias for `actionsRender`.            | `(originNode, info) => JSX.Element`                                                     | `-`                          |

### Image.PreviewGroup

| Property     | Description                              | Type                                                                                                | Default |
| ------------ | ---------------------------------------- | --------------------------------------------------------------------------------------------------- | ------- |
| `items`      | Preview items without child images.      | `(string \| { src?: string; alt?: string; width?: number \| string; height?: number \| string })[]` | `-`     |
| `fallback`   | Fallback image URL for grouped previews. | `string`                                                                                            | `-`     |
| `preview`    | Group preview configuration.             | `boolean \| GroupPreviewType`                                                                       | `true`  |
| `classNames` | Semantic DOM class names.                | `ImageSemanticClassNames \| ((info) => ImageSemanticClassNames)`                                    | `-`     |
| `styles`     | Semantic DOM styles.                     | `ImageSemanticStyles \| ((info) => ImageSemanticStyles)`                                            | `-`     |

### GroupPreviewType

Includes most `PreviewType` fields plus:

| Property         | Description                                | Type                                                 | Default |
| ---------------- | ------------------------------------------ | ---------------------------------------------------- | ------- |
| `current`        | Controlled current preview index.          | `number`                                             | `-`     |
| `defaultCurrent` | Initial current preview index.             | `number`                                             | `0`     |
| `onChange`       | Called when current preview index changes. | `(current: number, prevCurrent: number) => void`     | `-`     |
| `onOpenChange`   | Called when group preview opens or closes. | `(open: boolean, info: { current: number }) => void` | `-`     |
| `countRender`    | Custom progress text.                      | `(current: number, total: number) => JSX.Element`    | `-`     |

### ImageProgressConfig

| Property  | Description                     | Type                                                      | Default |
| --------- | ------------------------------- | --------------------------------------------------------- | ------- |
| `percent` | Progress percent.               | `number`                                                  | `-`     |
| `render`  | Custom progress content render. | `(progress: JSX.Element, percent: number) => JSX.Element` | `-`     |

### Semantic DOM

| Slot                             | Description                   |
| -------------------------------- | ----------------------------- |
| `root`                           | Image wrapper.                |
| `image`                          | Underlying image element.     |
| `cover`                          | Thumbnail cover.              |
| `placeholder.progress.root`      | Progress placeholder wrapper. |
| `placeholder.progress.content`   | Progress content.             |
| `placeholder.progress.rail`      | Progress rail.                |
| `placeholder.progress.indicator` | Progress text.                |
| `popup.root`                     | Preview popup root.           |
| `popup.mask`                     | Preview mask.                 |
| `popup.body`                     | Preview body.                 |
| `popup.close`                    | Preview close button.         |
| `popup.footer`                   | Preview footer.               |
| `popup.actions`                  | Preview action toolbar.       |
