# Spin

Spin displays a loading indicator for pages, sections, or inline content.

### Basic

```tsx
import { Button, Space, Spin } from '@ant-design-solid/core'
import { createEffect, createSignal, onCleanup } from 'solid-js'

const Demo1 = function () {
  return <Spin />
}

export default Demo1
```

### Fullscreen

```tsx
import { Button, Space, Spin } from '@ant-design-solid/core'
import { createEffect, createSignal, onCleanup } from 'solid-js'

const Demo6 = function () {
  const [fullscreen, setFullscreen] = createSignal(false)

  return (
    <>
      <Button onClick={() => setFullscreen(true)}>Show fullscreen spinner</Button>
      <Spin fullscreen spinning={fullscreen()} description="Loading fullscreen" />
    </>
  )
}

export default Demo6
```

### Progress

```tsx
import { Button, Flex, Space, Spin, Switch } from '@ant-design-solid/core'
import { createEffect, createSignal, onCleanup } from 'solid-js'

const Demo7 = function () {
  const [auto, setAuto] = createSignal(false)
  const [percent, setPercent] = createSignal(-50)

  createEffect(() => {
    const timer = window.setTimeout(() => {
      setPercent((value) => {
        const nextPercent = value + 5
        return nextPercent > 150 ? -50 : nextPercent
      })
    }, 100)

    onCleanup(() => window.clearTimeout(timer))
  })

  const mergedPercent = () => (auto() ? 'auto' : percent())

  return (
    <Flex align="center" gap="middle">
      <Switch
        checkedChildren="Auto"
        unCheckedChildren="Auto"
        checked={auto()}
        onChange={(checked) => {
          setAuto(checked)
          setPercent(-50)
        }}
      />
      <Spin percent={mergedPercent()} size="small" />
      <Spin percent={mergedPercent()} />
      <Spin percent={mergedPercent()} size="large" />
    </Flex>
  )
}

export default Demo7
```

### Custom indicator

```tsx
import { Button, Space, Spin } from '@ant-design-solid/core'
import { LoadingOutlined } from '@ant-design-solid/icons'
import { createEffect, createSignal, onCleanup } from 'solid-js'

const LoadingIcon = function () {
  return <LoadingOutlined spin style={{ color: '#1677ff', 'font-size': '24px' }} />
}

const Demo9 = function () {
  return <Spin indicator={<LoadingIcon />} />
}

export default Demo9
```

### Default indicator

```tsx
import { Button, Space, Spin } from '@ant-design-solid/core'
import { createEffect, createSignal, onCleanup } from 'solid-js'

const Demo10 = function () {
  createEffect(() => {
    Spin.setDefaultIndicator(<span style={{ color: '#1677ff' }}>Default loading</span>)
    onCleanup(() => Spin.setDefaultIndicator(undefined))
  })

  return <Spin description="Using global default indicator" />
}

export default Demo10
```

## API

| Property       | Description                                        | Type                                                                            | Default    |
| -------------- | -------------------------------------------------- | ------------------------------------------------------------------------------- | ---------- |
| `spinning`     | Controls whether the spinner is active.            | `boolean`                                                                       | `true`     |
| `size`         | Spinner size. `default` is kept for compatibility. | `'small' \| 'medium' \| 'large' \| 'default'`                                   | `'medium'` |
| `description`  | Loading description displayed with the spinner.    | `JSX.Element`                                                                   | `-`        |
| `tip`          | Deprecated loading text. Use `description`.        | `JSX.Element`                                                                   | `-`        |
| `delay`        | Delay in milliseconds before showing the spinner.  | `number`                                                                        | `-`        |
| `fullscreen`   | Renders the spinner as a fullscreen overlay.       | `boolean`                                                                       | `false`    |
| `indicator`    | Custom indicator content.                          | `JSX.Element`                                                                   | `-`        |
| `percent`      | Progress percent, or automatic simulated progress. | `number \| 'auto'`                                                              | `-`        |
| `wrapperClass` | Class applied to the nested loading wrapper.       | `string`                                                                        | `-`        |
| `classNames`   | Semantic DOM class names.                          | `Partial<Record<SpinSemanticSlot, string>> \| ((info) => Record...)`            | `-`        |
| `styles`       | Semantic DOM styles.                               | `Partial<Record<SpinSemanticSlot, JSX.CSSProperties>> \| ((info) => Record...)` | `-`        |
| `children`     | Nested content covered by loading overlay.         | `JSX.Element`                                                                   | `-`        |

## Static methods

| Method                | Description                             | Type                                            |
| --------------------- | --------------------------------------- | ----------------------------------------------- |
| `setDefaultIndicator` | Sets the global default Spin indicator. | `(indicator: JSX.Element \| undefined) => void` |

## Semantic DOM

| Slot          | Description                                |
| ------------- | ------------------------------------------ |
| `root`        | Root Spin node, or nested loading wrapper. |
| `section`     | Indicator and description section.         |
| `indicator`   | Indicator wrapper.                         |
| `description` | Description content.                       |
| `container`   | Nested children container.                 |
| `tip`         | Deprecated alias for `description`.        |
| `mask`        | Fullscreen or nested overlay mask.         |
