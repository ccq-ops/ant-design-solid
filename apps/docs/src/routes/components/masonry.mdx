# Masonry

A masonry layout component for displaying content with different heights.

### Basic

```tsx
import { createSignal } from 'solid-js'
import { Button, Card, Flex, Masonry, useToken } from '@ant-design-solid/core'
import { CloseOutlined } from '@ant-design-solid/icons'

const heights = [150, 50, 90, 70, 110, 150, 130, 80, 50, 90, 100, 150, 70, 50, 80]

const Demo4 = function () {
  const token = useToken()
  const [items, setItems] = createSignal(
    heights.map((height, index) => ({
      key: index,
      column: index % 4,
      data: height,
    })),
  )

  const removeItem = (removeKey) => {
    setItems((prevItems) => prevItems.filter(({ key }) => key !== removeKey))
  }

  const addItem = () => {
    setItems((prevItems) => [
      ...prevItems,
      {
        key: prevItems.length ? prevItems[prevItems.length - 1].key + 1 : 0,
        data: Math.floor(Math.random() * 100) + 50,
      },
    ])
  }

  return (
    <Flex vertical gap={16}>
      <Masonry
        columns={4}
        gutter={16}
        items={items()}
        itemRender={({ data, key }) => (
          <Card size="small" style={{ height: data, position: 'relative' }}>
            {Number(key) + 1}
            <Button
              aria-label={`Remove item ${Number(key) + 1}`}
              style={{
                position: 'absolute',
                top: `${token().paddingSM}px`,
                right: `${token().paddingSM}px`,
              }}
              size="small"
              icon={<CloseOutlined />}
              onClick={() => removeItem(key)}
            />
          </Card>
        )}
        onLayoutChange={(sortedItems) => {
          setItems((prevItems) => {
            let changed = false
            const nextItems = prevItems.map((item) => {
              const matchItem = sortedItems.find((sortedItem) => sortedItem.key === item.key)
              if (!matchItem || item.column === matchItem.column) return item
              changed = true
              return {
                ...item,
                column: matchItem.column,
              }
            })
            return changed ? nextItems : prevItems
          })
        }}
      />
      <Button block onClick={addItem}>
        Add Item
      </Button>
    </Flex>
  )
}

export default Demo4
```

## API

| Property             | Description                                                                   | Type                                                                               | Default |
| -------------------- | ----------------------------------------------------------------------------- | ---------------------------------------------------------------------------------- | ------- |
| `prefixCls`          | Custom CSS class prefix.                                                      | `string`                                                                           | `-`     |
| `class`              | Additional root class.                                                        | `string`                                                                           | `-`     |
| `classList`          | Conditional root classes.                                                     | `Record<string, boolean \| undefined>`                                             | `-`     |
| `rootClass`          | Additional root class using Solid naming.                                     | `string`                                                                           | `-`     |
| `rootClassName`      | Compatibility alias for root class. Prefer `rootClass` in Solid code.         | `string`                                                                           | `-`     |
| `style`              | Inline root style.                                                            | `JSX.CSSProperties \| string`                                                      | `-`     |
| `columns`            | Number of columns, can be a fixed value or a responsive configuration.        | `number \| Partial<Record<MasonryBreakpoint, number>>`                             | `3`     |
| `gutter`             | Spacing, fixed, responsive, or `[horizontal, vertical]`.                      | `MasonryGap \| [MasonryGap, MasonryGap]`                                           | `0`     |
| `items`              | Masonry items.                                                                | `MasonryItem<T>[]`                                                                 | `-`     |
| `itemRender`         | Custom item rendering function.                                               | `(item: MasonryRenderItem<T>) => JSX.Element`                                      | `-`     |
| `children`           | Manual masonry children used when items are not provided.                     | `JSX.Element`                                                                      | `-`     |
| `fresh`              | Whether to continuously monitor the size changes of child items.              | `boolean`                                                                          | `false` |
| `classNames`         | Semantic class names for `root` and `item`. Supports object or function.      | `MasonrySemanticClassNames \| ((info: { props }) => MasonrySemanticClassNamesMap)` | `-`     |
| `styles`             | Semantic inline styles for `root` and `item`. Supports object or function.    | `MasonrySemanticStyles \| ((info: { props }) => MasonrySemanticStylesMap)`         | `-`     |
| `onLayoutChange`     | Callback for column sorting changes.                                          | `(info: Array<{ key: MasonryItemKey; column: number }>) => void`                   | `-`     |
| `onLayoutInfoChange` | Called with columns, column heights, and item placement after layout changes. | `(info: MasonryLayoutInfo<T>) => void`                                             | `-`     |

### MasonryItem

| Property   | Description                                           | Type               | Default |
| ---------- | ----------------------------------------------------- | ------------------ | ------- |
| `key`      | Unique identifier for the item.                       | `string \| number` | `-`     |
| `data`     | Custom data storage.                                  | `T`                | `-`     |
| `children` | Custom display content, takes precedence over render. | `JSX.Element`      | `-`     |
| `column`   | Specifies the column to which the item belongs.       | `number`           | `-`     |
| `height`   | Height of the item.                                   | `number`           | `-`     |

### Gap

`Gap` represents the spacing between items. It can either be a fixed value or a responsive configuration.

```ts
type MasonryGap =
  | undefined
  | number
  | Partial<Record<'xs' | 'sm' | 'md' | 'lg' | 'xl' | 'xxl', number | undefined>>
type MasonryGutter = MasonryGap | [MasonryGap, MasonryGap]
```

### Semantic DOM

`classNames` and `styles` support: `root`, `item`.
