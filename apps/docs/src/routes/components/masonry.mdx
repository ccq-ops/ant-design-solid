# Masonry

Display variable-height content in balanced columns.

### Basic

```tsx
import { Card, Masonry } from '@ant-design-solid/core'

const baseItems = [
  { key: 1, data: { title: 'Mountain', height: 120, background: 'var(--docs-primary-soft)' } },
  { key: 2, data: { title: 'Forest', height: 180, background: 'var(--docs-surface-subtle)' } },
  {
    key: 3,
    data: {
      title: 'River',
      height: 96,
      background: 'color-mix(in srgb, var(--docs-primary-soft) 45%, var(--docs-surface-solid))',
    },
  },
  {
    key: 4,
    data: {
      title: 'City',
      height: 150,
      background: 'color-mix(in srgb, var(--docs-border-subtle) 30%, var(--docs-surface-solid))',
    },
  },
  {
    key: 5,
    data: {
      title: 'Desert',
      height: 110,
      background:
        'color-mix(in srgb, var(--docs-primary-soft-strong) 50%, var(--docs-surface-solid))',
    },
  },
  { key: 6, data: { title: 'Ocean', height: 210, background: 'var(--docs-surface-muted)' } },
]

function DemoCard(props) {
  return (
    <Card title={props.title} style={{ background: props.background }}>
      <div style={{ height: `${props.height}px`, color: 'var(--docs-text-secondary)' }}>
        Variable-height content creates a waterfall layout.
      </div>
    </Card>
  )
}

const Demo1 = function () {
  return (
    <Masonry
      items={baseItems}
      itemRender={(item) => (
        <DemoCard
          title={item.data.title}
          height={item.data.height}
          background={item.data.background}
        />
      )}
    />
  )
}

export default Demo1
```

### Dynamic items

```tsx
import { createSignal } from 'solid-js'
import { Button, Card, Masonry, Space } from '@ant-design-solid/core'

const baseItems = [
  { key: 1, data: { title: 'Mountain', height: 120, background: 'var(--docs-primary-soft)' } },
  { key: 2, data: { title: 'Forest', height: 180, background: 'var(--docs-surface-subtle)' } },
  {
    key: 3,
    data: {
      title: 'River',
      height: 96,
      background: 'color-mix(in srgb, var(--docs-primary-soft) 45%, var(--docs-surface-solid))',
    },
  },
  {
    key: 4,
    data: {
      title: 'City',
      height: 150,
      background: 'color-mix(in srgb, var(--docs-border-subtle) 30%, var(--docs-surface-solid))',
    },
  },
]

function DemoCard(props) {
  return (
    <Card title={props.title} style={{ background: props.background }}>
      <div style={{ height: `${props.height}px`, color: 'var(--docs-text-secondary)' }}>
        fresh keeps item measurements current when content changes.
      </div>
    </Card>
  )
}

const Demo5 = function () {
  const [items, setItems] = createSignal(baseItems)

  return (
    <Space direction="vertical" style={{ width: '100%' }}>
      <Button
        onClick={() =>
          setItems((previous) => [
            ...previous,
            {
              key: Date.now(),
              data: {
                title: `New ${previous.length + 1}`,
                height: 80 + ((previous.length * 37) % 140),
                background: 'var(--docs-primary-soft)',
              },
            },
          ])
        }
      >
        Add item
      </Button>
      <Masonry
        fresh
        columns={{ xs: 1, md: 3 }}
        gutter={16}
        items={items()}
        itemRender={(item) => (
          <DemoCard
            title={item.data.title}
            height={item.data.height}
            background={item.data.background}
          />
        )}
      />
    </Space>
  )
}

export default Demo5
```

### Semantic styles

```tsx
import { Card, Masonry } from '@ant-design-solid/core'

const items = [
  { key: 1, data: { title: 'Semantic A', height: 100 } },
  { key: 2, data: { title: 'Semantic B', height: 140 } },
  { key: 3, data: { title: 'Semantic C', height: 110 } },
]

const Demo6 = function () {
  return (
    <Masonry
      columns={3}
      gutter={12}
      rootClass="semantic-masonry"
      classNames={{ root: 'semantic-root', item: 'semantic-item' }}
      styles={({ props }) => ({
        root: { padding: props.columns === 3 ? '8px' : '0' },
        item: { outline: '1px dashed var(--docs-border)' },
      })}
      items={items}
      itemRender={(item) => (
        <Card title={item.data.title}>
          <div style={{ height: `${item.data.height}px` }}>Stable semantic slots.</div>
        </Card>
      )}
    />
  )
}

export default Demo6
```

### Layout change

```tsx
import { createSignal } from 'solid-js'
import { Card, Masonry, Space } from '@ant-design-solid/core'

const items = [
  { key: 'a', data: { title: 'A', height: 120 } },
  { key: 'b', data: { title: 'B', height: 170 } },
  { key: 'c', data: { title: 'C', height: 100 } },
]

const Demo7 = function () {
  const [summary, setSummary] = createSignal('')

  return (
    <Space direction="vertical" style={{ width: '100%' }}>
      <Masonry
        columns={2}
        gutter={12}
        items={items}
        onLayoutChange={(layout) =>
          setSummary(layout.map((item) => `${item.key}:${item.column + 1}`).join(', '))
        }
        itemRender={(item) => (
          <Card title={item.data.title}>
            <div style={{ height: `${item.data.height}px` }}>Column {item.column + 1}</div>
          </Card>
        )}
      />
      <div>{summary()}</div>
    </Space>
  )
}

export default Demo7
```

## API

| Property             | Description                                                                   | Type                                                                               | Default |
| -------------------- | ----------------------------------------------------------------------------- | ---------------------------------------------------------------------------------- | ------- |
| `prefixCls`          | Custom CSS class prefix.                                                      | `string`                                                                           | `-`     |
| `class`              | Additional root class.                                                        | `string`                                                                           | `-`     |
| `classList`          | Conditional root classes.                                                     | `Record<string, boolean \| undefined>`                                             | `-`     |
| `rootClass`          | Additional root class using Solid naming.                                     | `string`                                                                           | `-`     |
| `rootClassName`      | Compatibility alias for root class. Prefer `rootClass` in Solid code.         | `string`                                                                           | `-`     |
| `style`              | Inline root style.                                                            | `JSX.CSSProperties \| string`                                                      | `-`     |
| `columns`            | Column count or responsive breakpoint map.                                    | `number \| Partial<Record<MasonryBreakpoint, number>>`                             | `3`     |
| `gutter`             | Item spacing. Supports one gap or `[horizontal, vertical]`.                   | `MasonryGap \| [MasonryGap, MasonryGap]`                                           | `0`     |
| `items`              | Data items to render.                                                         | `MasonryItem<T>[]`                                                                 | `-`     |
| `itemRender`         | Render function for each data item.                                           | `(item: MasonryRenderItem<T>) => JSX.Element`                                      | `-`     |
| `children`           | Manual masonry children used when items are not provided.                     | `JSX.Element`                                                                      | `-`     |
| `fresh`              | Continuously monitors item size changes.                                      | `boolean`                                                                          | `false` |
| `classNames`         | Semantic class names for `root` and `item`. Supports object or function.      | `MasonrySemanticClassNames \| ((info: { props }) => MasonrySemanticClassNamesMap)` | `-`     |
| `styles`             | Semantic inline styles for `root` and `item`. Supports object or function.    | `MasonrySemanticStyles \| ((info: { props }) => MasonrySemanticStylesMap)`         | `-`     |
| `onLayoutChange`     | Called when item column order changes.                                        | `(info: Array<{ key: MasonryItemKey; column: number }>) => void`                   | `-`     |
| `onLayoutInfoChange` | Called with columns, column heights, and item placement after layout changes. | `(info: MasonryLayoutInfo<T>) => void`                                             | `-`     |

### MasonryItem

| Property   | Description                                        | Type               | Default |
| ---------- | -------------------------------------------------- | ------------------ | ------- |
| `key`      | Unique item key.                                   | `string \| number` | `-`     |
| `data`     | Custom data passed through to `itemRender`.        | `T`                | `-`     |
| `children` | Custom item content. Takes precedence over render. | `JSX.Element`      | `-`     |
| `column`   | Requested zero-based column index.                 | `number`           | `-`     |
| `height`   | Optional item height metadata.                     | `number`           | `-`     |

### Gap

```ts
type MasonryGap =
  | undefined
  | number
  | Partial<Record<'xs' | 'sm' | 'md' | 'lg' | 'xl' | 'xxl', number | undefined>>
type MasonryGutter = MasonryGap | [MasonryGap, MasonryGap]
```

### Semantic DOM

`classNames` and `styles` support: `root`, `item`.
