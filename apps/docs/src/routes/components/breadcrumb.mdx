### Route params and itemRender

:::preview

<BreadcrumbDemo008 />

export const BreadcrumbDemo008 = (() => {
  const routes = [
      { title: 'Home', path: '/' },
      { title: 'Users', path: 'users' },
      { title: 'User :id', path: ':id' },
  ];
  const Demo8 = function () {
      return (<Breadcrumb items={routes} params={{ id: '42' }} itemRender={(route, _params, _routes, paths) => {
              const last = paths.length === routes.length;
              const label = route.title;
              return last ? (<span>{label}</span>) : (<a href={`#/${paths.filter(Boolean).join('/')}`}>{label}</a>);
          }}/>);
  };

return Demo8
})()

---

```tsx
import { Breadcrumb } from '@solid-ant-design/core'

const routes = [
  { title: 'Home', path: '/' },
  { title: 'Users', path: 'users' },
  { title: 'User :id', path: ':id' },
]

const Demo8 = function () {
  return (
    <Breadcrumb
      items={routes}
      params={{ id: '42' }}
      itemRender={(route, _params, _routes, paths) => {
        const last = paths.length === routes.length
        const label = route.title
        return last ? (
          <span>{label}</span>
        ) : (
          <a href={`#/${paths.filter(Boolean).join('/')}`}>{label}</a>
        )
      }}
    />
  )
}

export default Demo8
```

## API

### Breadcrumb

| Property       | Description                                                    | Type                                                                  | Default |
| -------------- | -------------------------------------------------------------- | --------------------------------------------------------------------- | ------- |
| `items`        | Breadcrumb items rendered from data.                           | `BreadcrumbRouteType[]`                                               | `-`     |
| `routes`       | Legacy data source, compatible with antd v6 deprecated routes. | `BreadcrumbRouteType[]`                                               | `-`     |
| `params`       | Route params used by `path`, `title`, `breadcrumbName`.        | `Record<string, string \| number>`                                    | `{}`    |
| `separator`    | Default separator between items.                               | `JSX.Element`                                                         | `'/'`   |
| `dropdownIcon` | Icon rendered after dropdown breadcrumb items.                 | `JSX.Element`                                                         | `âŚ„`     |
| `itemRender`   | Custom item renderer.                                          | `(route, params, routes, paths) => JSX.Element`                       | `-`     |
| `rootClass`    | Root custom class.                                             | `string`                                                              | `-`     |
| `class`        | Root element class.                                            | `string`                                                              | `-`     |
| `style`        | Root element style.                                            | `JSX.CSSProperties \| string`                                         | `-`     |
| `classNames`   | Semantic slot classes.                                         | `Partial<Record<'root' \| 'item' \| 'separator', string>>`            | `-`     |
| `styles`       | Semantic slot styles.                                          | `Partial<Record<'root' \| 'item' \| 'separator', JSX.CSSProperties>>` | `-`     |
| `children`     | Manual breadcrumb item composition.                            | `JSX.Element`                                                         | `-`     |

### Breadcrumb item data

| Property         | Description                                                | Type                                             | Default |
| ---------------- | ---------------------------------------------------------- | ------------------------------------------------ | ------- |
| `type`           | Use `'separator'` to render an independent separator item. | `'separator'`                                    | `-`     |
| `key`            | Stable item key.                                           | `string \| number`                               | `index` |
| `title`          | Item label for data-driven breadcrumbs.                    | `JSX.Element`                                    | `-`     |
| `breadcrumbName` | Legacy label alias. Prefer `title`.                        | `string`                                         | `-`     |
| `href`           | Direct link for the item.                                  | `string`                                         | `-`     |
| `path`           | Route path segment accumulated into `#/{paths}`.           | `string`                                         | `-`     |
| `menu`           | Dropdown menu configuration.                               | `MenuProps` with breadcrumb menu items           | `-`     |
| `dropdownProps`  | Props passed to the dropdown wrapper.                      | `DropdownProps`                                  | `-`     |
| `onClick`        | Click handler for interactive breadcrumb items.            | `JSX.EventHandlerUnion<HTMLElement, MouseEvent>` | `-`     |
| `separator`      | Custom separator after this item.                          | `JSX.Element`                                    | `-`     |
| `class`          | Item custom class.                                         | `string`                                         | `-`     |
| `style`          | Item custom style.                                         | `JSX.CSSProperties`                              | `-`     |
| `children`       | Legacy child menu data. Prefer `menu`.                     | `BreadcrumbItemType[]`                           | `-`     |

### Breadcrumb.Item

| Property        | Description                         | Type                                             | Default |
| --------------- | ----------------------------------- | ------------------------------------------------ | ------- |
| `href`          | Renders this manual item as a link. | `string`                                         | `-`     |
| `menu`          | Dropdown menu configuration.        | `MenuProps`                                      | `-`     |
| `dropdownProps` | Props passed to Dropdown.           | `DropdownProps`                                  | `-`     |
| `dropdownIcon`  | Icon rendered after dropdown item.  | `JSX.Element`                                    | `âŚ„`     |
| `onClick`       | Click handler for this manual item. | `JSX.EventHandlerUnion<HTMLElement, MouseEvent>` | `-`     |
| `separator`     | Custom separator after this item.   | `JSX.Element`                                    | `'/'`   |
| `children`      | Item label content.                 | `JSX.Element`                                    | `-`     |

### Breadcrumb.Separator

| Property   | Description               | Type          | Default |
| ---------- | ------------------------- | ------------- | ------- |
| `children` | Manual separator content. | `JSX.Element` | `-`     |
