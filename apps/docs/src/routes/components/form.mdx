# Form

Form stores field values, validates rules, and coordinates common input controls. The examples below follow the Ant Design v6 Form demo order and translate each demo to Solid syntax.

### Basic Usage

import { createSignal } from 'solid-js'
import { Button, Form, Input, Radio, Space, message } from '@ant-design-solid/core'
import type { FormLayout } from '@ant-design-solid/core'

const Demo3 = function () {
  const [layout, setLayout] = createSignal<FormLayout>('horizontal')

  return (
    <Space direction="vertical" class="w-100">
      <Radio.Group
        value={layout()}
        options={[
          { label: 'Horizontal', value: 'horizontal' },
          { label: 'Vertical', value: 'vertical' },
          { label: 'Inline', value: 'inline' },
        ]}
        onChange={(value) => setLayout(value as FormLayout)}
      />
      <Form layout={layout()} onFinish={() => message.success('Submitted')}>
        <Form.Item label="Field A" name="fieldA">
          <Input placeholder="input placeholder" />
        </Form.Item>
        <Form.Item label="Field B" name="fieldB">
          <Input placeholder="input placeholder" />
        </Form.Item>
        <Form.Item>
          <Button type="primary" htmlType="submit">
            Submit
          </Button>
        </Form.Item>
      </Form>
    </Space>
  )
}

## API

### Form

| Property                | Description                                                                     | Type                                                           | Default        |
| ----------------------- | ------------------------------------------------------------------------------- | -------------------------------------------------------------- | -------------- |
| `component`             | Root element. Use `false` to render no root DOM node.                           | `false \| keyof JSX.IntrinsicElements \| Component`            | `'form'`       |
| `form`                  | External form instance created by useForm or createFormInstance.                | `FormInstance`                                                 | `-`            |
| `name`                  | Form name used by Form.Provider callbacks.                                      | `string`                                                       | `-`            |
| `fields`                | Controlled field state.                                                         | `FieldData[]`                                                  | `-`            |
| `initialValues`         | Initial field values applied when the form is mounted.                          | `FormValues`                                                   | `-`            |
| `layout`                | Form layout mode.                                                               | `'horizontal' \| 'vertical' \| 'inline'`                       | `'horizontal'` |
| `requiredMark`          | Required mark display style for labels.                                         | `boolean \| 'optional' \| ((label, info) => JSX.Element)`      | `true`         |
| `validateTrigger`       | Default event name or names that trigger validation for child Form.Item fields. | `string \| string[]`                                           | item trigger   |
| `validateMessages`      | Validation message templates.                                                   | `ValidateMessages`                                             | `-`            |
| `disabled`              | Default disabled state for Form-integrated controls.                            | `boolean`                                                      | `false`        |
| `size`                  | Default size for Form-integrated controls.                                      | `'small' \| 'middle' \| 'large'`                               | config size    |
| `variant`               | Default variant for supported Form-integrated controls.                         | `'outlined' \| 'borderless' \| 'filled' \| 'underlined'`       | `'outlined'`   |
| `feedbackIcons`         | Feedback icons for Form.Item with `hasFeedback`.                                | `FeedbackIcons`                                                | built-in icons |
| `classNames / styles`   | Semantic class and style maps.                                                  | `{ root?: string } / { root?: JSX.CSSProperties }`             | `-`            |
| `scrollToFirstError`    | Scroll to the first failed field after submit.                                  | `boolean \| (ScrollIntoViewOptions & { focus?: boolean })`     | `false`        |
| `clearOnDestroy`        | Clear values when the last Form owner using the instance unmounts.              | `boolean`                                                      | `false`        |
| `preserve`              | Default value preservation hint for fields.                                     | `boolean`                                                      | `true`         |
| `labelCol / wrapperCol` | Common label and control layout metadata passed to form items.                  | `unknown`                                                      | `-`            |
| `labelAlign / colon`    | Common label alignment and colon display.                                       | `'left' \| 'right' / boolean`                                  | `right / true` |
| `labelWrap / tooltip`   | Common label wrapping and tooltip.                                              | `boolean / FormTooltip`                                        | `-`            |
| `onFinish`              | Called with all values after validation succeeds on submit.                     | `(values: FormValues) => void`                                 | `-`            |
| `onFinishFailed`        | Called with values and error fields after validation fails.                     | `(errorInfo: ValidateErrorInfo) => void`                       | `-`            |
| `onValuesChange`        | Called whenever a field updates its value.                                      | `(changedValues: FormValues, allValues: FormValues) => void`   | `-`            |
| `onFieldsChange`        | Called when field values or field state change.                                 | `(changedFields: FieldData[], allFields: FieldData[]) => void` | `-`            |

### Form.Item

| Property                | Description                                                                                   | Type                                                | Default                           |
| ----------------------- | --------------------------------------------------------------------------------------------- | --------------------------------------------------- | --------------------------------- |
| `label`                 | Label displayed before the field control. Use `null` to render an empty label slot.           | `JSX.Element`                                       | `-`                               |
| `htmlFor`               | `for` attribute on the label element.                                                         | `string`                                            | `-`                               |
| `name`                  | Field name used to read and write values in the form store.                                   | `NamePath`                                          | `-`                               |
| `rules`                 | Validation rules for this field.                                                              | `Rule[]`                                            | `-`                               |
| `required`              | Adds a required rule and displays the required mark.                                          | `boolean`                                           | `false`                           |
| `messageVariables`      | Variables used by validate message templates.                                                 | `Record<string, string>`                            | `-`                               |
| `help`                  | Custom help text shown below the control.                                                     | `JSX.Element`                                       | first error/warning               |
| `validateStatus`        | Manually sets validation visual state.                                                        | `'success' \| 'warning' \| 'error' \| 'validating'` | `-`                               |
| `hasFeedback`           | Show feedback icon for the current validation status.                                         | `boolean \| { icons?: FeedbackIcons }`              | `false`                           |
| `layout`                | Item-level layout override.                                                                   | `'horizontal' \| 'vertical' \| 'inline'`            | Form layout                       |
| `classNames / styles`   | Semantic slots: root, label, content, help, helpItem, extra, feedbackIcon.                    | `Partial<Record<FormItemSemanticSlot, ...>>`        | `-`                               |
| `valuePropName`         | Prop name used by child controls for their value.                                             | `string`                                            | `'value'`                         |
| `trigger`               | Child event name that updates the field value.                                                | `string`                                            | `'onChange'`                      |
| `validateTrigger`       | Event name or names that trigger validation.                                                  | `string \| string[]`                                | `Form validateTrigger or trigger` |
| `shouldUpdate`          | For unnamed render areas, re-render when form values change or when a predicate returns true. | `boolean \| ((prevValues, nextValues) => boolean)`  | `-`                               |
| `labelCol / wrapperCol` | Item-level label and control layout metadata.                                                 | `unknown`                                           | `-`                               |
| `labelAlign / colon`    | Override label alignment or colon display.                                                    | `'left' \| 'right' / boolean`                       | Form value                        |
| `labelWrap / tooltip`   | Override label wrapping or render extra label help.                                           | `boolean / FormTooltip`                             | Form value                        |
| `getValueFromEvent`     | Customize how a field value is extracted from control events.                                 | `(...args: unknown[]) => unknown`                   | `-`                               |
| `getValueProps`         | Customize props derived from the current field value.                                         | `(value: unknown) => Record<string, unknown>`       | `-`                               |
| `normalize`             | Normalize a value before storing it in the form.                                              | `(value, prevValue, allValues) => unknown`          | `-`                               |
| `dependencies`          | Revalidate this field when dependent fields update.                                           | `NamePath[]`                                        | `-`                               |
| `validateFirst`         | Stop validation on the first failing rule, or run rules in parallel.                          | `boolean \| 'parallel'`                             | `false`                           |
| `validateDebounce`      | Debounce triggered validation by the given milliseconds.                                      | `number`                                            | `-`                               |
| `preserve`              | Keep a field value when the field is unmounted.                                               | `boolean`                                           | `true`                            |
| `noStyle`               | Register a field without rendering the styled Form.Item wrapper.                              | `boolean`                                           | `false`                           |
| `hidden`                | Hide the field while still collecting and validating its value.                               | `boolean`                                           | `false`                           |
| `extra`                 | Extra prompt content shown below validation help.                                             | `JSX.Element`                                       | `-`                               |
| `initialValue`          | Initial value for this field.                                                                 | `unknown`                                           | `-`                               |

### FormInstance

| Property            | Description                                                                    | Type                                                                                       |
| ------------------- | ------------------------------------------------------------------------------ | ------------------------------------------------------------------------------------------ |
| `getFieldValue`     | Reads one field value.                                                         | `(name: NamePath) => unknown`                                                              |
| `setFieldValue`     | Sets one field value and clears its errors.                                    | `(name: NamePath, value: unknown) => void`                                                 |
| `getFieldsValue`    | Returns registered values, selected values, all values, or filtered snapshots. | `() / (true, filter?) / (NamePath[], filter?) / ({ strict, filter })`                      |
| `setFieldsValue`    | Merges multiple values into the form store.                                    | `(values: FormValues) => void`                                                             |
| `setFields`         | Sets field values or field state such as errors and touched.                   | `(fields: FieldData[]) => void`                                                            |
| `resetFields`       | Resets all fields or the provided field names to their initial values.         | `(names?: NamePath[]) => void`                                                             |
| `validateFields`    | Validates all fields or selected fields and resolves with values.              | `(names?: NamePath[], config?: ValidateConfig) => Promise<FormValues>`                     |
| `getFieldError`     | Returns current errors for one field.                                          | `(name: NamePath) => string[]`                                                             |
| `getFieldsError`    | Returns error and warning state.                                               | `(names?: NamePath[]) => FieldError[]`                                                     |
| `getFieldWarnings`  | Returns current warnings for one field.                                        | `(name: NamePath) => string[]`                                                             |
| `scrollToField`     | Scrolls a registered field element into view and can focus its control.        | `(name: NamePath, options?: ScrollIntoViewOptions \| { focus?: boolean }) => void`         |
| `getFieldInstance`  | Returns the registered field DOM element when available.                       | `(name: NamePath) => unknown`                                                              |
| `isFieldTouched`    | Checks whether a field has been touched.                                       | `(name: NamePath) => boolean`                                                              |
| `isFieldsTouched`   | Checks whether any or all fields have been touched.                            | `(allTouched?: boolean) => boolean; (names?: NamePath[], allTouched?: boolean) => boolean` |
| `isFieldValidating` | Checks whether a field is validating.                                          | `(name: NamePath) => boolean`                                                              |
| `submit`            | Runs validation and calls onFinish or onFinishFailed.                          | `() => void`                                                                               |

### Rule

| Property          | Description                                                                                                                                    | Type                                                                                     | Default |
| ----------------- | ---------------------------------------------------------------------------------------------------------------------------------------------- | ---------------------------------------------------------------------------------------- | ------- |
| `required`        | Requires a non-empty value.                                                                                                                    | `boolean`                                                                                | `-`     |
| `type`            | Checks the runtime value type.                                                                                                                 | `'string' \| 'number' \| 'boolean' \| 'array' \| 'object' \| 'email' \| 'url' \| 'enum'` | `-`     |
| `min / max / len` | Validates string length, array length, or numeric value depending on the field.                                                                | `number`                                                                                 | `-`     |
| `pattern`         | Requires the value to match a regular expression.                                                                                              | `RegExp`                                                                                 | `-`     |
| `message`         | Error message shown when the rule fails.                                                                                                       | `string \| JSX.Element`                                                                  | `-`     |
| `fields`          | Object or array child rules keyed by property name or numeric index string.                                                                    | `Record<string, Rule>`                                                                   | `-`     |
| `defaultField`    | Rule applied to each item when validating an array value.                                                                                      | `Rule`                                                                                   | `-`     |
| `validator`       | Custom validator. Return a string, throw, or reject to report an error. Set validator.legacy = true to use the old (value, values) call shape. | `(rule, value, values?) => string \| void \| Promise<void>`                              | `-`     |
| `warningOnly`     | Store a failed rule result as warning instead of error.                                                                                        | `boolean`                                                                                | `false` |
| `validateTrigger` | Override the validation trigger for this rule.                                                                                                 | `string \| string[]`                                                                     | `-`     |
| `validateFirst`   | Stop validation on first failure or validate in parallel.                                                                                      | `boolean \| 'parallel'`                                                                  | `false` |
| `transform`       | Transform the value before this rule validates it.                                                                                             | `(value) => unknown`                                                                     | `-`     |
