# Flex

Flex arranges content with CSS flexbox and Ant Design style spacing, direction, wrapping, and alignment props.

### Basic

:::preview

  const Demo1 = function () {
      const [orientation, setOrientation] = createSignal('horizontal');

import { createSignal } from 'solid-js'

        ]}
      />

        ))}
      </Flex>
    </Flex>
  )
}

export default Demo1

import { createSignal } from 'solid-js'
import { Button, Flex, Radio, Slider } from '@solid-ant-design/core'

const Demo2 = function () {
  const [gapSize, setGapSize] = createSignal<FlexProps['gap']>('small')
  const [customGapSize, setCustomGapSize] = createSignal(0)

  return (
    <Flex gap="medium" vertical>
      <Radio.Group
        value={String(gapSize())}
        onChange={(value) => setGapSize(value as FlexProps['gap'])}
        options={['small', 'medium', 'large', 'customize'].map((size) => ({
          label: size,
          value: size,
        }))}
      />
      {gapSize() === 'customize' && (
        <Slider value={customGapSize()} onChange={(value) => setCustomGapSize(value as number)} />
      )}
      <Flex gap={gapSize() !== 'customize' ? gapSize() : customGapSize()}>
        <Button type="primary">Primary</Button>
        <Button>Default</Button>
        <Button type="dashed">Dashed</Button>
        <Button type="link">Link</Button>
      </Flex>
    </Flex>
  )
}

export default Demo2

## API

| Property        | Description                                                               | Type                                                                                                                                                              | Default        |
| --------------- | ------------------------------------------------------------------------- | ----------------------------------------------------------------------------------------------------------------------------------------------------------------- | -------------- |
| `align`         | Cross-axis alignment. Maps to CSS `align-items`.                          | `'center' \| 'start' \| 'end' \| 'flex-start' \| 'flex-end' \| 'self-start' \| 'self-end' \| 'baseline' \| 'normal' \| 'stretch'`                                 | `flex-start`   |
| `component`     | Wrapper element or Solid component.                                       | `ValidComponent`                                                                                                                                                  | `'div'`        |
| `flex`          | CSS `flex` shorthand applied to the root element.                         | `JSX.CSSProperties['flex']`                                                                                                                                       | `-`            |
| `gap`           | Gap between items. Presets use Flex component tokens.                     | `'small' \| 'middle' \| 'medium' \| 'large' \| number \| string \| [number, number]`                                                                              | `'medium'`     |
| `justify`       | Main-axis alignment. Maps to CSS `justify-content`.                       | `'flex-start' \| 'flex-end' \| 'start' \| 'end' \| 'center' \| 'space-between' \| 'space-around' \| 'space-evenly' \| 'stretch' \| 'normal' \| 'left' \| 'right'` | `-`            |
| `orientation`   | Direction of the flex layout. Takes priority over `vertical`.             | `'horizontal' \| 'vertical'`                                                                                                                                      | `'horizontal'` |
| `prefixCls`     | Custom CSS class prefix.                                                  | `string`                                                                                                                                                          | `-`            |
| `rootClassName` | Extra root class. Solid consumers should continue using `class` normally. | `string`                                                                                                                                                          | `-`            |
| `vertical`      | Shorthand for vertical orientation.                                       | `boolean`                                                                                                                                                         | `false`        |
| `wrap`          | Controls flex wrapping. Boolean true maps to `wrap`.                      | `boolean \| 'nowrap' \| 'wrap' \| 'wrap-reverse'`                                                                                                                 | `-`            |

## Design Token

| Token       | Description | Default alias |
| ----------- | ----------- | ------------- |
| `flexGapSM` | Small gap.  | `paddingXS`   |
| `flexGap`   | Middle gap. | `padding`     |
| `flexGapLG` | Large gap.  | `paddingLG`   |
