### Controlled

:::preview

<SliderDemo002 />

export const SliderDemo002 = (() => {
  const Demo2 = function () {
      const [value, setValue] = createSignal(30);
      return (<Space direction="vertical" class="w-80">
        <Slider value={value()} onChange={(next) => {
              if (typeof next === 'number')
                  setValue(next);
          }}/>
        <span>Current value: {value()}</span>
      </Space>);
  };

  return Demo2
})()

import { createSignal } from 'solid-js'

const Demo2 = function () {

      <Slider
        value={value()}
        onChange={(next) => {
          if (typeof next === 'number') setValue(next)
        }}
      />
      <span>Current value: {value()}</span>
    </Space>
  )
}

export default Demo2

## API

### Slider

| Property           | Description                                                               | Type                                                               | Default      |
| ------------------ | ------------------------------------------------------------------------- | ------------------------------------------------------------------ | ------------ |
| `value`            | Controlled slider value. Array form is used for range handles.            | `number \| number[]`                                               | `-`          |
| `defaultValue`     | Initial value for uncontrolled usage.                                     | `number \| number[]`                                               | `0 \| [0,0]` |
| `min`              | Minimum value.                                                            | `number`                                                           | `0`          |
| `max`              | Maximum value.                                                            | `number`                                                           | `100`        |
| `step`             | Value increment. Use `null` with marks to only allow mark/min/max values. | `number \| null`                                                   | `1`          |
| `disabled`         | Disables user interaction.                                                | `boolean`                                                          | `false`      |
| `keyboard`         | Enables keyboard handle movement.                                         | `boolean`                                                          | `true`       |
| `dots`             | Restricts movement to step or mark points and renders dots.               | `boolean`                                                          | `false`      |
| `included`         | Whether the selected track is filled.                                     | `boolean`                                                          | `true`       |
| `range`            | Enables range selection and optional range behavior.                      | `boolean \| SliderRangeConfig`                                     | `false`      |
| `marks`            | Tick mark labels keyed by numeric value.                                  | `Record<number, SliderMark>`                                       | `-`          |
| `orientation`      | Slider orientation. Takes priority over `vertical`.                       | `'horizontal' \| 'vertical'`                                       | `horizontal` |
| `vertical`         | Legacy shorthand for vertical orientation.                                | `boolean`                                                          | `false`      |
| `reverse`          | Reverses value direction.                                                 | `boolean`                                                          | `false`      |
| `tooltip`          | Tooltip configuration.                                                    | `SliderTooltipConfig`                                              | `-`          |
| `classNames`       | Semantic slot class names. Use `class` for the root element class.        | `SliderSemanticClassNames \| ((info) => SliderSemanticClassNames)` | `-`          |
| `styles`           | Semantic slot inline styles.                                              | `SliderSemanticStyles \| ((info) => SliderSemanticStyles)`         | `-`          |
| `ref`              | Imperative ref exposing `focus`, `blur`, and `nativeElement`.             | `SliderRef \| ((ref) => void)`                                     | `-`          |
| `tooltipVisible`   | Deprecated. Use `tooltip.open`.                                           | `boolean`                                                          | `-`          |
| `onChange`         | Called while value changes.                                               | `(value: SliderValue) => void`                                     | `-`          |
| `onChangeComplete` | Called after pointer or keyboard commit.                                  | `(value: SliderValue) => void`                                     | `-`          |
| `onAfterChange`    | Deprecated. Use `onChangeComplete`.                                       | `(value: SliderValue) => void`                                     | `-`          |

### SliderRangeConfig

| Property         | Description                                                                              | Type      | Default |
| ---------------- | ---------------------------------------------------------------------------------------- | --------- | ------- |
| `draggableTrack` | Allows dragging the selected range track.                                                | `boolean` | `false` |
| `editable`       | Allows adding handles by clicking the rail and removing handles by double-clicking them. | `boolean` | `false` |
| `minCount`       | Minimum handle count in editable mode.                                                   | `number`  | `0`     |
| `maxCount`       | Maximum handle count in editable mode.                                                   | `number`  | `-`     |

### SliderTooltipConfig

| Property             | Description                                       | Type                                        | Default         |
| -------------------- | ------------------------------------------------- | ------------------------------------------- | --------------- |
| `autoAdjustOverflow` | Reserved for antd-compatible tooltip placement.   | `boolean`                                   | `true`          |
| `open`               | Forces tooltip visibility on or off.              | `boolean`                                   | `-`             |
| `placement`          | Tooltip placement.                                | `'top' \| 'bottom' \| 'left' \| 'right'`    | `top`           |
| `getPopupContainer`  | Reserved for portal-based tooltip containers.     | `(triggerNode: HTMLElement) => HTMLElement` | `document.body` |
| `formatter`          | Formats tooltip content. Return `null` to hide.   | `(value: number) => JSX.Element \| null`    | identity        |
| `class`              | Tooltip class. Solid alias for React `className`. | `string`                                    | `-`             |
| `style`              | Tooltip inline style.                             | `JSX.CSSProperties`                         | `-`             |

### Semantic slots

`classNames` and `styles` support: `root`, `rail`, `tracks`, `track`, `handle`, `step`, `dot`, `mark`, `markText`, `tooltip`.

### SliderMark

| Property      | Description                             | Type                | Default |
| ------------- | --------------------------------------- | ------------------- | ------- |
| `JSX.Element` | Simple mark label.                      | `SliderMark`        | `-`     |
| `label`       | Mark label content for object marks.    | `JSX.Element`       | `-`     |
| `style`       | Inline style applied to the mark label. | `JSX.CSSProperties` | `-`     |
