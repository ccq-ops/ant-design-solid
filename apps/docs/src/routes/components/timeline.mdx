import { Timeline } from '@ant-design-solid/core'

# Timeline

Display chronological events with titles, content, placement, loading nodes, and semantic styling.

### Basic

:::preview

<TimelineDemo001 />

export const TimelineDemo001 = (() => {
  const items = [
      { title: '2026-06-01', content: 'Create project', color: 'green' },
      { title: '2026-06-02', content: 'Add core components' },
      { title: '2026-06-03', content: 'Polish documentation', color: 'red' },
  ];
  const Demo1 = function () {
      return <Timeline items={items}/>;
  };

  return Demo1
})()

## API

### Timeline

| Property        | Description                                        | Type                                                      | Default      |
| --------------- | -------------------------------------------------- | --------------------------------------------------------- | ------------ |
| `items`         | Timeline item data.                                | `TimelineItem[]`                                          | `-`          |
| `mode`          | Timeline item placement mode.                      | `'start' \| 'alternate' \| 'end'`                         | `'start'`    |
| `orientation`   | Timeline direction.                                | `'vertical' \| 'horizontal'`                              | `'vertical'` |
| `variant`       | Dot visual variant.                                | `'outlined' \| 'filled'`                                  | `'outlined'` |
| `titleSpan`     | Title area width or spacing before the dot center. | `number \| string`                                        | `12`         |
| `reverse`       | Reverses the item order.                           | `boolean`                                                 | `false`      |
| `class`         | Additional root class name.                        | `string`                                                  | `-`          |
| `className`     | Additional root class name alias.                  | `string`                                                  | `-`          |
| `rootClassName` | Additional root class name.                        | `string`                                                  | `-`          |
| `style`         | Root inline style.                                 | `JSX.CSSProperties`                                       | `-`          |
| `classNames`    | Semantic DOM class names.                          | `Partial<Record<TimelineSemanticKey, string>>`            | `-`          |
| `styles`        | Semantic DOM inline styles.                        | `Partial<Record<TimelineSemanticKey, JSX.CSSProperties>>` | `-`          |
| `prefixCls`     | Custom CSS class prefix.                           | `string`                                                  | `-`          |

### TimelineItem

| Property     | Description                                            | Type                                                 | Default  |
| ------------ | ------------------------------------------------------ | ---------------------------------------------------- | -------- |
| `key`        | Stable item key.                                       | `string \| number`                                   | `-`      |
| `title`      | Title content rendered beside the timeline item.       | `JSX.Element`                                        | `-`      |
| `content`    | Main timeline item content.                            | `JSX.Element`                                        | `-`      |
| `color`      | Preset semantic color or custom CSS color for the dot. | `'blue' \| 'red' \| 'green' \| 'gray' \| string`     | `'blue'` |
| `icon`       | Custom node icon.                                      | `JSX.Element`                                        | `-`      |
| `placement`  | Overrides item placement for this item.                | `'start' \| 'end'`                                   | `-`      |
| `loading`    | Shows the item in loading state.                       | `boolean`                                            | `false`  |
| `class`      | Additional item class name.                            | `string`                                             | `-`      |
| `className`  | Additional item class name.                            | `string`                                             | `-`      |
| `style`      | Additional item inline style.                          | `JSX.CSSProperties`                                  | `-`      |
| `classNames` | Item semantic DOM class names.                         | `Omit<TimelineSemanticClassNames, 'root' \| 'list'>` | `-`      |
| `styles`     | Item semantic DOM inline styles.                       | `Omit<TimelineSemanticStyles, 'root' \| 'list'>`     | `-`      |

### Semantic DOM

| Key           | Description                        | Component-level | Item-level |
| ------------- | ---------------------------------- | --------------- | ---------- |
| `root`        | Timeline root ordered list.        | Yes             | No         |
| `list`        | Reserved list slot.                | Yes             | No         |
| `item`        | Timeline item.                     | Yes             | Yes        |
| `itemTitle`   | Item title area.                   | Yes             | Yes        |
| `itemIcon`    | Item dot or custom icon.           | Yes             | Yes        |
| `itemContent` | Main item content area.            | Yes             | Yes        |
| `itemRail`    | Connecting rail between items.     | Yes             | Yes        |
| `itemWrapper` | Internal item wrapper.             | Yes             | Yes        |
| `itemSection` | Internal section around item body. | Yes             | Yes        |
| `itemHeader`  | Internal header around title/icon. | Yes             | Yes        |
