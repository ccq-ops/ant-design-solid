# Upload

Select files, upload them with a default request or custom request, intercept files before upload, and render a configurable upload list.

### Manual upload

:::preview

<UploadDemo003 />

export const UploadDemo003 = (() => {
  const Demo3 = function () {
      const [files, setFiles] = createSignal([]);
      return (<Space direction="vertical">
        <Upload fileList={files()} beforeUpload={() => false} onChange={({ fileList }) => setFiles(fileList)}>
          <Button>Select without uploading</Button>
        </Upload>
        <span>
          {files()
              .map((file) => file.name)
              .join(', ') || 'No files selected'}
        </span>
      </Space>);
  };

return Demo3
})()

---

```tsx
import { createSignal } from 'solid-js'
import { Button, Space, Upload } from '@solid-ant-design/core'
import type { UploadFile } from '@solid-ant-design/core'

const Demo3 = function () {
  const [files, setFiles] = createSignal<UploadFile[]>([])

  return (
    <Space direction="vertical">
      <Upload
        fileList={files()}
        beforeUpload={() => false}
        onChange={({ fileList }) => setFiles(fileList)}
      >
        <Button>Select without uploading</Button>
      </Upload>
      <span>
        {files()
          .map((file) => file.name)
          .join(', ') || 'No files selected'}
      </span>
    </Space>
  )
}

export default Demo3
```

export const UploadDemo009 = (() => {

      return (<Upload defaultFileList={[{ uid: '1', name: 'report.pdf', status: 'done', url: '/report.pdf' }]} showUploadList={{

              showDownloadIcon: true,

## API

### Upload

| Property                | Description                                                                                                                                                        | Type                                                                                                           | Default               |
| ----------------------- | ------------------------------------------------------------------------------------------------------------------------------------------------------------------ | -------------------------------------------------------------------------------------------------------------- | --------------------- |
| `accept`                | File types accepted by the native input, or an object with an additional Solid-side filter.                                                                        | `string \| { format: string; filter?: 'native' \| ((file: File) => boolean) }`                                 | `-`                   |
| `action`                | Upload URL or async URL resolver for the default request.                                                                                                          | `string \| (file: File) => string \| PromiseLike<string>`                                                      | `-`                   |
| `beforeUpload`          | Runs before upload. Return `false` to keep the file without uploading, `Upload.LIST_IGNORE` to skip it, or a `File`/`Blob`/`string` to replace the upload payload. | `(file: File, fileList: File[]) => void \| boolean \| string \| Blob \| File \| Promise<...>`                  | `-`                   |
| `capture`               | Native file input capture attribute.                                                                                                                               | `boolean \| 'user' \| 'environment'`                                                                           | `-`                   |
| `children`              | Upload trigger content.                                                                                                                                            | `JSX.Element`                                                                                                  | `-`                   |
| `class`                 | Root class. Solid replacement for React `className`.                                                                                                               | `string`                                                                                                       | `-`                   |
| `classNames`            | Semantic class names.                                                                                                                                              | `Partial<Record<'root' \| 'trigger' \| 'list' \| 'item', string>>`                                             | `-`                   |
| `customRequest`         | Override upload request behavior. Receives complete request options and `{ defaultRequest }`.                                                                      | `(options, info) => void`                                                                                      | `-`                   |
| `data`                  | Extra request params or async params resolver.                                                                                                                     | `Record<string, unknown> \| (file: UploadFile) => Record<string, unknown> \| Promise<Record<string, unknown>>` | `-`                   |
| `defaultFileList`       | Initial upload file list for uncontrolled usage.                                                                                                                   | `UploadFile[]`                                                                                                 | `-`                   |
| `directory`             | Enables folder selection through native directory input attributes.                                                                                                | `boolean`                                                                                                      | `false`               |
| `disabled`              | Disables file selection and file actions.                                                                                                                          | `boolean`                                                                                                      | `false`               |
| `fileList`              | Controlled upload file list.                                                                                                                                       | `UploadFile[]`                                                                                                 | `-`                   |
| `headers`               | Request headers for default request and custom request options.                                                                                                    | `Record<string, string>`                                                                                       | `-`                   |
| `iconRender`            | Custom file icon renderer.                                                                                                                                         | `(file: UploadFile, listType?: UploadListType) => JSX.Element`                                                 | `-`                   |
| `isImageUrl`            | Custom image detection for thumbnails.                                                                                                                             | `(file: UploadFile) => boolean`                                                                                | built-in image check  |
| `itemRender`            | Custom upload list item renderer.                                                                                                                                  | `(originNode, file, fileList, actions) => JSX.Element`                                                         | `-`                   |
| `listType`              | Upload list style.                                                                                                                                                 | `'text' \| 'picture' \| 'picture-card' \| 'picture-circle'`                                                    | `'text'`              |
| `maxCount`              | Maximum number of files kept in the list.                                                                                                                          | `number`                                                                                                       | `-`                   |
| `method`                | HTTP method for default request.                                                                                                                                   | `'POST' \| 'PUT' \| 'PATCH' \| 'post' \| 'put' \| 'patch'`                                                     | `'post'`              |
| `multiple`              | Allows selecting multiple files.                                                                                                                                   | `boolean`                                                                                                      | `false`               |
| `name`                  | Form field name used by the default request.                                                                                                                       | `string`                                                                                                       | `'file'`              |
| `openFileDialogOnClick` | Whether clicking the trigger opens the file dialog.                                                                                                                | `boolean`                                                                                                      | `true`                |
| `pastable`              | Enables paste-to-upload on the root element.                                                                                                                       | `boolean`                                                                                                      | `false`               |
| `previewFile`           | Custom preview data URL generator.                                                                                                                                 | `(file: File \| Blob) => PromiseLike<string>`                                                                  | `-`                   |
| `progress`              | Progress bar options.                                                                                                                                              | `{ strokeWidth?: number; showInfo?: boolean; strokeColor?: string }`                                           | `-`                   |
| `showUploadList`        | Shows the default list, or customizes list actions and extra content.                                                                                              | `boolean \| ShowUploadListConfig`                                                                              | `true`                |
| `styles`                | Semantic inline styles.                                                                                                                                            | `Partial<Record<'root' \| 'trigger' \| 'list' \| 'item', JSX.CSSProperties>>`                                  | `-`                   |
| `type`                  | Upload interaction type.                                                                                                                                           | `'select' \| 'drag'`                                                                                           | `'select'`            |
| `withCredentials`       | Sends cookies with default XMLHttpRequest upload.                                                                                                                  | `boolean`                                                                                                      | `false`               |
| `onChange`              | Called whenever upload state changes.                                                                                                                              | `(info: UploadChangeInfo) => void`                                                                             | `-`                   |
| `onDownload`            | Called when a download action is clicked.                                                                                                                          | `(file: UploadFile) => void`                                                                                   | open `file.url`       |
| `onDrop`                | Called when files are dropped on the upload area.                                                                                                                  | `(event: DragEvent) => void`                                                                                   | `-`                   |
| `onPreview`             | Called when a preview action or preview link is clicked.                                                                                                           | `(file: UploadFile) => void`                                                                                   | `-`                   |
| `onRemove`              | Called before removing a file. Return `false` or reject to keep it.                                                                                                | `(file: UploadFile) => void \| boolean \| Promise<void \| boolean>`                                            | `-`                   |
| `prefixCls`             | Custom CSS class prefix.                                                                                                                                           | `string`                                                                                                       | ConfigProvider prefix |

### Upload.Dragger

`Upload.Dragger` accepts the same props as `Upload` and sets `type="drag"` by default.

### ShowUploadListConfig

| Property           | Description                           | Type                                               |
| ------------------ | ------------------------------------- | -------------------------------------------------- |
| `extra`            | Extra content rendered for each file. | `JSX.Element \| (file: UploadFile) => JSX.Element` |
| `showPreviewIcon`  | Whether to show the preview action.   | `boolean \| (file: UploadFile) => boolean`         |
| `showRemoveIcon`   | Whether to show the remove action.    | `boolean \| (file: UploadFile) => boolean`         |
| `showDownloadIcon` | Whether to show the download action.  | `boolean \| (file: UploadFile) => boolean`         |
| `previewIcon`      | Custom preview icon.                  | `JSX.Element \| (file: UploadFile) => JSX.Element` |
| `removeIcon`       | Custom remove icon.                   | `JSX.Element \| (file: UploadFile) => JSX.Element` |
| `downloadIcon`     | Custom download icon.                 | `JSX.Element \| (file: UploadFile) => JSX.Element` |

### UploadFile

| Property           | Description                                   | Type                                                       |
| ------------------ | --------------------------------------------- | ---------------------------------------------------------- |
| `uid`              | Unique file id.                               | `string`                                                   |
| `name`             | File name.                                    | `string`                                                   |
| `size`             | File size.                                    | `number`                                                   |
| `fileName`         | Alternate file name.                          | `string`                                                   |
| `lastModified`     | Last modified timestamp.                      | `number`                                                   |
| `lastModifiedDate` | Last modified date.                           | `Date`                                                     |
| `status`           | Upload status.                                | `'ready' \| 'uploading' \| 'done' \| 'error' \| 'removed'` |
| `percent`          | Upload progress percent.                      | `number`                                                   |
| `originFileObj`    | Original native File object.                  | `File`                                                     |
| `url`              | Download or preview URL.                      | `string`                                                   |
| `thumbUrl`         | Thumbnail URL.                                | `string`                                                   |
| `crossOrigin`      | Thumbnail CORS attribute.                     | `'' \| 'anonymous' \| 'use-credentials'`                   |
| `response`         | Upload response value.                        | `unknown`                                                  |
| `error`            | Upload error value.                           | `unknown`                                                  |
| `linkProps`        | Extra link props or serialized link props.    | `JSX.AnchorHTMLAttributes<HTMLAnchorElement> \| string`    |
| `type`             | MIME type.                                    | `string`                                                   |
| `xhr`              | XMLHttpRequest instance from default request. | `XMLHttpRequest`                                           |
| `preview`          | Generated preview URL.                        | `string`                                                   |

### UploadChangeInfo

| Property   | Description                    | Type                  |
| ---------- | ------------------------------ | --------------------- |
| `file`     | Current changed file.          | `UploadFile`          |
| `fileList` | Current file list.             | `UploadFile[]`        |
| `event`    | Progress event when available. | `{ percent: number }` |

### UploadRequestOptions

| Property          | Description                                                            | Type                                                                 |
| ----------------- | ---------------------------------------------------------------------- | -------------------------------------------------------------------- |
| `action`          | Upload URL.                                                            | `string`                                                             |
| `data`            | Upload extra params.                                                   | `Record<string, unknown>`                                            |
| `filename`        | Form field name.                                                       | `string`                                                             |
| `file`            | Native File being uploaded.                                            | `File`                                                               |
| `withCredentials` | Whether to send credentials.                                           | `boolean`                                                            |
| `headers`         | Request headers.                                                       | `Record<string, string>`                                             |
| `method`          | Request method.                                                        | `UploadMethod`                                                       |
| `onProgress`      | Report upload progress. Accepts v6-style event and a number shorthand. | `(event: { percent: number } \| number, file?: UploadFile) => void`  |
| `onSuccess`       | Mark upload as successful.                                             | `(body?: unknown, fileOrXhr?: UploadFile \| XMLHttpRequest) => void` |
| `onError`         | Mark upload as failed.                                                 | `(event: unknown, body?: unknown) => void`                           |
