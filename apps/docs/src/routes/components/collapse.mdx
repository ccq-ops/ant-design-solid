import { Collapse, Space } from '@ant-design-solid/core'

# Collapse

Collapse displays content areas that can be expanded or collapsed.

### Basic

:::preview

<CollapseDemo001 />

export const CollapseDemo001 = (() => {
  const text = `A dog is a type of domesticated animal. Known for its loyalty and faithfulness, it can be found as a welcome guest in many households across the world.`;
  const items = [
      { key: '1', label: 'This is panel header 1', children: <p>{text}</p> },
      { key: '2', label: 'This is panel header 2', children: <p>{text}</p> },
      { key: '3', label: 'This is panel header 3', children: <p>{text}</p> },
  ];
  const Demo1 = function () {
      return <Collapse defaultActiveKey={['1']} items={items}/>;
  };

return Demo1
})()

## API

### Collapse

| Property               | Description                                                | Type                                                                                                                          | Default   |
| ---------------------- | ---------------------------------------------------------- | ----------------------------------------------------------------------------------------------------------------------------- | --------- |
| `items`                | Collapse panel items.                                      | `CollapseItem[]`                                                                                                              | `-`       |
| `activeKey`            | Controlled active panel key or keys.                       | `string \| number \| (string \| number)[]`                                                                                    | `-`       |
| `defaultActiveKey`     | Initial active panel key or keys.                          | `string \| number \| (string \| number)[]`                                                                                    | `-`       |
| `accordion`            | Allows only one panel to be open at a time.                | `boolean`                                                                                                                     | `false`   |
| `bordered`             | Whether to render borders.                                 | `boolean`                                                                                                                     | `true`    |
| `class`                | Additional root class.                                     | `string`                                                                                                                      | `-`       |
| `rootClass`            | Additional root class applied after semantic classes.      | `string`                                                                                                                      | `-`       |
| `classNames`           | Semantic structure class names.                            | `Record<'root' \| 'header' \| 'title' \| 'body' \| 'icon', string> \| (info) => Record<string, string>`                       | `-`       |
| `styles`               | Semantic structure inline styles.                          | `Record<'root' \| 'header' \| 'title' \| 'body' \| 'icon', JSX.CSSProperties> \| (info) => Record<string, JSX.CSSProperties>` | `-`       |
| `collapsible`          | Controls which part of panels toggles expansion.           | `'header' \| 'icon' \| 'disabled'`                                                                                            | `-`       |
| `destroyOnHidden`      | Destroy inactive panel content after it is hidden.         | `boolean`                                                                                                                     | `false`   |
| `destroyInactivePanel` | Deprecated. Use `destroyOnHidden` instead.                 | `boolean`                                                                                                                     | `false`   |
| `expandIcon`           | Custom expand icon renderer.                               | `(panelProps: CollapseExpandIconProps) => JSX.Element`                                                                        | `-`       |
| `expandIconPlacement`  | Expand icon placement.                                     | `'start' \| 'end'`                                                                                                            | `'start'` |
| `expandIconPosition`   | Deprecated. Use `expandIconPlacement` instead.             | `'start' \| 'end'`                                                                                                            | `-`       |
| `ghost`                | Makes the collapse borderless with transparent background. | `boolean`                                                                                                                     | `false`   |
| `size`                 | Collapse size.                                             | `'large' \| 'medium' \| 'middle' \| 'small'`                                                                                  | `medium`  |
| `onChange`             | Called with active key or keys after expansion changes.    | `(activeKey: CollapseActiveKey \| undefined) => void`                                                                         | `-`       |

### CollapseItem

| Property      | Description                               | Type                                            | Default |
| ------------- | ----------------------------------------- | ----------------------------------------------- | ------- |
| `key`         | Unique panel key.                         | `string \| number`                              | `-`     |
| `label`       | Panel header content.                     | `JSX.Element`                                   | `-`     |
| `children`    | Panel body content.                       | `JSX.Element`                                   | `-`     |
| `extra`       | Extra header content.                     | `JSX.Element`                                   | `-`     |
| `disabled`    | Disables the panel.                       | `boolean`                                       | `false` |
| `collapsible` | Panel-level collapsible behavior.         | `'header' \| 'icon' \| 'disabled'`              | `-`     |
| `forceRender` | Force render panel body before expansion. | `boolean`                                       | `false` |
| `showArrow`   | Whether to show the expand icon.          | `boolean`                                       | `true`  |
| `classNames`  | Panel semantic class names.               | `Record<'header' \| 'body', string>`            | `-`     |
| `styles`      | Panel semantic inline styles.             | `Record<'header' \| 'body', JSX.CSSProperties>` | `-`     |
| `class`       | Additional panel class.                   | `string`                                        | `-`     |
| `style`       | Panel inline style.                       | `JSX.CSSProperties`                             | `-`     |

### Design Token

| Token             | Description           | Default                  |
| ----------------- | --------------------- | ------------------------ |
| `headerBg`        | Header background.    | `transparent`            |
| `headerPadding`   | Header padding.       | `8px 16px`               |
| `headerPaddingSM` | Small header padding. | `4px 8px`                |
| `contentPadding`  | Content box padding.  | `16px`                   |
| `contentBg`       | Content background.   | `token.colorBgContainer` |
