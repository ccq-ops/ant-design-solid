<TransferDemo006 />

export const TransferDemo006 = (() => {
  const dataSource = [
      { key: '1', title: 'Alpha', description: 'First item' },
      { key: '2', title: 'Beta', description: 'Second item' },
      { key: '3', title: 'Gamma', description: 'Third item' },
  ];
  const Demo6 = function () {
      return (<Transfer dataSource={dataSource} footer={(_, info) => <small>{info?.direction} list footer</small>}>
        {(list) => (<div class="flex flex-col gap-1 p-2">
            <For each={list.filteredItems}>
              {(item) => {
                  const selected = () => list.selectedKeys.includes(item.key);
                  return (<button type="button" aria-pressed={selected()} class="flex items-center justify-between rounded border px-2 py-1 text-left transition-colors" classList={{
                          'border-blue-500 bg-blue-50 text-blue-700': selected(),
                          'border-[var(--docs-border)] bg-[var(--docs-surface-solid)] text-[var(--docs-text)] hover:border-blue-300': !selected(),
                      }} onClick={() => list.onItemSelect(item.key, !selected())}>
                    <span>{item.title}</span>
                    <span class="text-xs">{selected() ? 'Selected' : 'Select'}</span>
                  </button>);
              }}
            </For>
          </div>)}
      </Transfer>);
  };

return Demo6
})()

        <div class="flex flex-col gap-1 p-2">

            {(item) => {
              const selected = () => list.selectedKeys.includes(item.key)

              return (
                <button
                  type="button"
                  aria-pressed={selected()}
                  class="flex items-center justify-between rounded border px-2 py-1 text-left transition-colors"
                  classList={{
                    'border-blue-500 bg-blue-50 text-blue-700': selected(),
                    'border-[var(--docs-border)] bg-[var(--docs-surface-solid)] text-[var(--docs-text)] hover:border-blue-300':
                      !selected(),
                  }}
                  onClick={() => list.onItemSelect(item.key, !selected())}
                >
                  <span>{item.title}</span>
                  <span class="text-xs">{selected() ? 'Selected' : 'Select'}</span>
                </button>
              )
            }}
          </For>
        </div>

## API

### Transfer

Solid uses `class` instead of React's `className`. Ant Design's `rootClassName` is intentionally not exposed.

| Property              | Description                                             | Type                                                                                                     | Default                |
| --------------------- | ------------------------------------------------------- | -------------------------------------------------------------------------------------------------------- | ---------------------- |
| `actions`             | Labels or nodes for move-to-target and move-to-source.  | `JSX.Element[]`                                                                                          | `-`                    |
| `children`            | Custom list body render function.                       | `(props: TransferCustomListBodyProps<T>) => JSX.Element`                                                 | `-`                    |
| `class`               | Root class name.                                        | `string`                                                                                                 | `-`                    |
| `classNames`          | Semantic DOM class names.                               | `TransferSemanticClassNames`                                                                             | `-`                    |
| `dataSource`          | Items that can be moved between lists.                  | `T[]`                                                                                                    | `[]`                   |
| `defaultSelectedKeys` | Initial selected item keys for uncontrolled usage.      | `TransferKey[]`                                                                                          | `[]`                   |
| `defaultTargetKeys`   | Initial keys in the target list for uncontrolled usage. | `TransferKey[]`                                                                                          | `[]`                   |
| `disabled`            | Disables selection and transfer operations.             | `boolean`                                                                                                | `false`                |
| `filterOption`        | Custom item filter used by search.                      | `(inputValue: string, item: T, direction: TransferDirection) => boolean`                                 | `-`                    |
| `footer`              | Custom footer for each list.                            | `(props: TransferListProps<T>, info?: { direction }) => JSX.Element`                                     | `-`                    |
| `listStyle`           | Style applied to each list section.                     | `JSX.CSSProperties \| ({ direction }) => JSX.CSSProperties`                                              | `-`                    |
| `locale`              | Localized labels and empty/search text.                 | `Partial<TransferLocale>`                                                                                | `-`                    |
| `oneWay`              | Hides the move-left operation.                          | `boolean`                                                                                                | `false`                |
| `operationStyle`      | Style applied to operation buttons and area.            | `JSX.CSSProperties`                                                                                      | `-`                    |
| `operations`          | Deprecated alias for `actions`.                         | `JSX.Element[]`                                                                                          | `-`                    |
| `pagination`          | Limits visible items per list.                          | `boolean \| { pageSize?: number; simple?: boolean; showSizeChanger?: boolean; showLessItems?: boolean }` | `false`                |
| `prefixCls`           | Custom CSS class prefix.                                | `string`                                                                                                 | `-`                    |
| `render`              | Custom item renderer.                                   | `(item: T) => JSX.Element \| { label: JSX.Element; value: string } \| string \| number \| null`          | `-`                    |
| `rowKey`              | Derives a key when records do not provide `key`.        | `(record: T) => TransferKey`                                                                             | `-`                    |
| `selectAllLabels`     | Custom select-all labels for source and target.         | `TransferSelectAllLabel[]`                                                                               | `-`                    |
| `selectedKeys`        | Controlled selected item keys across both lists.        | `TransferKey[]`                                                                                          | `-`                    |
| `selectionsIcon`      | Custom icon shown near the select-all control.          | `JSX.Element`                                                                                            | `-`                    |
| `showSearch`          | Shows search inputs for filtering both lists.           | `boolean \| { placeholder?: string; defaultValue?: string }`                                             | `false`                |
| `showSelectAll`       | Shows select-all controls.                              | `boolean`                                                                                                | `true`                 |
| `status`              | Visual validation status.                               | `'' \| 'error' \| 'warning'`                                                                             | `''`                   |
| `style`               | Root style.                                             | `JSX.CSSProperties`                                                                                      | `-`                    |
| `styles`              | Semantic DOM styles.                                    | `TransferSemanticStyles`                                                                                 | `-`                    |
| `targetKeys`          | Controlled keys currently in the target list.           | `TransferKey[]`                                                                                          | `-`                    |
| `titles`              | Source and target list titles.                          | `JSX.Element[]`                                                                                          | `['Source', 'Target']` |
| `onChange`            | Called after items move between lists.                  | `(targetKeys, direction, moveKeys) => void`                                                              | `-`                    |
| `onScroll`            | Called when a list scrolls.                             | `(direction, event) => void`                                                                             | `-`                    |
| `onSearch`            | Called when a search input changes.                     | `(direction, value) => void`                                                                             | `-`                    |
| `onSelectChange`      | Called when source or target selected keys change.      | `(sourceSelectedKeys, targetSelectedKeys) => void`                                                       | `-`                    |

### TransferItem

| Property      | Description                              | Type          | Default |
| ------------- | ---------------------------------------- | ------------- | ------- |
| `key`         | Stable item key.                         | `TransferKey` | `-`     |
| `title`       | Item title content.                      | `JSX.Element` | `-`     |
| `description` | Optional item description.               | `JSX.Element` | `-`     |
| `disabled`    | Disables selecting and moving this item. | `boolean`     | `-`     |

### Static helpers

| Name                 | Description                        |
| -------------------- | ---------------------------------- |
| `Transfer.List`      | Low-level list renderer.           |
| `Transfer.Search`    | Search input used by Transfer.     |
| `Transfer.Operation` | Operation button used by Transfer. |
