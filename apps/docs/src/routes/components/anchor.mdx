import { Anchor, Card, Space } from '@solid-ant-design/core'

# Anchor

Anchor renders page-section navigation and highlights the current section while scrolling.

### Basic

:::preview

<AnchorDemo001 />

export const AnchorDemo001 = (() => {
  const items = [
      { key: 'part-1', href: '#anchor-basic-part-1', title: 'Part 1' },
      { key: 'part-2', href: '#anchor-basic-part-2', title: 'Part 2' },
      { key: 'part-3', href: '#anchor-basic-part-3', title: 'Part 3' },
  ];
  const Demo1 = function () {
      return (<Space align="start" class="w-full">
        <Anchor affix={false} showInkInFixed items={items}/>

          <Card id="anchor-basic-part-1" title="Part 1">
            Basic section content.
          </Card>
          <div class="h-6"/>
          <Card id="anchor-basic-part-2" title="Part 2">
            More content for the second anchor.
          </Card>
          <div class="h-6"/>
          <Card id="anchor-basic-part-3" title="Part 3">
            Final section content.
          </Card>
        </div>
      </Space>);
  };

return Demo1
})()

## API

Solid uses `class` instead of React `className`. The Solid root-layer prop is `rootClass`.

### Anchor

| Property           | Description                                                 | Type                                                                              | Default        |
| ------------------ | ----------------------------------------------------------- | --------------------------------------------------------------------------------- | -------------- |
| `items`            | Anchor links rendered as navigation items.                  | `AnchorItem[]`                                                                    | `-`            |
| `affix`            | Wraps the anchor in Affix behavior.                         | `boolean \| Omit<AffixProps, 'offsetTop' \| 'target' \| 'children'>`              | `true`         |
| `offsetTop`        | Top offset used by the affixed anchor.                      | `number`                                                                          | `0`            |
| `targetOffset`     | Scroll target offset used when navigating to anchors.       | `number`                                                                          | `-`            |
| `getContainer`     | Returns the scroll container observed by the anchor.        | `() => Window \| HTMLElement \| undefined \| null`                                | `() => window` |
| `bounds`           | Pixel tolerance when determining the current active anchor. | `number`                                                                          | `5`            |
| `getCurrentAnchor` | Overrides active anchor selection.                          | `(activeLink: string) => string`                                                  | `-`            |
| `direction`        | Anchor layout direction.                                    | `'vertical' \| 'horizontal'`                                                      | `'vertical'`   |
| `replace`          | Uses history replace instead of push for navigation.        | `boolean`                                                                         | `false`        |
| `showInkInFixed`   | Shows the indicator when `affix={false}`.                   | `boolean`                                                                         | `false`        |
| `rootClass`        | Extra class on the root element.                            | `string`                                                                          | `-`            |
| `classNames`       | Semantic DOM class names.                                   | `Record<'root' \| 'item' \| 'itemTitle' \| 'indicator', string> \| fn`            | `-`            |
| `styles`           | Semantic DOM inline styles.                                 | `Record<'root' \| 'item' \| 'itemTitle' \| 'indicator', JSX.CSSProperties> \| fn` | `-`            |
| `onClick`          | Called when a link is clicked.                              | `(event: MouseEvent, link: AnchorItem) => void`                                   | `-`            |
| `onChange`         | Called when the active link changes.                        | `(currentActiveLink: string) => void`                                             | `-`            |

### AnchorItem

| Property       | Description                                                      | Type               | Default |
| -------------- | ---------------------------------------------------------------- | ------------------ | ------- |
| `key`          | Optional stable item key.                                        | `string \| number` | `-`     |
| `href`         | Anchor href target.                                              | `string`           | `-`     |
| `target`       | Link target attribute.                                           | `string`           | `-`     |
| `title`        | Link label content.                                              | `JSX.Element`      | `-`     |
| `replace`      | Overrides replace behavior for this item.                        | `boolean`          | `false` |
| `targetOffset` | Overrides the component `targetOffset` for this link.            | `number`           | `-`     |
| `children`     | Nested anchor items. Not rendered when `direction="horizontal"`. | `AnchorItem[]`     | `-`     |

### Anchor.Link

`Anchor.Link` is supported for compatibility. Prefer `items` for new code.

| Property       | Description                                           | Type          | Default |
| -------------- | ----------------------------------------------------- | ------------- | ------- |
| `href`         | Anchor href target.                                   | `string`      | `-`     |
| `target`       | Link target attribute.                                | `string`      | `-`     |
| `title`        | Link label content.                                   | `JSX.Element` | `-`     |
| `replace`      | Uses history replace instead of push for navigation.  | `boolean`     | `false` |
| `targetOffset` | Overrides the component `targetOffset` for this link. | `number`      | `-`     |
| `children`     | Nested links. Not rendered when horizontal.           | `JSX.Element` | `-`     |

### Semantic DOM

| Slot        | Description              |
| ----------- | ------------------------ |
| `root`      | Root navigation element. |
| `item`      | Link item wrapper.       |
| `itemTitle` | Link title element.      |
| `indicator` | Active-link indicator.   |
