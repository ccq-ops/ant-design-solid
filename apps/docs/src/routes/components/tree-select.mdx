export const TreeSelectDemo005 = (() => {
  const treeData = [
      { title: 'China', value: 'china' },
      { title: 'Europe', value: 'europe' },
  ];
  const Demo = function () {
      const [value, setValue] = createSignal('europe');
      const [open, setOpen] = createSignal(false);
      const [search, setSearch] = createSignal('');
      return (<TreeSelect showSearch value={value()} open={open()} searchValue={search()} treeData={treeData} onChange={(next) => setValue(next)} onOpenChange={setOpen} onSearch={setSearch}/>);
  };

return Demo
})()

```tsx
import { createSignal } from 'solid-js'
import { TreeSelect } from '@solid-ant-design/core'

const treeData = [
  { title: 'China', value: 'china' },
  { title: 'Europe', value: 'europe' },
]

const Demo = function () {
  const [value, setValue] = createSignal('europe')
  const [open, setOpen] = createSignal(false)
  const [search, setSearch] = createSignal('')

  return (
    <TreeSelect
      showSearch
      value={value()}
      open={open()}
      searchValue={search()}
      treeData={treeData}
      onChange={(next) => setValue(next as string)}
      onOpenChange={setOpen}
      onSearch={setSearch}
    />
  )
}

export default Demo
```

## API

### TreeSelect

| Property                  | Description                                                            | Type                                                                     | Default                                                    |
| ------------------------- | ---------------------------------------------------------------------- | ------------------------------------------------------------------------ | ---------------------------------------------------------- |
| `allowClear`              | Show clear button or customize clear icon.                             | `boolean \| { clearIcon?: JSX.Element }`                                 | `false`                                                    |
| `class`                   | Root class. Use Solid `class` rather than React `className`.           | `string`                                                                 | `-`                                                        |
| `classNames`              | Semantic class names.                                                  | `Partial<Record<TreeSelectSemanticSlot, string>>`                        | `-`                                                        |
| `defaultOpen`             | Initial popup open state.                                              | `boolean`                                                                | `false`                                                    |
| `defaultValue`            | Initial selected value.                                                | `TreeSelectValue`                                                        | `-`                                                        |
| `disabled`                | Disable selector.                                                      | `boolean`                                                                | `false`                                                    |
| `fieldNames`              | Customize data field names.                                            | `{ label?: string; value?: string; children?: string }`                  | `{ label: 'title', value: 'value', children: 'children' }` |
| `getPopupContainer`       | Return popup portal container.                                         | `(triggerNode?: HTMLElement) => HTMLElement`                             | `document.body`                                            |
| `labelInValue`            | Wrap value with label.                                                 | `boolean`                                                                | `false`                                                    |
| `listHeight`              | Popup list height.                                                     | `number`                                                                 | `-`                                                        |
| `loadData`                | Load tree data asynchronously.                                         | `(node: TreeSelectNode) => Promise<unknown> \| unknown`                  | `-`                                                        |
| `maxCount`                | Maximum selectable count in multiple/checkable mode.                   | `number`                                                                 | `-`                                                        |
| `maxTagCount`             | Max displayed tags.                                                    | `number \| 'responsive'`                                                 | `-`                                                        |
| `maxTagPlaceholder`       | Placeholder for omitted tags.                                          | `JSX.Element \| (omittedValues: TreeSelectNode[]) => JSX.Element`        | `+ n ...`                                                  |
| `maxTagTextLength`        | Max displayed tag text length.                                         | `number`                                                                 | `-`                                                        |
| `multiple`                | Enable multiple selection.                                             | `boolean`                                                                | `false`                                                    |
| `notFoundContent`         | Empty search content.                                                  | `JSX.Element`                                                            | `Not Found`                                                |
| `open`                    | Controlled popup open state.                                           | `boolean`                                                                | `-`                                                        |
| `placeholder`             | Placeholder content.                                                   | `JSX.Element`                                                            | `-`                                                        |
| `placement`               | Popup placement.                                                       | `bottomLeft \| bottomRight \| topLeft \| topRight`                       | `bottomLeft`                                               |
| `popupMatchSelectWidth`   | Match popup width to selector or set a numeric width.                  | `boolean \| number`                                                      | `true`                                                     |
| `popupRender`             | Customize popup content.                                               | `(originNode: JSX.Element) => JSX.Element`                               | `-`                                                        |
| `prefix`                  | Selector prefix.                                                       | `JSX.Element`                                                            | `-`                                                        |
| `ref`                     | Exposes `focus()` and `blur()`.                                        | `TreeSelectRef \| ((ref) => void)`                                       | `-`                                                        |
| `searchValue`             | Controlled search value.                                               | `string`                                                                 | `-`                                                        |
| `showCheckedStrategy`     | Checked display strategy.                                              | `TreeSelect.SHOW_ALL \| TreeSelect.SHOW_PARENT \| TreeSelect.SHOW_CHILD` | `SHOW_CHILD`                                               |
| `showSearch`              | Enable search or configure search.                                     | `boolean \| TreeSelectSearchConfig`                                      | `false`                                                    |
| `size`                    | Selector size.                                                         | `large \| middle \| small`                                               | `middle`                                                   |
| `status`                  | Validation status.                                                     | `error \| warning`                                                       | `-`                                                        |
| `styles`                  | Semantic inline styles.                                                | `Partial<Record<TreeSelectSemanticSlot, JSX.CSSProperties>>`             | `-`                                                        |
| `suffixIcon`              | Custom suffix icon.                                                    | `JSX.Element \| null`                                                    | `<DownOutlined />`                                         |
| `switcherIcon`            | Custom tree switcher icon.                                             | `TreeIcon`                                                               | `-`                                                        |
| `tagRender`               | Custom tag render in multiple/checkable mode.                          | `(props: TreeSelectTagRenderProps) => JSX.Element`                       | `-`                                                        |
| `treeCheckable`           | Show tree checkboxes.                                                  | `boolean`                                                                | `false`                                                    |
| `treeCheckStrictly`       | Check nodes without parent-child association and force `labelInValue`. | `boolean`                                                                | `false`                                                    |
| `treeData`                | Tree data. Values must be unique.                                      | `TreeSelectNode[]`                                                       | `[]`                                                       |
| `treeDataSimpleMode`      | Use flat simple mode data.                                             | `boolean \| { id?: string; pId?: string; rootPId?: OptionValue }`        | `false`                                                    |
| `treeDefaultExpandAll`    | Expand all tree nodes by default.                                      | `boolean`                                                                | `false`                                                    |
| `treeDefaultExpandedKeys` | Initial expanded tree keys.                                            | `OptionValue[]`                                                          | `-`                                                        |
| `treeExpandAction`        | Expand when clicking or double-clicking title.                         | `false \| click \| doubleClick`                                          | `false`                                                    |
| `treeExpandedKeys`        | Controlled expanded tree keys.                                         | `OptionValue[]`                                                          | `-`                                                        |
| `treeIcon`                | Show tree node icons.                                                  | `boolean`                                                                | `false`                                                    |
| `treeLine`                | Show tree lines.                                                       | `boolean \| { showLeafIcon?: boolean \| TreeIcon }`                      | `false`                                                    |
| `treeLoadedKeys`          | Controlled loaded keys for `loadData`.                                 | `OptionValue[]`                                                          | `[]`                                                       |
| `treeNodeFilterProp`      | Node prop used for filtering.                                          | `string`                                                                 | `value`                                                    |
| `treeNodeLabelProp`       | Node prop used for selected display.                                   | `string`                                                                 | `title`                                                    |
| `treeTitleRender`         | Customize tree title render.                                           | `(nodeData: TreeSelectNode) => JSX.Element`                              | `-`                                                        |
| `value`                   | Controlled selected value.                                             | `TreeSelectValue`                                                        | `-`                                                        |
| `variant`                 | Selector variant.                                                      | `outlined \| borderless \| filled \| underlined`                         | `outlined`                                                 |
| `virtual`                 | Disable virtual scroll when `false`.                                   | `boolean`                                                                | `true`                                                     |
| `onChange`                | Called when value changes.                                             | `(value, label, extra) => void`                                          | `-`                                                        |
| `onOpenChange`            | Called when popup open state changes.                                  | `(open: boolean) => void`                                                | `-`                                                        |
| `onPopupScroll`           | Called when popup scrolls.                                             | `(event: UIEvent) => void`                                               | `-`                                                        |
| `onSearch`                | Called when search input changes.                                      | `(value: string) => void`                                                | `-`                                                        |
| `onSelect`                | Called when a tree node is selected.                                   | `(value, node, extra) => void`                                           | `-`                                                        |
| `onTreeExpand`            | Called when tree expansion changes.                                    | `(expandedKeys: OptionValue[]) => void`                                  | `-`                                                        |

### showSearch

| Property               | Description                           | Type                                           | Default |
| ---------------------- | ------------------------------------- | ---------------------------------------------- | ------- |
| `autoClearSearchValue` | Clear search after selection.         | `boolean`                                      | `true`  |
| `filterTreeNode`       | Custom filter.                        | `boolean \| (inputValue, treeNode) => boolean` | `true`  |
| `searchValue`          | Controlled search value.              | `string`                                       | `-`     |
| `treeNodeFilterProp`   | Node prop used for default filtering. | `string`                                       | `value` |
| `onSearch`             | Search callback.                      | `(value: string) => void`                      | `-`     |

### TreeNode

`TreeSelect.TreeNode` is available for antd API parity, though `treeData` is recommended.

| Property          | Description                                 | Type          | Default |
| ----------------- | ------------------------------------------- | ------------- | ------- |
| `checkable`       | Show checkbox for this node when checkable. | `boolean`     | `true`  |
| `disableCheckbox` | Disable node checkbox.                      | `boolean`     | `false` |
| `disabled`        | Disable node.                               | `boolean`     | `false` |
| `isLeaf`          | Mark node as leaf.                          | `boolean`     | `false` |
| `key`             | Tree key. Defaults to value.                | `OptionValue` | `value` |
| `selectable`      | Whether the node can be selected.           | `boolean`     | `true`  |
| `title`           | Node title.                                 | `JSX.Element` | `-`     |
| `value`           | Unique node value.                          | `OptionValue` | `-`     |
