### Controlled

```tsx
import { createSignal } from 'solid-js'
import { Rate, Space } from '@ant-design-solid/core'

const Demo4 = function () {
  const [value, setValue] = createSignal(3)

  return (
    <Space direction="vertical">
      <Rate value={value()} onChange={setValue} />
      <span>Current value: {value()}</span>
    </Space>
  )
}

export default Demo4
```

```tsx

```

### Focus

```tsx
import { Rate, Space, Button } from '@ant-design-solid/core'
import type { RateRef } from '@ant-design-solid/core'

const Demo9 = function () {
  const rateRef: { current?: RateRef } = {}

  return (
    <Space>
      <Button onClick={() => rateRef.current?.focus()}>Focus</Button>
      <Button onClick={() => rateRef.current?.blur()}>Blur</Button>
      <Rate ref={rateRef} defaultValue={3} />
    </Space>
  )
}

export default Demo9
```

## API

### Rate

| Property        | Description                                                | Type                                                                | Default  |
| --------------- | ---------------------------------------------------------- | ------------------------------------------------------------------- | -------- |
| `value`         | Controlled rating value.                                   | `number`                                                            | `-`      |
| `defaultValue`  | Initial rating value for uncontrolled usage.               | `number`                                                            | `0`      |
| `count`         | Number of rating items.                                    | `number`                                                            | `5`      |
| `allowHalf`     | Allows selecting half values.                              | `boolean`                                                           | `false`  |
| `allowClear`    | Allows clearing the rating by clicking the selected value. | `boolean`                                                           | `true`   |
| `disabled`      | Disables user interaction.                                 | `boolean`                                                           | `false`  |
| `keyboard`      | Enables keyboard interaction.                              | `boolean`                                                           | `true`   |
| `size`          | Rating item size.                                          | `'small' \| 'medium' \| 'large'`                                    | `medium` |
| `character`     | Custom rating character or render function.                | `JSX.Element \| ((props: RateCharacterRenderProps) => JSX.Element)` | `-`      |
| `tooltips`      | Tooltip config for each rating item.                       | `(string \| Omit<TooltipProps, 'children'>)[]`                      | `-`      |
| `ref`           | Imperative methods and native root element.                | `RateRef \| { current?: RateRef } \| ((ref: RateRef) => void)`      | `-`      |
| `class`         | Custom root class.                                         | `string`                                                            | `-`      |
| `style`         | Custom root style.                                         | `JSX.CSSProperties`                                                 | `-`      |
| `onChange`      | Called when the rating value changes.                      | `(next: number) => void`                                            | `-`      |
| `onHoverChange` | Called when hover value changes.                           | `(next: number) => void`                                            | `-`      |
| `onFocus`       | Called when the root receives focus.                       | `JSX.FocusEventHandlerUnion<HTMLDivElement, FocusEvent>`            | `-`      |
| `onBlur`        | Called when the root loses focus.                          | `JSX.FocusEventHandlerUnion<HTMLDivElement, FocusEvent>`            | `-`      |
| `onKeyDown`     | Called before internal keyboard handling.                  | `JSX.EventHandlerUnion<HTMLDivElement, KeyboardEvent>`              | `-`      |

### RateRef

| Method          | Description                   |
| --------------- | ----------------------------- |
| `focus()`       | Focuses the Rate root node.   |
| `blur()`        | Blurs the Rate root node.     |
| `nativeElement` | Native Rate root DOM element. |
