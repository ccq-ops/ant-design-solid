# Splitter

Splitter separates content into resizable panels.

### Basic

```tsx
import { Splitter } from '@ant-design-solid/core'

const panelStyle = {
  display: 'flex',
  'align-items': 'center',
  'justify-content': 'center',
  height: '100%',
  background: 'var(--docs-surface-subtle)',
}

const Demo1 = function () {
  return (
    <Splitter style={{ height: '160px' }}>
      <Splitter.Panel defaultSize="40%" style={panelStyle}>
        Left
      </Splitter.Panel>
      <Splitter.Panel style={panelStyle}>Right</Splitter.Panel>
    </Splitter>
  )
}

export default Demo1
```

### Collapsible

```tsx
import { Splitter } from '@ant-design-solid/core'

const panelStyle = {
  display: 'flex',
  'align-items': 'center',
  'justify-content': 'center',
  height: '100%',
  background: 'var(--docs-surface-subtle)',
}

const Demo4 = function () {
  return (
    <Splitter
      collapsible={{ motion: true }}
      onCollapse={(collapsed, sizes) => {
        console.log(collapsed, sizes)
      }}
      style={{ height: '180px' }}
    >
      <Splitter.Panel defaultSize={180} min={80} collapsible={{ end: true }} style={panelStyle}>
        Left
      </Splitter.Panel>
      <Splitter.Panel collapsible={{ start: true, showCollapsibleIcon: 'auto' }} style={panelStyle}>
        Right
      </Splitter.Panel>
    </Splitter>
  )
}

export default Demo4
```

### Semantic styles

```tsx
import { Splitter } from '@ant-design-solid/core'

const panelStyle = {
  display: 'flex',
  'align-items': 'center',
  'justify-content': 'center',
  height: '100%',
}

const Demo7 = function () {
  return (
    <Splitter
      classNames={{
        root: 'docs-splitter-root',
        panel: 'docs-splitter-panel',
        dragger: { default: 'docs-splitter-dragger', active: 'docs-splitter-dragger-active' },
      }}
      styles={{
        root: { height: '160px' },
        panel: { background: 'var(--docs-surface-subtle)' },
        dragger: {
          default: { color: 'var(--docs-text-secondary)' },
          active: { color: 'var(--docs-primary)' },
        },
      }}
    >
      <Splitter.Panel defaultSize="40%" style={panelStyle}>
        Left
      </Splitter.Panel>
      <Splitter.Panel style={panelStyle}>Right</Splitter.Panel>
    </Splitter>
  )
}

export default Demo7
```

### Destroy hidden content

```tsx
import { Splitter } from '@ant-design-solid/core'

const panelStyle = {
  display: 'flex',
  'align-items': 'center',
  'justify-content': 'center',
  height: '100%',
  background: 'var(--docs-surface-subtle)',
}

const Demo8 = function () {
  return (
    <Splitter destroyOnHidden collapsible={{ motion: true }} style={{ height: '160px' }}>
      <Splitter.Panel defaultSize={160} collapsible={{ end: true }} style={panelStyle}>
        Hidden content is unmounted after collapse.
      </Splitter.Panel>
      <Splitter.Panel style={panelStyle}>Right</Splitter.Panel>
    </Splitter>
  )
}

export default Demo8
```

## API

### Splitter

| Property               | Description                                                           | Type                                                                      | Default        |
| ---------------------- | --------------------------------------------------------------------- | ------------------------------------------------------------------------- | -------------- |
| `classNames`           | Customize class for semantic slots.                                   | `{ root?: string; panel?: string; dragger?: string \| DraggerClass }`     | `-`            |
| `collapsible`          | Global collapse configuration.                                        | `{ motion?: boolean; icon?: { start?: JSX.Element; end?: JSX.Element } }` | `-`            |
| `collapsibleIcon`      | Custom collapsible icon. Use `collapsible.icon` instead.              | `{ start?: JSX.Element; end?: JSX.Element }`                              | `-`            |
| `destroyOnHidden`      | Destroy panel content when collapsed or when panel size is `0`.       | `boolean`                                                                 | `false`        |
| `draggerIcon`          | Custom dragger icon.                                                  | `JSX.Element`                                                             | `-`            |
| `layout`               | Panel layout direction. Kept for compatibility; prefer `orientation`. | `'horizontal' \| 'vertical'`                                              | `'horizontal'` |
| `lazy`                 | Defer panel size updates until dragging ends.                         | `boolean`                                                                 | `false`        |
| `onCollapse`           | Called when a panel expands or collapses.                             | `(collapsed: boolean[], sizes: number[]) => void`                         | `-`            |
| `onDraggerDoubleClick` | Called when a dragger is double clicked.                              | `(index: number) => void`                                                 | `-`            |
| `onResizeStart`        | Called when dragging a resize bar starts.                             | `(sizes: number[]) => void`                                               | `-`            |
| `onResize`             | Called while panel sizes change.                                      | `(sizes: number[]) => void`                                               | `-`            |
| `onResizeEnd`          | Called when dragging a resize bar ends.                               | `(sizes: number[]) => void`                                               | `-`            |
| `orientation`          | Panel layout direction.                                               | `'horizontal' \| 'vertical'`                                              | `'horizontal'` |
| `styles`               | Customize style for semantic slots.                                   | `{ root?: CSSProperties; panel?: CSSProperties; dragger?: DraggerStyle }` | `-`            |
| `vertical`             | Use vertical orientation. `orientation` has higher priority.          | `boolean`                                                                 | `false`        |

### Splitter.Panel

| Property          | Description                                                      | Type                                                                                     | Default |
| ----------------- | ---------------------------------------------------------------- | ---------------------------------------------------------------------------------------- | ------- |
| `collapsible`     | Quick folding configuration.                                     | `boolean \| { start?: boolean; end?: boolean; showCollapsibleIcon?: boolean \| 'auto' }` | `false` |
| `defaultSize`     | Initial panel size. Number means px; string supports percent.    | `number \| string \| undefined`                                                          | `-`     |
| `destroyOnHidden` | Destroy content when this panel is collapsed or size is `0`.     | `boolean`                                                                                | `-`     |
| `max`             | Maximum panel size. Number means px; string supports percent.    | `number \| string \| undefined`                                                          | `-`     |
| `min`             | Minimum panel size. Number means px; string supports percent.    | `number \| string \| undefined`                                                          | `-`     |
| `resizable`       | Whether adjacent resize bars can resize this panel.              | `boolean`                                                                                | `true`  |
| `size`            | Controlled panel size. Number means px; string supports percent. | `number \| string \| undefined`                                                          | `-`     |

### Semantic DOM

| Slot      | Description             |
| --------- | ----------------------- |
| `root`    | Root Splitter element.  |
| `panel`   | Panel element.          |
| `dragger` | Resize dragger element. |
