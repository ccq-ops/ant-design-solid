# Tooltip

### Hover

```tsx
import { createSignal } from 'solid-js'
import { Button, Space, Tooltip } from '@ant-design-solid/core'

const Demo1 = function () {
  return (
    <Tooltip title="Helpful text">
      <Button>Hover</Button>
    </Tooltip>
  )
}

export default Demo1
```

### Triggers

```tsx
import { createSignal } from 'solid-js'
import { Button, Space, Tooltip } from '@ant-design-solid/core'

const Demo2 = function () {
  return (
    <Space wrap>
      <Tooltip trigger="click" title="Clicked">
        <Button>Click</Button>
      </Tooltip>
      <Tooltip trigger="focus" title="Focused">
        <Button>Focus</Button>
      </Tooltip>
    </Space>
  )
}

export default Demo2
```

### Controlled

```tsx
import { createSignal } from 'solid-js'
import { Button, Space, Tooltip } from '@ant-design-solid/core'

const Demo4 = function () {
  return (
    <Space>
      <Tooltip open={open()} onOpenChange={setOpen} title="Controlled text">
        <Button>Controlled</Button>
      </Tooltip>
      <Button onClick={() => setOpen((next) => !next)}>{open() ? 'Close' : 'Open'}</Button>
    </Space>
  )
}

export default Demo4
```

## API

| Property            | Description                                           | Type                                         | Default   |
| ------------------- | ----------------------------------------------------- | -------------------------------------------- | --------- |
| `title`             | Tooltip content.                                      | `JSX.Element`                                | `-`       |
| `placement`         | Overlay placement relative to the trigger.            | `TooltipPlacement`                           | `-`       |
| `trigger`           | Interaction that opens the overlay.                   | `'hover' \| 'click' \| 'focus'`              | `'hover'` |
| `open`              | Controlled overlay open state.                        | `boolean`                                    | `-`       |
| `defaultOpen`       | Initial overlay open state for uncontrolled usage.    | `boolean`                                    | `false`   |
| `onOpenChange`      | Called when overlay open state changes.               | `(open: boolean) => void`                    | `-`       |
| `mouseEnterDelay`   | Delay before opening on hover.                        | `number`                                     | `-`       |
| `mouseLeaveDelay`   | Delay before closing after hover leaves.              | `number`                                     | `-`       |
| `overlayClass`      | Additional class for the overlay.                     | `string`                                     | `-`       |
| `overlayStyle`      | Inline style for the overlay.                         | `JSX.CSSProperties`                          | `-`       |
| `zIndex`            | Overrides overlay z-index.                            | `number`                                     | `-`       |
| `getPopupContainer` | Returns the element used to mount the overlay portal. | `(triggerNode?: HTMLElement) => HTMLElement` | `-`       |
