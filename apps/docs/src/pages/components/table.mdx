# Table

### Basic

```tsx
import { Badge, Table, Tag } from '@ant-design-solid/core'

const data = [
  { key: '1', name: 'Ada Lovelace', age: 32, status: 'active' },
  { key: '2', name: 'Grace Hopper', age: 36, status: 'idle' },
]

const columns = [
  { title: 'Name', dataIndex: 'name' },
  { title: 'Age', dataIndex: 'age', align: 'right' },
  {
    title: 'Status',
    dataIndex: 'status',
    render: (value) =>
      value === 'active' ? (
        <Tag color="success">Active</Tag>
      ) : (
        <Badge status="default" text="Idle" />
      ),
  },
]

const Demo1 = function () {
  return <Table columns={columns} dataSource={data} />
}

export default Demo1
```

### Sort and filter

```tsx
import { Badge, Table, Tag } from '@ant-design-solid/core'

const data = [
  { key: '1', name: 'Ada Lovelace', age: 32, status: 'active' },
  { key: '2', name: 'Grace Hopper', age: 36, status: 'idle' },
  { key: '3', name: 'Linus Torvalds', age: 28, status: 'active' },
]

const columns = [
  {
    title: 'Name',
    dataIndex: 'name',
    sorter: (a, b) => a.name.localeCompare(b.name),
  },
  {
    title: 'Age',
    dataIndex: 'age',
    align: 'right',
    sorter: (a, b) => a.age - b.age,
    defaultSortOrder: 'ascend',
  },
  {
    title: 'Status',
    dataIndex: 'status',
    filters: [
      { text: 'Active', value: 'active' },
      { text: 'Idle', value: 'idle' },
    ],
    onFilter: (value, record) => record.status === value,
    render: (value) =>
      value === 'active' ? (
        <Tag color="success">Active</Tag>
      ) : (
        <Badge status="default" text="Idle" />
      ),
  },
]

const Demo5 = function () {
  return <Table columns={columns} dataSource={data} pagination={false} />
}

export default Demo5
```

### Controlled sort and filter

```tsx
import { createSignal } from 'solid-js'
import { Table } from '@ant-design-solid/core'

const data = [
  { key: '1', name: 'Ada Lovelace', age: 32, status: 'active' },
  { key: '2', name: 'Grace Hopper', age: 36, status: 'idle' },
  { key: '3', name: 'Linus Torvalds', age: 28, status: 'active' },
]

const Demo18 = function () {
  const [sortOrder, setSortOrder] = createSignal<'ascend' | 'descend' | null>('ascend')
  const [filteredStatus, setFilteredStatus] = createSignal<(string | number | boolean)[]>([
    'active',
  ])

  const columns = [
    { title: 'Name', dataIndex: 'name' },
    {
      title: 'Age',
      dataIndex: 'age',
      sorter: (a, b) => a.age - b.age,
      sortOrder: sortOrder(),
    },
    {
      title: 'Status',
      dataIndex: 'status',
      filters: [
        { text: 'Active', value: 'active' },
        { text: 'Idle', value: 'idle' },
      ],
      filteredValue: filteredStatus(),
      onFilter: (value, record) => record.status === value,
    },
  ]

  return (
    <Table
      columns={columns}
      dataSource={data}
      pagination={false}
      onChange={(_pagination, filters, sorter) => {
        setSortOrder(sorter.order)
        setFilteredStatus(filters.status ?? [])
      }}
    />
  )
}

export default Demo18
```

## API

### Table

| Property       | Description                                      | Type                                                                                                                   | Default                 |
| -------------- | ------------------------------------------------ | ---------------------------------------------------------------------------------------------------------------------- | ----------------------- |
| `columns`      | Table column definitions.                        | `TableColumn<T>[]`                                                                                                     | `-`                     |
| `dataSource`   | Data records rendered as rows.                   | `T[]`                                                                                                                  | `-`                     |
| `rowKey`       | Record field or function used as row key.        | `keyof T \| string \| ((record: T, index: number) => string \| number)`                                                | `record.key \|\| index` |
| `loading`      | Shows the loading indicator.                     | `boolean \| { spinning?: boolean; tip?: JSX.Element }`                                                                 | `false`                 |
| `emptyText`    | Empty state content.                             | `JSX.Element`                                                                                                          | `'No data'`             |
| `locale`       | Locale text. Currently supports empty text.      | `{ emptyText?: JSX.Element \| (() => JSX.Element) }`                                                                   | `-`                     |
| `pagination`   | Pagination config, or `false` to disable paging. | `false \| TablePaginationConfig`                                                                                       | `{}`                    |
| `rowSelection` | Row selection config.                            | `TableRowSelection<T>`                                                                                                 | `-`                     |
| `expandable`   | Expandable row config.                           | `TableExpandableConfig<T>`                                                                                             | `-`                     |
| `summary`      | Summary row renderer for current page data.      | `(currentData: T[]) => JSX.Element`                                                                                    | `-`                     |
| `scroll`       | Basic horizontal and vertical scroll config.     | `TableScrollConfig`                                                                                                    | `-`                     |
| `title`        | Table title renderer for current page data.      | `(currentData: T[]) => JSX.Element`                                                                                    | `-`                     |
| `footer`       | Table footer renderer for current page data.     | `(currentData: T[]) => JSX.Element`                                                                                    | `-`                     |
| `tableLayout`  | Native table layout mode.                        | `'auto' \| 'fixed'`                                                                                                    | `fixed` with ellipsis   |
| `size`         | Table size.                                      | `'small' \| 'middle' \| 'large'`                                                                                       | `'middle'`              |
| `bordered`     | Whether to render bordered table styling.        | `boolean`                                                                                                              | `false`                 |
| `showHeader`   | Whether to render the header row.                | `boolean`                                                                                                              | `true`                  |
| `className`    | Additional class name for the root wrapper.      | `string`                                                                                                               | `-`                     |
| `rowClassName` | Row class name.                                  | `string \| ((record: T, index: number) => string)`                                                                     | `-`                     |
| `onRow`        | Returns extra props for each table row.          | `(record: T, index: number) => JSX.HTMLAttributes<HTMLTableRowElement>`                                                | `-`                     |
| `onHeaderRow`  | Returns extra props for the header row.          | `(columns: TableColumn<T>[], index: number) => JSX.HTMLAttributes<HTMLTableRowElement>`                                | `-`                     |
| `onChange`     | Called after pagination, sorting, or filtering.  | `(pagination: TableChangePagination, filters, sorter: TableSorterResult<T>, extra: TableCurrentDataSource<T>) => void` | `-`                     |

### TableScrollConfig

| Property                   | Description                                               | Type                       | Default |
| -------------------------- | --------------------------------------------------------- | -------------------------- | ------- |
| `x`                        | Enables horizontal scroll and sets table min-width.       | `string \| number \| true` | `-`     |
| `y`                        | Enables vertical scroll and sets container max height.    | `string \| number`         | `-`     |
| `scrollToFirstRowOnChange` | Compatibility field. Currently accepted but not acted on. | `boolean`                  | `-`     |

### TableChangePagination

| Property   | Description                     | Type     |
| ---------- | ------------------------------- | -------- |
| `current`  | Current page.                   | `number` |
| `pageSize` | Current page size.              | `number` |
| `total`    | Total rows after table process. | `number` |

### TableCurrentDataSource

| Property            | Description                               | Type                               |
| ------------------- | ----------------------------------------- | ---------------------------------- |
| `currentDataSource` | Current processed data after filter/sort. | `T[]`                              |
| `action`            | Change trigger.                           | `'paginate' \| 'sort' \| 'filter'` |

### TableSorterResult

| Property    | Description               | Type                                                 |
| ----------- | ------------------------- | ---------------------------------------------------- |
| `column`    | Sorted column definition. | `TableColumn<T>`                                     |
| `order`     | Current sort order.       | `'ascend' \| 'descend' \| null`                      |
| `field`     | Sorted column data field. | `keyof T \| string \| readonly (string \| number)[]` |
| `columnKey` | Sorted column key.        | `string \| number`                                   |

### TableRowSelection

| Property                 | Description                                  | Type                                                                                   | Default      |
| ------------------------ | -------------------------------------------- | -------------------------------------------------------------------------------------- | ------------ |
| `type`                   | Selection type.                              | `'checkbox' \| 'radio'`                                                                | `'checkbox'` |
| `selectedRowKeys`        | Controlled selected row keys.                | `(string \| number)[]`                                                                 | `-`          |
| `defaultSelectedRowKeys` | Initial selected row keys.                   | `(string \| number)[]`                                                                 | `[]`         |
| `columnTitle`            | Selection column title.                      | `JSX.Element`                                                                          | `-`          |
| `columnWidth`            | Selection column width.                      | `string \| number`                                                                     | `48`         |
| `getCheckboxProps`       | Props for each checkbox or radio input.      | `(record: T) => Partial<JSX.InputHTMLAttributes<HTMLInputElement>>`                    | `-`          |
| `onChange`               | Called when selected rows change.            | `(selectedRowKeys, selectedRows, info: { type: 'all' \| 'none' \| 'single' }) => void` | `-`          |
| `onSelect`               | Called when a row is selected or deselected. | `(record: T, selected: boolean, selectedRows: T[], nativeEvent: Event) => void`        | `-`          |

### TableExpandableConfig

| Property                 | Description                             | Type                                                                           | Default |
| ------------------------ | --------------------------------------- | ------------------------------------------------------------------------------ | ------- |
| `columnTitle`            | Expand column title.                    | `JSX.Element`                                                                  | `-`     |
| `columnWidth`            | Expand column width.                    | `string \| number`                                                             | `48`    |
| `defaultExpandAllRows`   | Expands all rows initially.             | `boolean`                                                                      | `false` |
| `defaultExpandedRowKeys` | Initial expanded row keys.              | `(string \| number)[]`                                                         | `[]`    |
| `expandedRowKeys`        | Controlled expanded row keys.           | `(string \| number)[]`                                                         | `-`     |
| `expandedRowRender`      | Expanded content renderer.              | `(record: T, index: number, indent: number, expanded: boolean) => JSX.Element` | `-`     |
| `expandedRowClassName`   | Expanded row class name.                | `string \| ((record: T, index: number, indent: number) => string)`             | `-`     |
| `expandRowByClick`       | Expands rows by clicking the data row.  | `boolean`                                                                      | `false` |
| `rowExpandable`          | Whether a record can be expanded.       | `(record: T) => boolean`                                                       | `-`     |
| `onExpand`               | Called when a row expands or collapses. | `(expanded: boolean, record: T) => void`                                       | `-`     |
| `onExpandedRowsChange`   | Called when expanded row keys change.   | `(expandedRows: (string \| number)[]) => void`                                 | `-`     |

### TablePaginationConfig

Table pagination reuses the Pagination component API and adds antd-compatible placement fields.

| Property           | Description                                   | Type                                                                                                         | Default             |
| ------------------ | --------------------------------------------- | ------------------------------------------------------------------------------------------------------------ | ------------------- |
| `current`          | Controlled current page.                      | `number`                                                                                                     | `-`                 |
| `defaultCurrent`   | Initial current page.                         | `number`                                                                                                     | `1`                 |
| `pageSize`         | Controlled page size.                         | `number`                                                                                                     | `-`                 |
| `defaultPageSize`  | Initial page size.                            | `number`                                                                                                     | `10`                |
| `total`            | Total item count. Defaults to processed rows. | `number`                                                                                                     | `data.length`       |
| `showSizeChanger`  | Shows page size selector.                     | `boolean \| SelectProps`                                                                                     | `false`             |
| `pageSizeOptions`  | Page size options.                            | `(string \| number)[]`                                                                                       | `[10, 20, 50, 100]` |
| `showQuickJumper`  | Shows quick jumper.                           | `boolean \| { goButton?: JSX.Element }`                                                                      | `false`             |
| `hideOnSinglePage` | Hides pagination when only one page.          | `boolean`                                                                                                    | `false`             |
| `showTotal`        | Renders total text.                           | `(total: number, range: [number, number]) => JSX.Element`                                                    | `-`                 |
| `placement`        | antd-compatible pagination placement values.  | `Array<'topStart' \| 'topCenter' \| 'topEnd' \| 'bottomStart' \| 'bottomCenter' \| 'bottomEnd' \| 'none'>`   | `-`                 |
| `position`         | Deprecated antd-compatible placement values.  | `Array<'topLeft' \| 'topCenter' \| 'topRight' \| 'bottomLeft' \| 'bottomCenter' \| 'bottomRight' \| 'none'>` | `-`                 |
