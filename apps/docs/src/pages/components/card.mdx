# Card

### Basic

```tsx
import { Button, Card, Space } from '@ant-design-solid/core'

const Demo1 = function () {
  return (
    <Card class="w-80">
      <p>Card content</p>
      <p>Card content</p>
      <p>Card content</p>
    </Card>
  )
}

export default Demo1
```

### Title and extra

```tsx
import { Button, Card, Space } from '@ant-design-solid/core'

const Demo2 = function () {
  return (
    <Card title="Card title" extra={<a href="#more">More</a>} class="w-90">
      <p>Content with a header and an extra action.</p>
    </Card>
  )
}

export default Demo2
```

### Cover and actions

```tsx
import { Button, Card, Space } from '@ant-design-solid/core'

const Demo3 = function () {
  return (
    <Card
      title="Project"
      cover={
        <div class="flex h-[120px] items-center justify-center bg-gradient-to-br from-blue-50 to-green-50 font-semibold text-blue-600">
          Cover
        </div>
      }
      actions={[
        <a href="#edit">Edit</a>,
        <a href="#share">Share</a>,
        <a href="#more-actions">More</a>,
      ]}
      class="w-90"
    >
      <p>Cards can include a visual cover and a footer action bar.</p>
    </Card>
  )
}

export default Demo3
```

### Variants

```tsx
import { Button, Card, Space } from '@ant-design-solid/core'

const Demo4 = function () {
  return (
    <Space wrap align="start">
      <Card hoverable size="small" title="Small" class="w-60">
        Small cards reduce header and body padding.
      </Card>
      <Card bordered={false} title="Borderless" class="w-60">
        Borderless cards sit directly on the surrounding surface.
      </Card>
      <Card hoverable title="Hoverable" class="w-60">
        Hoverable cards receive an elevated treatment on hover.
      </Card>
    </Space>
  )
}

export default Demo4
```

### With actions

```tsx
import { Button, Card, Space } from '@ant-design-solid/core'

const Demo5 = function () {
  return (
    <Card actions={[<Button type="text">Edit</Button>, <Button type="text">Archive</Button>]}>
      Action slots accept any JSX element, including buttons.
    </Card>
  )
}

export default Demo5
```

## API

| Property    | Description                                             | Type                   | Default     |
| ----------- | ------------------------------------------------------- | ---------------------- | ----------- |
| `title`     | Card title rendered in the header.                      | `JSX.Element`          | `-`         |
| `extra`     | Extra content rendered on the right side of the header. | `JSX.Element`          | `-`         |
| `cover`     | Cover content rendered before the card body.            | `JSX.Element`          | `-`         |
| `actions`   | Footer action items.                                    | `JSX.Element[]`        | `-`         |
| `bordered`  | Whether to render a card border.                        | `boolean`              | `true`      |
| `hoverable` | Enables hover elevation styling.                        | `boolean`              | `false`     |
| `size`      | Card size.                                              | `'default' \| 'small'` | `'default'` |
| `children`  | Card body content.                                      | `JSX.Element`          | `-`         |
