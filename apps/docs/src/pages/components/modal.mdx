# Modal

Modal displays important content in a focused dialog above the page.

### Basic

```tsx
import { createSignal } from 'solid-js'
import { Button, Modal, Space, message } from '@ant-design-solid/core'

const Demo1 = function () {
  const [basicOpen, setBasicOpen] = createSignal(false)

  return (
<Button type="primary" onClick={() => setBasicOpen(true)}>
          Open modal
        </Button>
        <Modal
          open={basicOpen()}
          title="Basic modal"
          onOk={() => {
            setBasicOpen(false)
          }}
          onCancel={() => {
            setBasicOpen(false)
          }}
        >
          Modal content stays mounted in a portal while open.
        </Modal>
  )
}

export default Demo1
```

            setButtonsOpen(false)

### Confirm loading

```tsx
import { createSignal } from 'solid-js'
import { Button, Modal, Space, message } from '@ant-design-solid/core'

const Demo4 = function () {
  const [loadingOpen, setLoadingOpen] = createSignal(false)

  const [loading, setLoading] = createSignal(false)

  function handleLoadingOk() {
  setLoading(true)
  window.setTimeout(() => {
  setLoading(false)
  setLoadingOpen(false)
  message.success('Saved')
  }, 800)
  }

  return (
<Button onClick={() => setLoadingOpen(true)}>Open async modal</Button>
        <Modal
          open={loadingOpen()}
          title="Confirm loading"
          confirmLoading={loading()}
          onOk={handleLoadingOk}
          onCancel={() => {
            setLoadingOpen(false)
          }}
        >
          Click OK to simulate an async operation.
        </Modal>
  )
}

export default Demo4
```

### Custom container

```tsx
import { createSignal } from 'solid-js'
import { Button, Modal, Space, message } from '@ant-design-solid/core'

const Demo6 = function () {
  const [containerOpen, setContainerOpen] = createSignal(false)

  const [inlineOpen, setInlineOpen] = createSignal(false)

  let containerRef

  return (
<Space wrap>
          <Button onClick={() => setContainerOpen(true)}>Mount into container</Button>
          <Button onClick={() => setInlineOpen(true)}>Render inline</Button>
        </Space>
        <div
          ref={containerRef}
          style={{
            position: 'relative',
            margin: '12px 0',
            padding: '12px',
            border: '1px dashed #d9d9d9',
            'min-height': '48px',
          }}
        >
          Custom container target
        </div>
        <Modal
          open={containerOpen()}
          title="Custom container"
          getContainer={() => containerRef}
          onOk={() => {
            setContainerOpen(false)
          }}
          onCancel={() => {
            setContainerOpen(false)
          }}
        >
          This modal portal is mounted into the dashed container.
        </Modal>
        <div style={{ position: 'relative' }}>
          <Modal
            open={inlineOpen()}
            title="Inline modal"
            getContainer={false}
            onOk={() => {
              setInlineOpen(false)
            }}
            onCancel={() => {
              setInlineOpen(false)
            }}
          >
            Setting getContainer to false renders the modal inline.
          </Modal>
        </div>
  )
}

export default Demo6
```

### Render control

```tsx
import { createSignal } from 'solid-js'
import { Button, Modal, Space, message } from '@ant-design-solid/core'

const Demo8 = function () {
  const [forceOpen, setForceOpen] = createSignal(false)

  const [destroyOpen, setDestroyOpen] = createSignal(false)

  const [renderOpen, setRenderOpen] = createSignal(false)

  return (
<Space wrap>
          <Button onClick={() => setForceOpen(true)}>Force rendered modal</Button>
          <Button onClick={() => setDestroyOpen(true)}>Destroy on hidden</Button>
          <Button onClick={() => setRenderOpen(true)}>Wrapped render</Button>
        </Space>
        <Modal
          open={forceOpen()}
          forceRender
          title="Force render"
          afterOpenChange={(open) => message.info(`Force modal open: ${open}`)}
          onOk={() => {
            setForceOpen(false)
          }}
          onCancel={() => {
            setForceOpen(false)
          }}
        >
          This modal content is rendered before first open.
        </Modal>
        <Modal
          open={destroyOpen()}
          destroyOnHidden
          title="Destroy on hidden"
          onOk={() => {
            setDestroyOpen(false)
          }}
          onCancel={() => {
            setDestroyOpen(false)
          }}
        >
          Content is unmounted after the modal is hidden.
        </Modal>
        <Modal
          open={renderOpen()}
          title="modalRender"
          modalRender={(node) => <section style={{ border: '2px solid #1677ff' }}>{node}</section>}
          onOk={() => {
            setRenderOpen(false)
          }}
          onCancel={() => {
            setRenderOpen(false)
          }}
        >
          modalRender wraps the dialog node.
        </Modal>
  )
}

export default Demo8
```

### Static method advanced config

```tsx
import { createSignal } from 'solid-js'
import { Button, Modal, Space, message } from '@ant-design-solid/core'

const Demo10 = function () {
  function openAdvancedConfirm() {
    const modal = Modal.confirm({
      title: 'Archive project?',
      content: 'You can restore it from the archive later.',
      icon: <span style={{ color: '#faad14' }}>!</span>,
      okText: 'Archive',
      okButtonProps: { danger: true },
      cancelButtonProps: { type: 'text' },
      afterClose: () => message.info('Confirm closed'),
      onOk: (close) => {
        message.success('Archived')
        close?.()
      },
    })

    window.setTimeout(() => {
      modal.update((prev) => ({ title: `${prev.title} (updated)` }))
    }, 600)
  }

  return <Button onClick={openAdvancedConfirm}>Open advanced confirm</Button>
}

export default Demo10
```

## API

### Modal

| Property            | Description                                                                                                  | Type                                                          | Default                 |
| ------------------- | ------------------------------------------------------------------------------------------------------------ | ------------------------------------------------------------- | ----------------------- |
| `open`              | Controls whether the modal is visible.                                                                       | `boolean`                                                     | `false`                 |
| `title`             | Modal title content.                                                                                         | `JSX.Element`                                                 | `-`                     |
| `footer`            | Custom footer content. Set to null to hide the footer, or use a render function to wrap the default buttons. | `JSX.Element \| ((originNode, extra) => JSX.Element) \| null` | `OK and Cancel buttons` |
| `okText`            | Text for the default OK button.                                                                              | `JSX.Element`                                                 | `'OK'`                  |
| `cancelText`        | Text for the default Cancel button.                                                                          | `JSX.Element`                                                 | `'Cancel'`              |
| `confirmLoading`    | Shows loading state on the default OK button.                                                                | `boolean`                                                     | `false`                 |
| `closable`          | Shows or configures the close button in the top corner.                                                      | `boolean \| ModalClosableConfig`                              | `true`                  |
| `closeIcon`         | Custom close icon.                                                                                           | `JSX.Element`                                                 | `<CloseOutlined />`     |
| `mask`              | Shows or configures the background mask.                                                                     | `boolean \| ModalMaskConfig`                                  | `true`                  |
| `maskClosable`      | Closes the modal when clicking the wrapper outside the dialog. Ignored when mask.closable is provided.       | `boolean`                                                     | `true`                  |
| `keyboard`          | Closes the top modal when pressing Escape.                                                                   | `boolean`                                                     | `true`                  |
| `centered`          | Vertically centers the modal dialog.                                                                         | `boolean`                                                     | `false`                 |
| `width`             | Modal width.                                                                                                 | `number \| string`                                            | `520`                   |
| `zIndex`            | Overrides modal z-index.                                                                                     | `number`                                                      | `-`                     |
| `okType`            | Button type for the default OK button.                                                                       | `ButtonType`                                                  | `'primary'`             |
| `okButtonProps`     | Props for the default OK button.                                                                             | `ButtonProps`                                                 | `-`                     |
| `cancelButtonProps` | Props for the default Cancel button.                                                                         | `ButtonProps`                                                 | `-`                     |
| `className`         | Class name for the modal dialog.                                                                             | `string`                                                      | `-`                     |
| `wrapClassName`     | Class name for the modal wrapper.                                                                            | `string`                                                      | `-`                     |
| `classNames`        | Semantic DOM class names.                                                                                    | `ModalClassNames`                                             | `-`                     |
| `styles`            | Semantic DOM inline styles.                                                                                  | `ModalStyles`                                                 | `-`                     |
| `destroyOnHidden`   | Unmount children when the modal is hidden.                                                                   | `boolean`                                                     | `false`                 |
| `forceRender`       | Render modal content before it is opened.                                                                    | `boolean`                                                     | `false`                 |
| `getContainer`      | Custom mount node, selector, or false for inline rendering.                                                  | `HTMLElement \| () => HTMLElement \| string \| false`         | `document.body`         |
| `modalRender`       | Custom render wrapper for the modal dialog node.                                                             | `(node: JSX.Element) => JSX.Element`                          | `-`                     |
| `loading`           | Shows a loading body placeholder.                                                                            | `boolean`                                                     | `false`                 |
| `onOk`              | Called when the default OK button is clicked.                                                                | `() => void \| Promise<void>`                                 | `-`                     |
| `onCancel`          | Called when the modal requests to close.                                                                     | `() => void`                                                  | `-`                     |
| `afterClose`        | Called after the modal changes from open to closed.                                                          | `() => void`                                                  | `-`                     |
| `afterOpenChange`   | Called when the open state changes.                                                                          | `(open: boolean) => void`                                     | `-`                     |

| `closeIcon`         | Custom close icon.                                                           | `JSX.Element`                                                 | `-`         |
| `mask`              | Shows or configures the background mask.                                     | `boolean \| ModalMaskConfig`                                  | `true`      |
| `maskClosable`      | Allows closing by clicking outside.                                          | `boolean`                                                     | `false`     |
| `keyboard`          | Allows closing with Escape.                                                  | `boolean`                                                     | `true`      |

| `styles`            | Semantic DOM inline styles.                                                  | `ModalStyles`                                                 | `-`         |
| `okType`            | Button type for the OK button.                                               | `ButtonType`                                                  | `'primary'` |
| `okButtonProps`     | Props for the OK button.                                                     | `ButtonProps`                                                 | `-`         |
| `cancelButtonProps` | Props for the Cancel button.                                                 | `ButtonProps`                                                 | `-`         |
| `footer`            | Custom footer content.                                                       | `JSX.Element \| ((originNode, extra) => JSX.Element) \| null` | `-`         |
| `icon`              | Custom confirm icon.                                                         | `JSX.Element`                                                 | `-`         |
| `afterClose`        | Called after the dialog is destroyed.                                        | `() => void`                                                  | `-`         |
| `getContainer`      | Custom mount node, selector, or false for inline rendering.                  | `HTMLElement \| () => HTMLElement \| string \| false`         | `-`         |
| `modalRender`       | Custom render wrapper for the dialog node.                                   | `(node: JSX.Element) => JSX.Element`                          | `-`         |
| `destroyOnHidden`   | Unmount children when hidden.                                                | `boolean`                                                     | `false`     |
| `forceRender`       | Render dialog content before it opens.                                       | `boolean`                                                     | `false`     |
| `onOk`              | Called when OK is clicked. Receives a close function for manual closing.     | `(close?: () => void) => void \| Promise<void>`               | `-`         |
| `onCancel`          | Called when Cancel is clicked. Receives a close function for manual closing. | `(close?: () => void) => void \| Promise<void>`               | `-`         |

### ModalMaskConfig

| Property   | Description                                                                 | Type      | Default |
| ---------- | --------------------------------------------------------------------------- | --------- | ------- |
| `enabled`  | Whether to render the mask element.                                         | `boolean` | `true`  |
| `blur`     | Adds a backdrop blur effect to the mask.                                    | `boolean` | `false` |
| `closable` | Whether wrapper clicks close the modal. Takes precedence over maskClosable. | `boolean` | `-`     |

### ModalClosableConfig

| Property     | Description                                         | Type          | Default |
| ------------ | --------------------------------------------------- | ------------- | ------- |
| `closeIcon`  | Custom icon rendered inside the close button.       | `JSX.Element` | `-`     |
| `disabled`   | Disables the close button action.                   | `boolean`     | `false` |
| `onClose`    | Called when the close button is clicked.            | `() => void`  | `-`     |
| `afterClose` | Called after the modal changes from open to closed. | `() => void`  | `-`     |

| --------- | -------------------------------------------------- | ----------------------------- | ------- |

| `header`  | Header container.                                  | `string \| JSX.CSSProperties` | `-`     |
| `title`   | Title element.                                     | `string \| JSX.CSSProperties` | `-`     |
| `body`    | Body container.                                    | `string \| JSX.CSSProperties` | `-`     |
| `footer`  | Footer container.                                  | `string \| JSX.CSSProperties` | `-`     |

### Footer render extra

| Property          | Description                      | Type                | Default |
| ----------------- | -------------------------------- | ------------------- | ------- |
| `originNode`      | The default footer node.         | `JSX.Element`       | `-`     |
| `extra.OkBtn`     | Default OK button component.     | `() => JSX.Element` | `-`     |

### Modal function return

| Property  | Description                                                                | Type                                         | Default |
| --------- | -------------------------------------------------------------------------- | -------------------------------------------- | ------- |
| `destroy` | Destroys the current static modal instance.                                | `() => void`                                 | `-`     |
| `update`  | Updates the current static modal config. Supports object or function form. | `(config \| (prevConfig) => config) => void` | `-`     |
