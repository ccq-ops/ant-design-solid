# Slider

### Basic

```tsx
import { createSignal } from 'solid-js'
import { Slider, Space } from '@ant-design-solid/core'

const Demo1 = function () {
  return <Slider defaultValue={30} />
}

export default Demo1
```

### Controlled

```tsx
import { createSignal } from 'solid-js'
import { Slider, Space } from '@ant-design-solid/core'

const Demo2 = function () {
  const [value, setValue] = createSignal(30)

  return (
    <Space direction="vertical" class="w-80">
      <Slider
        value={value()}
        onChange={(next) => {
          if (typeof next === 'number') setValue(next)
        }}
      />
      <span>Current value: {value()}</span>
    </Space>
  )
}

export default Demo2
```

### Marks

```tsx
import { createSignal } from 'solid-js'
import { Slider, Space } from '@ant-design-solid/core'

const Demo4 = function () {
  return (
    <Slider
      defaultValue={26}
      tooltipVisible
      marks={{
        0: '0°C',
        26: '26°C',
        37: { label: '37°C', style: { color: '#1677ff' } },
        100: '100°C',
      }}
    />
  )
}

export default Demo4
```

## API

### Slider

| Property         | Description                                             | Type                           | Default |
| ---------------- | ------------------------------------------------------- | ------------------------------ | ------- |
| `value`          | Controlled slider value. Tuple form is used for ranges. | `number \| [number, number]`   | `-`     |
| `defaultValue`   | Initial value for uncontrolled usage.                   | `number \| [number, number]`   | `-`     |
| `min`            | Minimum value.                                          | `number`                       | `0`     |
| `max`            | Maximum value.                                          | `number`                       | `100`   |
| `step`           | Value increment.                                        | `number`                       | `1`     |
| `disabled`       | Disables user interaction.                              | `boolean`                      | `false` |
| `range`          | Enables two-handle range selection.                     | `boolean`                      | `false` |
| `marks`          | Tick mark labels keyed by numeric value.                | `Record<number, SliderMark>`   | `-`     |
| `vertical`       | Renders the slider vertically.                          | `boolean`                      | `false` |
| `tooltipVisible` | Forces the value tooltip to be visible.                 | `boolean`                      | `-`     |
| `onChange`       | Called while value changes.                             | `(value: SliderValue) => void` | `-`     |
| `onAfterChange`  | Called after a committed change.                        | `(value: SliderValue) => void` | `-`     |

### SliderMark

| Property      | Description                             | Type                | Default |
| ------------- | --------------------------------------- | ------------------- | ------- |
| `JSX.Element` | Simple mark label.                      | `SliderMark`        | `-`     |
| `label`       | Mark label content for object marks.    | `JSX.Element`       | `-`     |
| `style`       | Inline style applied to the mark label. | `JSX.CSSProperties` | `-`     |
