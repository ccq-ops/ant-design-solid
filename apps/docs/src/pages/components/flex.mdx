# Flex

### Basic

```tsx
import { Button, Flex } from '@ant-design-solid/core'

const Demo1 = function () {
  return (
    <Flex gap="large" wrap>
      <Button>One</Button>
      <Button>Two</Button>
      <Button>Three</Button>
    </Flex>
  )
}

export default Demo1
```

### Vertical

```tsx
import { Button, Flex } from '@ant-design-solid/core'

const Demo2 = function () {
  return (
    <Flex vertical gap={8} align="flex-start">
      <Button>A</Button>
      <Button>B</Button>
    </Flex>
  )
}

export default Demo2
```

### Alignment

```tsx
import { Button, Flex } from '@ant-design-solid/core'

const Demo3 = function () {
  return (
    <Flex justify="space-between" align="center" class="w-full bg-gray-100 p-3">
      <span>Left</span>
      <Button type="primary">Right</Button>
    </Flex>
  )
}

export default Demo3
```

### Semantic element

```tsx
import { Button, Flex } from '@ant-design-solid/core'

const Demo4 = function () {
  return (
    <Flex component="section" gap={[8, 16]} wrap="wrap-reverse">
      <span>Section</span>
      <span>content</span>
    </Flex>
  )
}

export default Demo4
```

## API

| Property    | Description                                            | Type                                                                                            | Default    |
| ----------- | ------------------------------------------------------ | ----------------------------------------------------------------------------------------------- | ---------- |
| `vertical`  | Stacks children vertically.                            | `boolean`                                                                                       | `false`    |
| `wrap`      | Controls flex wrapping. Boolean true maps to wrap.     | `boolean \| 'nowrap' \| 'wrap' \| 'wrap-reverse'`                                               | `-`        |
| `justify`   | Maps to CSS justify-content.                           | `'flex-start' \| 'center' \| 'flex-end' \| 'space-between' \| 'space-around' \| 'space-evenly'` | `-`        |
| `align`     | Maps to CSS align-items.                               | `'flex-start' \| 'center' \| 'flex-end' \| 'stretch' \| 'baseline'`                             | `-`        |
| `gap`       | Horizontal and vertical gap. Presets use Space tokens. | `'small' \| 'middle' \| 'large' \| number \| [number, number]`                                  | `'middle'` |
| `component` | HTML element used as the wrapper.                      | `keyof JSX.IntrinsicElements`                                                                   | `'div'`    |
| `prefixCls` | Custom CSS class prefix.                               | `string`                                                                                        | `-`        |
