# Calendar

Display dates in a month or year panel with selectable and customizable cells.

### Basic

```tsx
import { createSignal } from 'solid-js'
import { Calendar, Space } from '@ant-design-solid/core'

const Demo1 = function () {
  return <Calendar />
}

export default Demo1
```

### Card

```tsx
import { createSignal } from 'solid-js'
import { Calendar, Space } from '@ant-design-solid/core'

const Demo2 = function () {
  return <Calendar fullscreen={false} defaultValue="2026-06-15" />
}

export default Demo2
```

### Custom date content

```tsx
import { createSignal } from 'solid-js'
import { Calendar, Space } from '@ant-design-solid/core'

const Demo3 = function () {
  return (
    <Calendar
      defaultValue="2026-06-15"
      dateCellRender={(date) => (date.getDate() === 15 ? <span>Release</span> : null)}
    />
  )
}

export default Demo3
```

  return (
    <Space direction="vertical">
      <Calendar

```

### Disabled dates

```

### Custom header

```tsx
import { createSignal } from 'solid-js'
import { Calendar, Space } from '@ant-design-solid/core'

function sameDay(a, b) {
  return (
    a.getFullYear() === b.getFullYear() &&
    a.getMonth() === b.getMonth() &&
    a.getDate() === b.getDate()
  )
}

const Demo6 = function () {
  return (
    <Calendar
      fullscreen={false}
      defaultValue="2026-06-15"
      headerRender={({ value, onChange, onModeChange }) => (
        <Space>
          <button
            type="button"
            onClick={() => onChange(new Date(value.getFullYear(), value.getMonth() - 1, 1))}
          >
            Previous
          </button>
          <button type="button" onClick={() => onModeChange('year')}>
            Year view
          </button>
        </Space>
      )}
      dateCellRender={(date) =>
        sameDay(date, new Date(2026, 5, 15)) ? <span>Selected</span> : null
      }
    />
  )
}

export default Demo6
```

## API

| Property              | Description                                               | Type                                                  | Default   |
| --------------------- | --------------------------------------------------------- | ----------------------------------------------------- | --------- |
| `value`               | Controlled selected date.                                 | `Date \| string`                                      | `-`       |
| `defaultValue`        | Initial selected date for uncontrolled usage.             | `Date \| string`                                      | `-`       |
| `mode`                | Controlled panel mode.                                    | `'month' \| 'year'`                                   | `'month'` |
| `defaultMode`         | Initial panel mode for uncontrolled usage.                | `'month' \| 'year'`                                   | `'month'` |
| `fullscreen`          | Whether to render full-size calendar layout.              | `boolean`                                             | `true`    |
| `disabledDate`        | Disables date or month cells.                             | `(date: Date) => boolean`                             | `-`       |
| `dateCellRender`      | Additional content renderer for date cells.               | `(date: Date) => JSX.Element`                         | `-`       |
| `dateFullCellRender`  | Full custom renderer for date cells.                      | `(date: Date) => JSX.Element`                         | `-`       |
| `monthCellRender`     | Additional content renderer for month cells in year mode. | `(date: Date) => JSX.Element`                         | `-`       |
| `monthFullCellRender` | Full custom renderer for month cells in year mode.        | `(date: Date) => JSX.Element`                         | `-`       |
| `headerRender`        | Custom header renderer with panel controls.               | `(config: CalendarHeaderRenderConfig) => JSX.Element` | `-`       |
| `onSelect`            | Called whenever a selectable date or month is selected.   | `(date: Date) => void`                                | `-`       |
| `onChange`            | Called when selected value changes.                       | `(date: Date) => void`                                | `-`       |
| `onPanelChange`       | Called when panel date or mode changes.                   | `(date: Date, mode: CalendarMode) => void`            | `-`       |
| `prefixCls`           | Custom CSS class prefix.                                  | `string`                                              | `-`       |
