# ColorPicker

Provide color selection with presets, alpha, formats, and controlled state.

### Basic

```tsx
import { createSignal } from 'solid-js'
import { ColorPicker, Space } from '@ant-design-solid/core'

const Demo1 = function () {
  return <ColorPicker defaultValue="#1677ff" />
}

export default Demo1
```

### Controlled

```tsx
import { createSignal } from 'solid-js'
import { ColorPicker, Space } from '@ant-design-solid/core'

const Demo2 = function () {
  const [value, setValue] = createSignal('#1677ff')

  return (
    <Space direction="vertical">
      <ColorPicker
        value={value()}
        showText
        onChange={(color) => setValue(color?.toHexString() ?? '')}
      />
      <span>Current value: {value()}</span>
    </Space>
  )
}

export default Demo2
```

### Controlled open

```tsx
import { createSignal } from 'solid-js'
import { ColorPicker, Space } from '@ant-design-solid/core'

const Demo8 = function () {
  const [open, setOpen] = createSignal(false)

  return (
    <Space>
      <ColorPicker open={open()} onOpenChange={setOpen} defaultValue="#1677ff" />
      <span>Open: {String(open())}</span>
    </Space>
  )
}

export default Demo8
```

### Custom panel render

```tsx
import { createSignal } from 'solid-js'
import { ColorPicker, Space } from '@ant-design-solid/core'

const Demo9 = function () {
  return (
    <ColorPicker
      panelRender={(panel) => (
        <div
          style={{
            border: '1px solid #1677ff',
            padding: '8px',
            'border-radius': '8px',
          }}
        >
          {panel}
        </div>
      )}
    />
  )
}

export default Demo9
```

## API

### ColorPicker

| Property            | Description                                                     | Type                                                                      | Default               |
| ------------------- | --------------------------------------------------------------- | ------------------------------------------------------------------------- | --------------------- |
| `value`             | Controlled color value.                                         | `ColorPickerValue`                                                        | `-`                   |
| `defaultValue`      | Initial color value for uncontrolled usage.                     | `ColorPickerValue`                                                        | `-`                   |
| `onChange`          | Called when the color changes with Color object and hex string. | `(value: Color \| undefined, hex: string) => void`                        | `-`                   |
| `onChangeComplete`  | Called after a committed color change.                          | `(value: Color \| undefined) => void`                                     | `-`                   |
| `open`              | Controlled popup open state.                                    | `boolean`                                                                 | `-`                   |
| `defaultOpen`       | Initial popup open state.                                       | `boolean`                                                                 | `false`               |
| `onOpenChange`      | Called when popup open state changes.                           | `(open: boolean) => void`                                                 | `-`                   |
| `disabled`          | Disables the picker.                                            | `boolean`                                                                 | `false`               |
| `size`              | Picker size.                                                    | `'small' \| 'middle' \| 'large'`                                          | `ConfigProvider size` |
| `placement`         | Popup placement.                                                | `DropdownPlacement`                                                       | `'bottomLeft'`        |
| `trigger`           | Interaction that opens the popup.                               | `'click' \| 'hover'`                                                      | `'click'`             |
| `format`            | Controlled color input format.                                  | `'hex' \| 'rgb' \| 'hsb'`                                                 | `'hex'`               |
| `defaultFormat`     | Initial color input format.                                     | `'hex' \| 'rgb' \| 'hsb'`                                                 | `'hex'`               |
| `disabledAlpha`     | Disables alpha editing.                                         | `boolean`                                                                 | `false`               |
| `allowClear`        | Shows a clear button.                                           | `boolean`                                                                 | `false`               |
| `showText`          | Shows selected color text or custom text renderer.              | `boolean \| ((color: Color \| undefined) => JSX.Element)`                 | `-`                   |
| `presets`           | Preset color groups.                                            | `ColorPickerPreset[]`                                                     | `-`                   |
| `panelRender`       | Custom panel renderer.                                          | `(panel: JSX.Element, extra: ColorPickerPanelRenderExtra) => JSX.Element` | `-`                   |
| `popupClass`        | Additional popup class.                                         | `string`                                                                  | `-`                   |
| `popupStyle`        | Popup inline style.                                             | `JSX.CSSProperties`                                                       | `-`                   |
| `zIndex`            | Custom popup z-index.                                           | `number`                                                                  | `-`                   |
| `getPopupContainer` | Returns popup portal container.                                 | `(triggerNode?: HTMLElement) => HTMLElement`                              | `-`                   |

### ColorPickerPreset

| Property | Description          | Type          | Default |
| -------- | -------------------- | ------------- | ------- |
| `label`  | Preset group label.  | `JSX.Element` | `-`     |
| `colors` | Preset color values. | `string[]`    | `-`     |
