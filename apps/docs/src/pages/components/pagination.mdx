# Pagination

### Basic

```tsx
import { createSignal } from 'solid-js'
import { Pagination, Space } from '@ant-design-solid/core'

const Demo1 = function () {
  return <Pagination total={50} />
}

export default Demo1
```

### More pages

```tsx
import { createSignal } from 'solid-js'

  return <Pagination total={500} defaultCurrent={6} />

```

### Controlled

```tsx
import { createSignal } from 'solid-js'
import { Pagination, Space } from '@ant-design-solid/core'

const Demo3 = function () {
  const [current, setCurrent] = createSignal(2)

  return (
    <Space direction="vertical">
      <Pagination current={current()} total={80} onChange={setCurrent} />
      <span>Current page: {current()}</span>
    </Space>
  )
}

export default Demo3
```

### Size changer


### Custom item render

```tsx
import { createSignal } from 'solid-js'
import { Pagination, Space } from '@ant-design-solid/core'

const Demo11 = function () {
  return (
    <Pagination
      total={120}
      itemRender={(page, type, originalElement) => {
        if (type === 'prev') return <span>Previous</span>
        if (type === 'next') return <span>Next</span>
        return <strong>{page}</strong>
      }}
    />
  )
}

export default Demo11
```

## API

| Property                       | Description                                                                | Type                                                                                            | Default             |
| ------------------------------ | -------------------------------------------------------------------------- | ----------------------------------------------------------------------------------------------- | ------------------- |
| `align`                        | Aligns pagination items within the container.                              | `'start' \| 'center' \| 'end'`                                                                  | `-`                 |
| `class`                        | Additional class name for the root pagination element.                     | `string`                                                                                        | `-`                 |
| `classNames`                   | Semantic class names for internal slots.                                   | `Partial<Record<PaginationSemanticSlot, string>>`                                               | `-`                 |
| `current`                      | Controlled current page.                                                   | `number`                                                                                        | `-`                 |
| `defaultCurrent`               | Initial current page for uncontrolled usage.                               | `number`                                                                                        | `1`                 |
| `defaultPageSize`              | Initial items per page for uncontrolled usage.                             | `number`                                                                                        | `10`                |
| `disabled`                     | Disables pagination controls.                                              | `boolean`                                                                                       | `false`             |
| `hideOnSinglePage`             | Hides pagination when total fits on one page.                              | `boolean`                                                                                       | `false`             |
| `itemRender`                   | Customizes page, previous, and next item content.                          | `(page: number, type: 'page' \| 'prev' \| 'next', originalElement: JSX.Element) => JSX.Element` | `-`                 |
| `pageSize`                     | Controlled items per page.                                                 | `number`                                                                                        | `-`                 |
| `pageSizeOptions`              | Options for page size selection.                                           | `(string \| number)[]`                                                                          | `[10, 20, 50, 100]` |
| `responsive`                   | Adds a responsive class for responsive pagination styling.                 | `boolean`                                                                                       | `false`             |
| `showLessItems`                | Shows fewer page items around the current page.                            | `boolean`                                                                                       | `false`             |
| `showQuickJumper`              | Shows quick jump input. Object form supports a custom goButton.            | `boolean \| { goButton?: JSX.Element }`                                                         | `false`             |
| `showSizeChanger`              | Shows page size selection. Object form is passed to the internal select.   | `boolean \| SelectHTMLAttributes<HTMLSelectElement>`                                            | `-`                 |
| `showTitle`                    | Shows title attributes on page controls.                                   | `boolean`                                                                                       | `true`              |
| `showTotal`                    | Renders custom total text.                                                 | `(total: number, range: [number, number]) => JSX.Element`                                       | `-`                 |
| `simple`                       | Uses the compact simple layout, optionally with a read-only input.         | `boolean \| { readOnly?: boolean }`                                                             | `false`             |
| `size`                         | Controls pagination size.                                                  | `'large' \| 'medium' \| 'small'`                                                                | `'medium'`          |
| `style`                        | Inline style for the root pagination element.                              | `JSX.CSSProperties`                                                                             | `-`                 |
| `styles`                       | Semantic inline styles for internal slots.                                 | `Partial<Record<PaginationSemanticSlot, JSX.CSSProperties>>`                                    | `-`                 |
| `total`                        | Total number of items.                                                     | `number`                                                                                        | `0`                 |
| `totalBoundaryShowSizeChanger` | Shows the size changer automatically when total is larger than this value. | `number`                                                                                        | `50`                |
| `onChange`                     | Called when page or page size changes.                                     | `(page: number, pageSize: number) => void`                                                      | `-`                 |
| `onShowSizeChange`             | Called when page size changes.                                             | `(current: number, size: number) => void`                                                       | `-`                 |

## Semantic DOM

| Property      | Description                       | Type                   | Default |
| ------------- | --------------------------------- | ---------------------- | ------- |
| `root`        | Root nav element.                 | `classNames \| styles` | `-`     |
| `list`        | The page item list.               | `classNames \| styles` | `-`     |
| `item`        | Each page list item.              | `classNames \| styles` | `-`     |
| `itemButton`  | Page, previous, and next buttons. | `classNames \| styles` | `-`     |
| `prev`        | Previous item wrapper.            | `classNames \| styles` | `-`     |
| `next`        | Next item wrapper.                | `classNames \| styles` | `-`     |
| `ellipsis`    | Ellipsis item.                    | `classNames \| styles` | `-`     |
| `simplePager` | Simple mode input wrapper.        | `classNames \| styles` | `-`     |
| `input`       | Simple and quick jumper input.    | `classNames \| styles` | `-`     |
| `select`      | Page size select.                 | `classNames \| styles` | `-`     |
| `quickJumper` | Quick jumper wrapper.             | `classNames \| styles` | `-`     |
| `totalText`   | Total text wrapper.               | `classNames \| styles` | `-`     |
| `goButton`    | Quick jumper go button wrapper.   | `classNames \| styles` | `-`     |
