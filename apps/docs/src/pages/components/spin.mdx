# Spin

Spin displays a loading indicator for pages, sections, or inline content.

### Basic

```tsx
import { Button, Space, Spin } from '@ant-design-solid/core'
import { createEffect, createSignal, onCleanup } from 'solid-js'

const Demo1 = function () {
  return <Spin />
}

export default Demo1
```

### Size

```tsx
import { Button, Space, Spin } from '@ant-design-solid/core'
import { createEffect, createSignal, onCleanup } from 'solid-js'

const Demo2 = function () {
  return (
    <Space align="center">
      <Spin size="small" />
      <Spin />
      <Spin size="large" />
    </Space>
  )
}

export default Demo2
```

### Nested

```tsx
import { Button, Space, Spin } from '@ant-design-solid/core'
import { createEffect, createSignal, onCleanup } from 'solid-js'

const Demo4 = function () {
  return (
    <Spin spinning tip="Loading content">
      <div
        style={{
          padding: '24px',
          border: '1px solid var(--docs-border)',
          'border-radius': '8px',
          background: 'var(--docs-surface-solid)',
        }}
      >
        Content remains mounted while the spinner overlays it.
      </div>
    </Spin>
  )
}

export default Demo4
```

### Delay

```tsx
import { Button, Space, Spin } from '@ant-design-solid/core'
import { createEffect, createSignal, onCleanup } from 'solid-js'

const Demo5 = function () {
  return (
    <Spin spinning delay={500} tip="Loading after delay">
      <div
        style={{
          padding: '24px',
          border: '1px solid var(--docs-border)',
          'border-radius': '8px',
          background: 'var(--docs-surface-solid)',
        }}
      >
        The indicator appears only if loading lasts at least 500ms.
      </div>
    </Spin>
  )
}

export default Demo5
```

### Fullscreen


## API

| Property     | Description                                       | Type                              | Default     |
| ------------ | ------------------------------------------------- | --------------------------------- | ----------- |
| `spinning`   | Controls whether the spinner is active.           | `boolean`                         | `true`      |
| `size`       | Spinner size.                                     | `'small' \| 'default' \| 'large'` | `'default'` |
| `tip`        | Loading text displayed with the spinner.          | `JSX.Element`                     | `-`         |
| `delay`      | Delay in milliseconds before showing the spinner. | `number`                          | `-`         |
| `fullscreen` | Renders the spinner as a fullscreen overlay.      | `boolean`                         | `false`     |
| `indicator`  | Custom indicator content.                         | `JSX.Element`                     | `-`         |
| `children`   | Nested content covered by loading overlay.        | `JSX.Element`                     | `-`         |
