# Steps

### Basic

```tsx
import { createSignal } from 'solid-js'
import { Space, Steps } from '@ant-design-solid/core'

const Demo1 = function () {
  return (
    <Steps
      current={1}
      items={[{ title: 'Finished' }, { title: 'In Progress' }, { title: 'Waiting' }]}
    />
  )
}

export default Demo1
```

### Navigation

```tsx
import { createSignal } from 'solid-js'
import { Space, Steps } from '@ant-design-solid/core'

const Demo5 = function () {
  const [current, setCurrent] = createSignal(0)

  return (
    <Space direction="vertical">
      <Steps
        type="navigation"
        current={current()}
        onChange={setCurrent}
        items={[
          { title: 'Start' },
          { title: 'Review' },
          { title: 'Disabled', disabled: true },
          { title: 'Done' },
        ]}
      />
      <span>Current step: {current() + 1}</span>
    </Space>
  )
}

export default Demo5
```

## API

### Steps

| Property    | Description                                         | Type                                         | Default        |
| ----------- | --------------------------------------------------- | -------------------------------------------- | -------------- |
| `items`     | Step items rendered in order.                       | `StepItem[]`                                 | `-`            |
| `current`   | Current step index.                                 | `number`                                     | `0`            |
| `status`    | Status for the current step.                        | `'wait' \| 'process' \| 'finish' \| 'error'` | `-`            |
| `direction` | Step layout direction.                              | `'horizontal' \| 'vertical'`                 | `'horizontal'` |
| `size`      | Step size.                                          | `'default' \| 'small'`                       | `'default'`    |
| `type`      | Step style type.                                    | `'default' \| 'navigation'`                  | `'default'`    |
| `onChange`  | Called when a clickable step changes current index. | `(current: number) => void`                  | `-`            |

### StepItem

| Property      | Description                       | Type                                         | Default |
| ------------- | --------------------------------- | -------------------------------------------- | ------- |
| `title`       | Step title.                       | `JSX.Element`                                | `-`     |
| `description` | Step description.                 | `JSX.Element`                                | `-`     |
| `status`      | Per-step status override.         | `'wait' \| 'process' \| 'finish' \| 'error'` | `-`     |
| `icon`        | Custom step icon.                 | `JSX.Element`                                | `-`     |
| `disabled`    | Disables navigation to this step. | `boolean`                                    | `-`     |
