# Segmented

### Basic

```tsx
import { Segmented, Space } from '@ant-design-solid/core'

const Demo1 = function () {
  return <Segmented options={['Daily', 'Weekly', 'Monthly']} />
}

export default Demo1
```

### Object options

```tsx
import { Segmented, Space } from '@ant-design-solid/core'

const Demo2 = function () {
  return (
    <Segmented
      options={[
        { label: 'List', value: 'list', icon: '☰' },
        { label: 'Grid', value: 'grid', icon: '▦' },
        { label: 'Disabled', value: 'disabled', disabled: true },
      ]}
      defaultValue="grid"
    />
  )
}

export default Demo2
```

### Sizes

```tsx
import { Segmented, Space } from '@ant-design-solid/core'

const Demo3 = function () {
  return (
    <Space direction="vertical">
      <Segmented size="small" options={['Small', 'Middle']} />
      <Segmented options={['Middle', 'Large']} />
      <Segmented size="large" options={['Large', 'Huge']} />
    </Space>
  )
}

export default Demo3
```

### Block and disabled

```tsx
import { Segmented, Space } from '@ant-design-solid/core'
```


### Tooltips

### Semantic styles

```tsx
import { Segmented } from '@ant-design-solid/core'

const Demo7 = function () {
  return (
    <Segmented
      options={['Open', 'Closed', 'Archived']}
      rootClass="demo-segmented-root"
      classNames={{ item: 'demo-segmented-item', label: 'demo-segmented-label' }}
      styles={{ root: { maxWidth: '360px' }, item: { minWidth: '96px' } }}
    />
  )
}

export default Demo7
```

## API

### Segmented

| Property       | Description                                                                         | Type                                                                   | Default        |
| -------------- | ----------------------------------------------------------------------------------- | ---------------------------------------------------------------------- | -------------- |
| `options`      | Options rendered by the segmented control.                                          | `SegmentedOption[]`                                                    | `[]`           |
| `value`        | Controlled selected value.                                                          | `string \| number`                                                     | `-`            |
| `defaultValue` | Initial selected value for uncontrolled usage.                                      | `string \| number`                                                     | `-`            |
| `disabled`     | Disables the whole control.                                                         | `boolean`                                                              | `false`        |
| `block`        | Makes the segmented control fill its container width.                               | `boolean`                                                              | `false`        |
| `size`         | Control size. `medium` is the antd v6 spelling; `middle` is kept for compatibility. | `'small' \| 'medium' \| 'middle' \| 'large'`                           | `-`            |
| `orientation`  | Arrangement direction. Takes priority over `vertical`.                              | `'horizontal' \| 'vertical'`                                           | `'horizontal'` |
| `vertical`     | Vertical arrangement shorthand.                                                     | `boolean`                                                              | `false`        |
| `shape`        | Segmented shape.                                                                    | `'default' \| 'round'`                                                 | `'default'`    |
| `name`         | Name attribute applied to each radio-like item.                                     | `string`                                                               | `-`            |
| `prefixCls`    | Custom CSS class prefix.                                                            | `string`                                                               | `-`            |
| `rootClass`    | Additional class on the root element. Solid alias for antd `rootClassName`.         | `string`                                                               | `-`            |
| `classNames`   | Semantic classes for internal slots. Supports object or function.                   | `SegmentedSemanticClassNames \| (info) => SegmentedSemanticClassNames` | `-`            |
| `styles`       | Semantic inline styles for internal slots. Supports object or function.             | `SegmentedSemanticStyles \| (info) => SegmentedSemanticStyles`         | `-`            |
| `onChange`     | Called with the selected value after user interaction.                              | `(value: SegmentedValue) => void`                                      | `-`            |

### SegmentedOption

| Property           | Description                                                               | Type                                       | Default |
| ------------------ | ------------------------------------------------------------------------- | ------------------------------------------ | ------- |
| `string \| number` | Primitive option used as both label and value.                            | `SegmentedOption`                          | `-`     |
| `label`            | Option label content for object options. Optional when `icon` is present. | `JSX.Element`                              | `-`     |
| `value`            | Option value.                                                             | `string \| number`                         | `-`     |
| `disabled`         | Disables this option.                                                     | `boolean`                                  | `-`     |
| `icon`             | Icon rendered before the label.                                           | `JSX.Element`                              | `-`     |
| `class`            | Additional class for this option.                                         | `string`                                   | `-`     |
| `tooltip`          | Tooltip for this option.                                                  | `string \| Omit<TooltipProps, 'children'>` | `-`     |

### Semantic DOM

`classNames` and `styles` support these slots: `root`, `item`, `icon`, `label`.
