# DatePicker

Select dates, date ranges, and date-time values. Like antd, DatePicker values are `dayjs` objects rather than strings.

### Basic dayjs value

```tsx
import dayjs from 'dayjs'
import { createSignal } from 'solid-js'
import {
  CalendarOutlined,
  ClockCircleOutlined,
  DoubleLeftOutlined,
  DoubleRightOutlined,
  LeftOutlined,
  RightOutlined,
} from '@ant-design-solid/icons'
import { DatePicker, Space } from '@ant-design-solid/core'

const Demo1 = function () {
  return <DatePicker defaultValue={dayjs('2026-06-15')} />
}

export default Demo1
```

### Controlled open and value

```tsx
import dayjs from 'dayjs'
import { createSignal } from 'solid-js'
import {
  CalendarOutlined,
  ClockCircleOutlined,
  DoubleLeftOutlined,
  DoubleRightOutlined,
  LeftOutlined,
  RightOutlined,
} from '@ant-design-solid/icons'
import { DatePicker, Space } from '@ant-design-solid/core'

const Demo2 = function () {
  const [value, setValue] = createSignal(dayjs('2026-06-15'))

  const [open, setOpen] = createSignal(false)

  const selectedLabel = () => value()?.format('YYYY-MM-DD') ?? 'none'

  return (
    <Space direction="vertical">
      <DatePicker value={value()} open={open()} onChange={setValue} onOpenChange={setOpen} />
      <span>Selected date: {selectedLabel()}</span>
    </Space>
  )
}

export default Demo2
```

### Need confirm and onOk

```tsx
import dayjs from 'dayjs'
import { createSignal } from 'solid-js'
import {
  CalendarOutlined,
  ClockCircleOutlined,
  DoubleLeftOutlined,
  DoubleRightOutlined,
  LeftOutlined,
  RightOutlined,
} from '@ant-design-solid/icons'
import { DatePicker, Space } from '@ant-design-solid/core'

const Demo11 = function () {
  const [confirmedValue, setConfirmedValue] = createSignal(dayjs('2026-06-15 09:30'))

  const [confirmedLabel, setConfirmedLabel] = createSignal('2026-06-15 09:30:00')

  return (
    <Space direction="vertical">
      <DatePicker
        showTime
        needConfirm
        value={confirmedValue()}
        onChange={setConfirmedValue}
        onOk={(date) => setConfirmedLabel(date?.format('YYYY-MM-DD HH:mm:ss') ?? 'none')}
      />
      <span>Confirmed value: {confirmedLabel()}</span>
    </Space>
  )
}

export default Demo11
```

## API

DatePicker and RangePicker values use `dayjs`. The `dateString` callback argument is only the formatted string representation of those dayjs values.

| Property                | Description                                                                     | Type                                                                                                           | Default         |
| ----------------------- | ------------------------------------------------------------------------------- | -------------------------------------------------------------------------------------------------------------- | --------------- |
| `value`                 | Controlled selected value. DatePicker values use dayjs objects.                 | `Dayjs \| null; Dayjs[] when multiple`                                                                         | `-`             |
| `defaultValue`          | Initial selected value for uncontrolled usage. Values use dayjs objects.        | `Dayjs \| null; Dayjs[] when multiple`                                                                         | `-`             |
| `format`                | Display and emitted date string format. Formatting is applied to dayjs values.  | `string \| string[] \| ((value: Dayjs) => string) \| { format: string; type?: 'mask' }`                        | `'YYYY-MM-DD'`  |
| `picker`                | Panel granularity for the dayjs value selection.                                | `'date' \| 'week' \| 'month' \| 'quarter' \| 'year' \| 'time'`                                                 | `'date'`        |
| `multiple`              | Allows selecting multiple dayjs values in one DatePicker.                       | `boolean`                                                                                                      | `false`         |
| `showWeek`              | Shows the Week header and week-number column in the date panel.                 | `boolean`                                                                                                      | `false`         |
| `needConfirm`           | Requires pressing OK before committing the selected dayjs value.                | `boolean`                                                                                                      | `false`         |
| `placeholder`           | Placeholder text when no date is selected.                                      | `string`                                                                                                       | `'Select date'` |
| `disabled`              | Disables the picker.                                                            | `boolean`                                                                                                      | `false`         |
| `allowClear`            | Shows a clear button when a value is selected.                                  | `boolean \| { clearIcon?: JSX.Element }`                                                                       | `false`         |
| `variant`               | Visual variant of the selector.                                                 | `'outlined' \| 'borderless' \| 'filled' \| 'underlined'`                                                       | `'outlined'`    |
| `status`                | Validation status styling.                                                      | `'' \| 'error' \| 'warning'`                                                                                   | `-`             |
| `size`                  | Selector size.                                                                  | `'small' \| 'middle' \| 'large'`                                                                               | `'middle'`      |
| `classNames`            | Semantic slot class names.                                                      | `Partial<Record<DatePickerSemanticSlot, string>>`                                                              | `-`             |
| `styles`                | Semantic slot inline styles.                                                    | `Partial<Record<DatePickerSemanticSlot, JSX.CSSProperties>>`                                                   | `-`             |
| `prefix`                | Custom prefix content inside the selector.                                      | `JSX.Element`                                                                                                  | `-`             |
| `suffixIcon`            | Custom suffix icon inside the selector.                                         | `JSX.Element`                                                                                                  | `-`             |
| `prevIcon`              | Custom previous navigation icon.                                                | `JSX.Element`                                                                                                  | `-`             |
| `nextIcon`              | Custom next navigation icon.                                                    | `JSX.Element`                                                                                                  | `-`             |
| `superPrevIcon`         | Custom super previous navigation icon.                                          | `JSX.Element`                                                                                                  | `-`             |
| `superNextIcon`         | Custom super next navigation icon.                                              | `JSX.Element`                                                                                                  | `-`             |
| `components`            | Overrides supported internal slots such as input and panel.                     | `Partial<Record<'input' \| 'panel' \| PickerMode, Component>>`                                                 | `-`             |
| `previewValue`          | Controlled preview value for hover/range preview parity.                        | `Dayjs \| [Dayjs \| null, Dayjs \| null] \| null`                                                              | `-`             |
| `open`                  | Controlled popup open state.                                                    | `boolean`                                                                                                      | `-`             |
| `defaultOpen`           | Initial popup open state.                                                       | `boolean`                                                                                                      | `false`         |
| `pickerValue`           | Controlled panel date. The panel value uses dayjs.                              | `Dayjs`                                                                                                        | `-`             |
| `defaultPickerValue`    | Initial panel date for uncontrolled usage. The panel value uses dayjs.          | `Dayjs`                                                                                                        | `-`             |
| `disabledDate`          | Disables dayjs date cells.                                                      | `(current: Dayjs, info: { type: PickerType }) => boolean`                                                      | `-`             |
| `disabledTime`          | Disables time options for a selected dayjs value.                               | `(date: Dayjs \| null) => DisabledTimeConfig`                                                                  | `-`             |
| `showTime`              | Adds a time panel and keeps the value as a dayjs object.                        | `boolean \| ShowTimeOptions`                                                                                   | `false`         |
| `showNow`               | Shows the Now shortcut when time selection is enabled.                          | `boolean`                                                                                                      | `false`         |
| `presets`               | Quick selections. Preset values are dayjs objects or callbacks returning dayjs. | `Array<{ label: JSX.Element; value: Dayjs \| (() => Dayjs) }>`                                                 | `-`             |
| `cellRender`            | Custom cell rendering for dayjs panel values.                                   | `(current: Dayjs, info: CellRenderInfo) => JSX.Element`                                                        | `-`             |
| `dateRender`            | Legacy custom date cell rendering for dayjs values.                             | `(current: Dayjs, today: Dayjs) => JSX.Element`                                                                | `-`             |
| `renderExtraFooter`     | Renders extra content in the popup footer.                                      | `(mode: PickerMode) => JSX.Element`                                                                            | `-`             |
| `panelRender`           | Wraps or replaces the generated picker panel.                                   | `(panel: JSX.Element) => JSX.Element`                                                                          | `-`             |
| `onChange`              | Called when selection changes. The value argument uses dayjs.                   | `(value: Dayjs \| null, dateString: string) => void; multiple: (value: Dayjs[], dateString: string[]) => void` | `-`             |
| `onSelect`              | Called when a dayjs cell value is selected.                                     | `(value: Dayjs) => void`                                                                                       | `-`             |
| `onOpenChange`          | Called when popup open state changes.                                           | `(open: boolean) => void`                                                                                      | `-`             |
| `onOk`                  | Called when an OK-confirmed dayjs value is accepted.                            | `(value?: Dayjs \| null \| Dayjs[]) => void`                                                                   | `-`             |
| `prefixCls`             | Custom CSS class prefix.                                                        | `string`                                                                                                       | `-`             |
| `zIndex`                | Custom popup z-index.                                                           | `number`                                                                                                       | `-`             |
| `getPopupContainer`     | Returns popup portal container.                                                 | `(triggerNode?: HTMLElement) => HTMLElement`                                                                   | `-`             |
| `mode`                  | Controlled panel mode.                                                          | `PickerMode`                                                                                                   | `-`             |
| `minDate`               | Minimum selectable and navigable dayjs date.                                    | `Dayjs`                                                                                                        | `-`             |
| `maxDate`               | Maximum selectable and navigable dayjs date.                                    | `Dayjs`                                                                                                        | `-`             |
| `placement`             | Popup placement relative to the selector.                                       | `'bottomLeft' \| 'bottomRight' \| 'topLeft' \| 'topRight'`                                                     | `'bottomLeft'`  |
| `autoFocus`             | Focuses the input on mount.                                                     | `boolean`                                                                                                      | `-`             |
| `inputReadOnly`         | Sets the text input to read-only.                                               | `boolean`                                                                                                      | `-`             |
| `preserveInvalidOnBlur` | Keeps invalid typed text when the input blurs.                                  | `boolean`                                                                                                      | `-`             |
| `popupClassName`        | Custom popup class name.                                                        | `string`                                                                                                       | `-`             |
| `dropdownClassName`     | Alias for popupClassName.                                                       | `string`                                                                                                       | `-`             |
| `popupStyle`            | Custom popup inline style.                                                      | `JSX.CSSProperties`                                                                                            | `-`             |
| `bordered`              | Legacy border switch. false maps to borderless.                                 | `boolean`                                                                                                      | `-`             |
| `order`                 | Sorts multiple values before emitting them.                                     | `boolean`                                                                                                      | `true`          |
| `tagRender`             | Custom tag renderer when multiple is enabled.                                   | `(props: TagRenderProps) => JSX.Element`                                                                       | `-`             |
| `onPanelChange`         | Called when panel date or mode changes.                                         | `(value: Dayjs, mode: PickerMode) => void`                                                                     | `-`             |
| `onFocus`               | Called when the input receives focus.                                           | `(event: FocusEvent) => void`                                                                                  | `-`             |
| `onBlur`                | Called when the input loses focus.                                              | `(event: FocusEvent) => void`                                                                                  | `-`             |
| `onKeyDown`             | Called for input keyboard events.                                               | `(event: KeyboardEvent) => void`                                                                               | `-`             |

## RangePicker API

| Property             | Description                                                                     | Type                                                                                                                      | Default                      |
| -------------------- | ------------------------------------------------------------------------------- | ------------------------------------------------------------------------------------------------------------------------- | ---------------------------- |
| `value`              | Controlled range value. RangePicker values use dayjs objects.                   | `[Dayjs \| null, Dayjs \| null] \| null`                                                                                  | `-`                          |
| `defaultValue`       | Initial range value for uncontrolled usage. Values use dayjs objects.           | `[Dayjs \| null, Dayjs \| null] \| null`                                                                                  | `-`                          |
| `placeholder`        | Placeholder text for the start and end inputs.                                  | `[string, string]`                                                                                                        | `['Start date', 'End date']` |
| `disabled`           | Disables both inputs or an individual range side.                               | `boolean \| [boolean, boolean]`                                                                                           | `false`                      |
| `allowEmpty`         | Allows clearing the start or end dayjs value independently.                     | `[boolean, boolean]`                                                                                                      | `-`                          |
| `showTime`           | Adds a time panel and keeps range values as dayjs objects.                      | `boolean \| RangeShowTimeOptions`                                                                                         | `false`                      |
| `presets`            | Quick range selections. Preset endpoints are dayjs objects, null, or callbacks. | `Array<{ label: JSX.Element; value: [Dayjs \| (() => Dayjs) \| null, Dayjs \| (() => Dayjs) \| null] \| (() => tuple) }>` | `-`                          |
| `onChange`           | Called when the selected range changes. The dates argument uses dayjs values.   | `(dates: [Dayjs \| null, Dayjs \| null] \| null, dateStrings: [string, string]) => void`                                  | `-`                          |
| `onCalendarChange`   | Called while selecting range endpoints. The dates argument uses dayjs values.   | `(dates, dateStrings, info: { range: 'start' \| 'end' }) => void`                                                         | `-`                          |
| `id`                 | Input id for start and end fields.                                              | `string \| [string, string]`                                                                                              | `-`                          |
| `defaultPickerValue` | Initial panel date for uncontrolled usage.                                      | `Dayjs \| [Dayjs, Dayjs]`                                                                                                 | `-`                          |
| `pickerValue`        | Controlled panel date for one or both range panels.                             | `Dayjs \| [Dayjs, Dayjs]`                                                                                                 | `-`                          |
| `disabledTime`       | Disables time options for the active range endpoint.                            | `(date: Dayjs \| null) => DisabledTimeConfig`                                                                             | `-`                          |
| `order`              | Orders start and end values before emitting them.                               | `boolean`                                                                                                                 | `true`                       |
| `separator`          | Custom separator content between start and end inputs.                          | `JSX.Element`                                                                                                             | `-`                          |
| `onOk`               | Called when an OK-confirmed range value is accepted.                            | `(dates: [Dayjs \| null, Dayjs \| null] \| null) => void`                                                                 | `-`                          |
