# TreeSelect

Select a value from hierarchical tree data.

### Basic

```tsx
import { createSignal } from 'solid-js'
import { TreeSelect } from '@ant-design-solid/core'

const treeData = [
  {
    title: 'Asia',
    value: 'asia',
    children: [
      { title: 'China', value: 'china' },
      { title: 'Japan', value: 'japan', disabled: true },
    ],
  },
  { title: 'Europe', value: 'europe' },
]

const Demo1 = function () {
  return <TreeSelect placeholder="Select area" treeData={treeData} />
}

export default Demo1
```

### Controlled

```tsx
import { createSignal } from 'solid-js'
import { TreeSelect } from '@ant-design-solid/core'

const treeData = [
  {
    title: 'Asia',
    value: 'asia',
    children: [
      { title: 'China', value: 'china' },
      { title: 'Japan', value: 'japan', disabled: true },
    ],
  },
  { title: 'Europe', value: 'europe' },
]

const Demo4 = function () {
  const [value, setValue] = createSignal('europe')

  return <TreeSelect value={value()} onChange={(next) => setValue(next)} treeData={treeData} />
}

export default Demo4
```

## API

### TreeSelect

| Property              | Description                                    | Type                                                                           | Default |
| --------------------- | ---------------------------------------------- | ------------------------------------------------------------------------------ | ------- |
| `treeData`            | Hierarchical option nodes.                     | `TreeSelectNode[]`                                                             | `-`     |
| `value`               | Controlled selected value.                     | `OptionValue`                                                                  | `-`     |
| `defaultValue`        | Initial selected value for uncontrolled usage. | `OptionValue`                                                                  | `-`     |
| `open`                | Controlled dropdown open state.                | `boolean`                                                                      | `-`     |
| `defaultOpen`         | Initial dropdown open state.                   | `boolean`                                                                      | `false` |
| `placeholder`         | Placeholder content.                           | `JSX.Element`                                                                  | `-`     |
| `disabled`            | Disables the selector.                         | `boolean`                                                                      | `false` |
| `allowClear`          | Shows a clear button when a value is selected. | `boolean`                                                                      | `false` |
| `defaultExpandedKeys` | Initially expanded tree node values.           | `OptionValue[]`                                                                | `-`     |
| `prefixCls`           | Custom CSS class prefix.                       | `string`                                                                       | `-`     |
| `zIndex`              | Custom dropdown z-index.                       | `number`                                                                       | `-`     |
| `getPopupContainer`   | Returns dropdown portal container.             | `(triggerNode?: HTMLElement) => HTMLElement`                                   | `-`     |
| `onChange`            | Called when selection changes.                 | `(value: OptionValue \| undefined, node: TreeSelectNode \| undefined) => void` | `-`     |
| `onOpenChange`        | Called when dropdown open state changes.       | `(open: boolean) => void`                                                      | `-`     |

### TreeSelectNode

| Property   | Description              | Type               | Default |
| ---------- | ------------------------ | ------------------ | ------- |
| `title`    | Node display title.      | `JSX.Element`      | `-`     |
| `value`    | Node option value.       | `OptionValue`      | `-`     |
| `disabled` | Disables this node.      | `boolean`          | `false` |
| `children` | Child tree select nodes. | `TreeSelectNode[]` | `-`     |
