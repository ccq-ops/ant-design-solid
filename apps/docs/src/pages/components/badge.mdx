# Badge

### Count

```tsx
import { Badge, Button, Space } from '@ant-design-solid/core'

const Demo1 = function () {
  return (
    <Space wrap>
      <Badge count={5}>
        <Button>Inbox</Button>
      </Badge>
      <Badge count={120} overflowCount={99}>
        <Button>Alerts</Button>
      </Badge>
      <Badge count={0} showZero>
        <Button>Zero</Button>
      </Badge>
    </Space>
  )
}

export default Demo1
```

### Overflow count

```tsx
import { Badge, Button, Space } from '@ant-design-solid/core'

const Demo2 = function () {
  return (
    <Space wrap>
      <Badge count={1000} overflowCount={999}>
        <Button>Messages</Button>
      </Badge>
      <Badge count="NEW">
        <Button>Release</Button>
      </Badge>
    </Space>
  )

export default Demo2
```

### Status

```tsx
import { Badge, Button, Space } from '@ant-design-solid/core'

const Demo4 = function () {
  return (
    <Space direction="vertical">
      <Badge status="success" text="Active" />
      <Badge status="processing" text="Processing" />
      <Badge status="warning" text="Warning" />
      <Badge status="error" text="Error" />
      <Badge status="default" text="Default" />
    </Space>
  )
}

export default Demo4
```

## API

| Property        | Description                                                        | Type                                                             | Default |
| --------------- | ------------------------------------------------------------------ | ---------------------------------------------------------------- | ------- |
| `count`         | Badge count content.                                               | `number \| string`                                               | `-`     |
| `dot`           | Shows a small dot instead of count content.                        | `boolean`                                                        | `false` |
| `status`        | Standalone status badge style. Used when no children are provided. | `'success' \| 'processing' \| 'default' \| 'error' \| 'warning'` | `-`     |
| `text`          | Text displayed beside a standalone status dot.                     | `JSX.Element`                                                    | `-`     |
| `overflowCount` | Maximum numeric count before showing plus text.                    | `number`                                                         | `99`    |
| `showZero`      | Shows zero counts instead of hiding the badge.                     | `boolean`                                                        | `false` |
| `children`      | Element wrapped by the badge.                                      | `JSX.Element`                                                    | `-`     |
