# Menu

Menu displays a list of navigation actions with optional groups and submenus.

### Basic

```tsx
import { Menu, Space } from '@ant-design-solid/core'

const items = [
  { key: 'mail', label: 'Mail' },
  {
    key: 'settings',
    label: 'Settings',
    children: [
      { key: 'profile', label: 'Profile' },
      { key: 'security', label: 'Security' },
    ],
  },
  { type: 'divider' },
  {
    type: 'group',
    label: 'Workspace',
    children: [
      { key: 'team', label: 'Team' },
      { key: 'projects', label: 'Projects' },
    ],
  },
]

const Demo1 = function () {
  return (
    <Menu
      items={[
        { key: 'mail', label: 'Mail' },
        { key: 'team', label: 'Team' },
        { key: 'projects', label: 'Projects' },
      ]}
      defaultSelectedKeys={['mail']}
      class="w-60"
    />
  )
}

export default Demo1
```

### Collapsed

```tsx
import { Menu, Space } from '@ant-design-solid/core'

const items = [
  { key: 'mail', label: 'Mail' },
  {
    key: 'settings',
    label: 'Settings',
    children: [
      { key: 'profile', label: 'Profile' },
      { key: 'security', label: 'Security' },
    ],
  },
  { type: 'divider' },
  {
    type: 'group',
    label: 'Workspace',
    children: [
      { key: 'team', label: 'Team' },
      { key: 'projects', label: 'Projects' },
    ],
  },
]

const Demo6 = function () {
  return (
    <Space align="start">
      <Menu
        inlineCollapsed
        defaultSelectedKeys={['mail']}
        defaultOpenKeys={['settings']}
        items={[
          { key: 'mail', label: 'Mail', icon: '✉️' },
          {
            key: 'settings',
            label: 'Settings',
            icon: '⚙️',
            children: [{ key: 'profile', label: 'Profile' }],
          },
        ]}
      />
      <Menu
        defaultSelectedKeys={['mail']}
        defaultOpenKeys={['settings']}
        items={[
          { key: 'mail', label: 'Mail', icon: '✉️' },
          {
            key: 'settings',
            label: 'Settings',
            icon: '⚙️',
            children: [{ key: 'profile', label: 'Profile' }],
          },
        ]}
        class="w-55"
      />
    </Space>
  )
}

export default Demo6
```

## API

### Menu

| Property              | Description                                        | Type                                         | Default      |
| --------------------- | -------------------------------------------------- | -------------------------------------------- | ------------ |
| `items`               | Menu items, submenus, groups, and dividers.        | `MenuItem[]`                                 | `-`          |
| `mode`                | Menu layout mode.                                  | `'vertical' \| 'horizontal' \| 'inline'`     | `'vertical'` |
| `selectedKeys`        | Controlled selected item keys.                     | `string[]`                                   | `-`          |
| `defaultSelectedKeys` | Initial selected item keys for uncontrolled usage. | `string[]`                                   | `-`          |
| `openKeys`            | Controlled open submenu keys.                      | `string[]`                                   | `-`          |
| `defaultOpenKeys`     | Initial open submenu keys for uncontrolled usage.  | `string[]`                                   | `-`          |
| `inlineCollapsed`     | Collapses inline menus.                            | `boolean`                                    | `false`      |
| `zIndex`              | Overrides popup submenu z-index.                   | `number`                                     | `-`          |
| `getPopupContainer`   | Returns popup submenu portal container.            | `(triggerNode?: HTMLElement) => HTMLElement` | `-`          |
| `onClick`             | Called when an enabled menu item is clicked.       | `(info: MenuClickInfo) => void`              | `-`          |
| `onSelect`            | Called when selected keys change.                  | `(info: MenuSelectInfo) => void`             | `-`          |
| `onOpenChange`        | Called when open submenu keys change.              | `(openKeys: string[]) => void`               | `-`          |

### MenuItem

| Property   | Description                                   | Type                                          | Default |
| ---------- | --------------------------------------------- | --------------------------------------------- | ------- |
| `key`      | Stable menu key for item and submenu entries. | `string`                                      | `-`     |
| `label`    | Menu label content.                           | `JSX.Element`                                 | `-`     |
| `icon`     | Optional leading icon.                        | `JSX.Element`                                 | `-`     |
| `disabled` | Disables item interaction.                    | `boolean`                                     | `-`     |
| `type`     | Entry type: item, submenu, group, or divider. | `'item' \| 'submenu' \| 'group' \| 'divider'` | `-`     |
| `children` | Nested entries for submenus and groups.       | `MenuItem[]`                                  | `-`     |
