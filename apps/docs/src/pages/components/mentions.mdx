# Mentions

Mentions lets users select people or topics while typing in a textarea.

### Basic

```tsx
import { createSignal } from 'solid-js'
import { Form, Mentions, Space } from '@ant-design-solid/core'

const peopleOptions = [
  { label: 'Alice', value: 'alice' },
  { label: 'Bob', value: 'bob' },
  { label: 'Charlie', value: 'charlie', disabled: true },
]

const Demo1 = function () {
  return <Mentions placeholder="Mention someone with @" options={peopleOptions} />
}

export default Demo1
```

### Default value and clear

```tsx
import { createSignal } from 'solid-js'
import { Form, Mentions, Space } from '@ant-design-solid/core'

const peopleOptions = [
  { label: 'Alice', value: 'alice' },

const Demo2 = function () {

```

### Controlled

```tsx
import { createSignal } from 'solid-js'
import { Form, Mentions, Space } from '@ant-design-solid/core'

const peopleOptions = [
  { label: 'Alice', value: 'alice' },
  { label: 'Bob', value: 'bob' },
  { label: 'Charlie', value: 'charlie', disabled: true },
]

const Demo3 = function () {
  const [value, setValue] = createSignal('Hello @alice ')

  return (
    <Space direction="vertical" class="w-80">
      <Mentions value={value()} onChange={setValue} options={peopleOptions} />
      <span>Value: {value()}</span>
    </Space>
  )
}

export default Demo3
```

### Multiple prefixes

```tsx
import { createSignal } from 'solid-js'
import { Form, Mentions, Space } from '@ant-design-solid/core'

const topicOptions = [
  { label: 'Design', value: 'design' },
  { label: 'Solid', value: 'solid' },
  { label: 'Components', value: 'components' },
]

const Demo4 = function () {
  return (
    <Space direction="vertical" class="w-80">
      <Mentions
        prefix={['@', '#']}
        placeholder="Type @ or # to open suggestions"
        options={topicOptions}
      />
    </Space>
  )
}

export default Demo4
```

### Disabled and form

```tsx
import { createSignal } from 'solid-js'
import { Form, Mentions, Space } from '@ant-design-solid/core'

const peopleOptions = [
  { label: 'Alice', value: 'alice' },
  { label: 'Bob', value: 'bob' },
  { label: 'Charlie', value: 'charlie', disabled: true },
]

const Demo5 = function () {
  return (
    <Space direction="vertical" class="w-80">
      <Mentions disabled defaultValue="Disabled @alice" options={peopleOptions} />
      <Form initialValues={{ comment: 'Hi @bob ' }}>
        <Form.Item name="comment">
          <Mentions options={peopleOptions} />
        </Form.Item>
      </Form>
    </Space>
  )
}

export default Demo5
```

## API

### Mentions

| Property            | Description                                                   | Type                                                                   | Default |
| ------------------- | ------------------------------------------------------------- | ---------------------------------------------------------------------- | ------- |
| `value`             | Controlled textarea value.                                    | `string`                                                               | `-`     |
| `defaultValue`      | Initial textarea value for uncontrolled usage.                | `string`                                                               | `-`     |
| `open`              | Controlled suggestions popup open state.                      | `boolean`                                                              | `-`     |
| `defaultOpen`       | Initial suggestions popup open state.                         | `boolean`                                                              | `false` |
| `options`           | Suggestion options.                                           | `MentionsOption[]`                                                     | `-`     |
| `prefix`            | Mention trigger prefix or prefixes.                           | `string \| string[]`                                                   | `'@'`   |
| `split`             | Text inserted after a selected mention.                       | `string`                                                               | `' '`   |
| `placeholder`       | Textarea placeholder text.                                    | `string`                                                               | `-`     |
| `disabled`          | Disables editing and suggestions.                             | `boolean`                                                              | `false` |
| `allowClear`        | Shows a clear affordance when the value is not empty.         | `boolean`                                                              | `false` |
| `filterOption`      | Controls filtering or supplies a custom option filter.        | `boolean \| ((inputValue: string, option: MentionsOption) => boolean)` | `-`     |
| `validateSearch`    | Controls whether text after a prefix should open suggestions. | `(text: string, props: MentionsProps) => boolean`                      | `-`     |
| `prefixCls`         | Custom CSS class prefix.                                      | `string`                                                               | `-`     |
| `zIndex`            | Overrides popup z-index.                                      | `number`                                                               | `-`     |
| `getPopupContainer` | Returns the element used to mount the popup portal.           | `(triggerNode?: HTMLElement) => HTMLElement`                           | `-`     |
| `onChange`          | Called when textarea value changes.                           | `(value: string) => void`                                              | `-`     |
| `onSelect`          | Called when an option is selected.                            | `(option: MentionsOption, prefix: string) => void`                     | `-`     |
| `onSearch`          | Called while searching after a prefix.                        | `(text: string, prefix: string) => void`                               | `-`     |
| `onOpenChange`      | Called when popup open state changes.                         | `(open: boolean) => void`                                              | `-`     |

### MentionsOption

| Property   | Description                                             | Type          | Default |
| ---------- | ------------------------------------------------------- | ------------- | ------- |
| `label`    | Option display label. Falls back to value when omitted. | `JSX.Element` | `-`     |
| `value`    | Mention value inserted after the prefix.                | `string`      | `-`     |
| `disabled` | Disables selecting this option.                         | `boolean`     | `-`     |
