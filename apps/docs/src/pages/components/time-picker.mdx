# TimePicker

Select a time from hour, minute, and second columns.

### Basic

```tsx
import { createSignal } from 'solid-js'
import { Space, TimePicker } from '@ant-design-solid/core'

const Demo1 = function () {
  return <TimePicker />
}

export default Demo1
```

### HH:mm

```tsx
import { createSignal } from 'solid-js'
import { Space, TimePicker } from '@ant-design-solid/core'

const Demo2 = function () {
  return <TimePicker format="HH:mm" minuteStep={15} />
}

export default Demo2
```

### Controlled

```tsx
import { createSignal } from 'solid-js'
import { Space, TimePicker } from '@ant-design-solid/core'

const Demo3 = function () {
  const [value, setValue] = createSignal('09:30:00')

  const [open, setOpen] = createSignal(false)

  return (
    <Space direction="vertical">
      <TimePicker value={value()} open={open()} onChange={setValue} onOpenChange={setOpen} />
      <span>Selected time: {value() ?? 'none'}</span>
    </Space>
  )
}

export default Demo3
```

### Disabled time

```tsx
import { createSignal } from 'solid-js'
import { Space, TimePicker } from '@ant-design-solid/core'

const Demo4 = function () {
  return (
    <TimePicker
      disabledHours={() => [0, 1, 2, 3, 4, 5, 6]}
      disabledMinutes={(hour) => (hour === 12 ? [0, 15, 30, 45] : [])}
    />
  )
}

export default Demo4
```

### Clearable

```tsx
import { createSignal } from 'solid-js'
```

## API

| Property            | Description                                                  | Type                                                                                   | Default         |
| ------------------- | ------------------------------------------------------------ | -------------------------------------------------------------------------------------- | --------------- |
| `value`             | Controlled time value.                                       | `string`                                                                               | `-`             |
| `defaultValue`      | Initial time value for uncontrolled usage.                   | `string`                                                                               | `-`             |
| `format`            | Time display and emitted value format.                       | `'HH:mm:ss' \| 'HH:mm'`                                                                | `'HH:mm:ss'`    |
| `placeholder`       | Placeholder text when no time is selected.                   | `string`                                                                               | `'Select time'` |
| `disabled`          | Disables the picker.                                         | `boolean`                                                                              | `false`         |
| `allowClear`        | Shows a clear button when a value is selected.               | `boolean`                                                                              | `false`         |
| `open`              | Controlled popup open state.                                 | `boolean`                                                                              | `-`             |
| `defaultOpen`       | Initial popup open state.                                    | `boolean`                                                                              | `false`         |
| `minuteStep`        | Minute option interval.                                      | `number`                                                                               | `1`             |
| `secondStep`        | Second option interval.                                      | `number`                                                                               | `1`             |
| `disabledHours`     | Returns disabled hour values.                                | `() => number[]`                                                                       | `-`             |
| `disabledMinutes`   | Returns disabled minute values for selected hour.            | `(selectedHour: number \| undefined) => number[]`                                      | `-`             |
| `disabledSeconds`   | Returns disabled second values for selected hour and minute. | `(selectedHour: number \| undefined, selectedMinute: number \| undefined) => number[]` | `-`             |
| `onChange`          | Called when selection changes.                               | `(value: string \| undefined) => void`                                                 | `-`             |
| `onOpenChange`      | Called when popup open state changes.                        | `(open: boolean) => void`                                                              | `-`             |
| `prefixCls`         | Custom CSS class prefix.                                     | `string`                                                                               | `-`             |
| `zIndex`            | Custom popup z-index.                                        | `number`                                                                               | `-`             |
| `getPopupContainer` | Returns popup portal container.                              | `(triggerNode?: HTMLElement) => HTMLElement`                                           | `-`             |
