# Select

Select lets users choose one value from a dropdown list.

### Basic

```tsx
import { createSignal } from 'solid-js'
import { Form, Select, Space } from '@ant-design-solid/core'

const options = [
  { label: 'Apple', value: 'apple' },
  { label: 'Orange', value: 'orange' },
  { label: 'Pear', value: 'pear' },
]

const Demo1 = function () {
  return <Select class="w-60" placeholder="Select fruit" options={options} />
}

export default Demo1
```

### Controlled

```tsx
import { createSignal } from 'solid-js'
import { Form, Select, Space } from '@ant-design-solid/core'

const options = [
  { label: 'Apple', value: 'apple' },
  { label: 'Orange', value: 'orange' },
  { label: 'Pear', value: 'pear' },
]

const Demo2 = function () {
  const [value, setValue] = createSignal('orange')

  return (
    <Space direction="vertical">
      <Select class="w-60" value={value()} onChange={(next) => setValue(next)} options={options} />
      <span>Selected: {String(value() ?? 'none')}</span>
    </Space>
  )
}

export default Demo2
```

### Disabled options

```tsx
import { createSignal } from 'solid-js'
import { Form, Select, Space } from '@ant-design-solid/core'

const options = [
  { label: 'Apple', value: 'apple' },
  { label: 'Orange', value: 'orange' },
  { label: 'Pear', value: 'pear' },
]

const disabledOptions = [
  { label: 'Available', value: 'available' },
  { label: 'Disabled', value: 'disabled', disabled: true },
  { label: 'Pending', value: 'pending' },
]

const Demo4 = function () {
  return (
    <Space direction="vertical" class="w-60">
      <Select placeholder="Pick status" options={disabledOptions} />
      <Select disabled placeholder="Disabled select" options={disabledOptions} />
    </Space>
  )
}

export default Demo4
```

### Open state

```tsx
import { createSignal } from 'solid-js'
import { Form, Select, Space } from '@ant-design-solid/core'

const options = [
  { label: 'Apple', value: 'apple' },
  { label: 'Orange', value: 'orange' },
  { label: 'Pear', value: 'pear' },
]

const Demo5 = function () {
  const [open, setOpen] = createSignal(false)

  return (
    <Space direction="vertical">
      <Select class="w-60" open={open()} onOpenChange={setOpen} options={options} />
      <span>Dropdown is {open() ? 'open' : 'closed'}</span>
    </Space>
  )
}

export default Demo5
```

### In Form

```tsx
import { createSignal } from 'solid-js'
import { Form, Select, Space } from '@ant-design-solid/core'

const options = [
  { label: 'Apple', value: 'apple' },
  { label: 'Orange', value: 'orange' },
  { label: 'Pear', value: 'pear' },
]

const Demo6 = function () {
  return (
    <Form initialValues={{ fruit: 'apple' }}>
      <Form.Item label="Fruit" name="fruit">
        <Select class="w-60" options={options} />
      </Form.Item>
    </Form>
  )
}

export default Demo6
```

## API

| Property            | Description                                                            | Type                                         | Default |
| ------------------- | ---------------------------------------------------------------------- | -------------------------------------------- | ------- |
| `value`             | Controlled selected value.                                             | `string \| number \| boolean`                | `-`     |
| `defaultValue`      | Initial selected value for uncontrolled usage.                         | `string \| number \| boolean`                | `-`     |
| `open`              | Controlled dropdown open state.                                        | `boolean`                                    | `-`     |
| `defaultOpen`       | Initial dropdown open state.                                           | `boolean`                                    | `false` |
| `options`           | Select options. Primitive options are normalized to label/value pairs. | `OptionInput[]`                              | `-`     |
| `placeholder`       | Placeholder displayed when no value is selected.                       | `JSX.Element`                                | `-`     |
| `disabled`          | Disables opening and value changes.                                    | `boolean`                                    | `false` |
| `allowClear`        | Shows a clear button when a value is selected.                         | `boolean`                                    | `false` |
| `zIndex`            | Overrides dropdown z-index.                                            | `number`                                     | `-`     |
| `getPopupContainer` | Returns the element used to mount the dropdown portal.                 | `(triggerNode?: HTMLElement) => HTMLElement` | `-`     |
| `onChange`          | Called when the selected value changes.                                | `(value, option) => void`                    | `-`     |
| `onOpenChange`      | Called when the dropdown open state changes.                           | `(open: boolean) => void`                    | `-`     |
