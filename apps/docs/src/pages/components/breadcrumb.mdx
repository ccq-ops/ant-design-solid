# Breadcrumb

### Basic

```tsx
import { Breadcrumb, Space, message } from '@ant-design-solid/core'

const Demo1 = function () {
  return (
    <Breadcrumb
      items={[
        { title: 'Home', href: '/' },
        { title: 'Components', href: '/components' },
        { title: 'Breadcrumb' },
      ]}
    />
  )
}

export default Demo1
```

### Custom separator

```tsx
import { Breadcrumb, Space, message } from '@ant-design-solid/core'

const Demo2 = function () {
  return (
    <Breadcrumb
      separator=">"
      items={[
        { title: 'Home', href: '/' },
        { title: 'Library', separator: '→' },
        { title: 'Breadcrumb' },
      ]}
    />
  )
}

export default Demo2
```

### Clickable item

```tsx
import { Breadcrumb, Space, message } from '@ant-design-solid/core'

const Demo3 = function () {
  return (
    <Breadcrumb
      items={[
        { title: 'Home', href: '/' },
        {
          title: 'Reload',
          onClick: () => message.success('Reloaded'),
        },
        { title: 'Current' },
      ]}
    />
  )
}

export default Demo3
```

### Manual composition

```tsx
import { Breadcrumb, Space, message } from '@ant-design-solid/core'

const Demo4 = function () {
  return (
    <Space direction="vertical">
      <Breadcrumb>
        <Breadcrumb.Item href="/">Home</Breadcrumb.Item>
        <Breadcrumb.Item href="/components">Components</Breadcrumb.Item>
        <Breadcrumb.Item>Breadcrumb</Breadcrumb.Item>
      </Breadcrumb>
    </Space>
  )
}

export default Demo4
```

## API

### Breadcrumb

| Property    | Description                          | Type                   | Default |
| ----------- | ------------------------------------ | ---------------------- | ------- |
| `items`     | Breadcrumb items rendered from data. | `BreadcrumbItemType[]` | `-`     |
| `separator` | Default separator between items.     | `JSX.Element`          | `'/'`   |
| `children`  | Manual breadcrumb item composition.  | `JSX.Element`          | `-`     |

### Breadcrumb item data

| Property    | Description                                     | Type                                             | Default |
| ----------- | ----------------------------------------------- | ------------------------------------------------ | ------- |
| `title`     | Item label for data-driven breadcrumbs.         | `JSX.Element`                                    | `-`     |
| `href`      | Renders the item as a link when provided.       | `string`                                         | `-`     |
| `onClick`   | Click handler for interactive breadcrumb items. | `JSX.EventHandlerUnion<HTMLElement, MouseEvent>` | `-`     |
| `separator` | Custom separator after this item.               | `JSX.Element`                                    | `-`     |

### Breadcrumb.Item

| Property   | Description                         | Type                                             | Default |
| ---------- | ----------------------------------- | ------------------------------------------------ | ------- |
| `href`     | Renders this manual item as a link. | `string`                                         | `-`     |
| `onClick`  | Click handler for this manual item. | `JSX.EventHandlerUnion<HTMLElement, MouseEvent>` | `-`     |
| `children` | Item label content.                 | `JSX.Element`                                    | `-`     |
