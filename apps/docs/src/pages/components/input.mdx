# Input

Input captures single-line text and includes affix, clear, count, status and variant states.

### Basic

```tsx
import { createSignal } from 'solid-js'
import { Form, Input, Space } from '@ant-design-solid/core'
import { SearchOutlined, UserOutlined } from '@ant-design-solid/icons'

const Demo1 = function () {
  return <Input placeholder="Basic usage" />
}

export default Demo1
```

### Controlled

```tsx
import { createSignal } from 'solid-js'
import { Form, Input, Space } from '@ant-design-solid/core'
import { SearchOutlined, UserOutlined } from '@ant-design-solid/icons'

function ControlledInputDemo() {
  const [value, setValue] = createSignal('Solid')

  return (
    <Space direction="vertical" class="w-80">
      <Input value={value()} onInput={(event) => setValue(event.currentTarget.value)} />
      <span class="text-sm text-gray-500">Current value: {value()}</span>
    </Space>
  )
}

const Demo2 = function () {
  return <ControlledInputDemo />
}

export default Demo2
```

### Search

```tsx
import { createSignal } from 'solid-js'
import { Form, Input, Space } from '@ant-design-solid/core'
import { SearchOutlined, UserOutlined } from '@ant-design-solid/icons'

function SearchDemo() {
  const [message, setMessage] = createSignal('Search result will appear here')

  return (
    <Space direction="vertical" class="w-96">
      <Input.Search
        allowClear
        placeholder="Search docs"
        onSearch={(value, _event, info) => setMessage(`Searched "${value}" from ${info.source}`)}
      />
      <Input.Search
        defaultValue="Solid"
        enterButton="Search"
        loading
        onSearch={(value) => setMessage(`Button search: ${value}`)}
      />
      <span class="text-sm text-gray-500">{message()}</span>
    </Space>
  )
}

const Demo10 = function () {
  return <SearchDemo />
}

export default Demo10
```

### Password

```tsx
import { Form, Input, Space } from '@ant-design-solid/core'

function PasswordDemo() {
  const [visible, setVisible] = createSignal(false)

  return (
    <Space direction="vertical" class="w-80">

        defaultValue="controlled"
      />
      <span class="text-sm text-gray-500">
        Controlled visibility: {visible() ? 'visible' : 'hidden'}
      </span>
    </Space>
  )
}

const Demo11 = function () {
  return <PasswordDemo />
}

export default Demo11
```

## API

### Input

| Property       | Description                                                                                                        | Type                                                                   | Default    |
| -------------- | ------------------------------------------------------------------------------------------------------------------ | ---------------------------------------------------------------------- | ---------- |
| `size`         | Input size. Defaults to the current ConfigProvider component size.                                                 | `'small' \| 'middle' \| 'large'`                                       | `-`        |
| `status`       | Validation visual status.                                                                                          | `'error' \| 'warning'`                                                 | `-`        |
| `variant`      | Visual variant of the input wrapper.                                                                               | `'outlined' \| 'borderless' \| 'filled' \| 'underlined'`               | `outlined` |
| `prefix`       | Element rendered before the input value.                                                                           | `JSX.Element`                                                          | `-`        |
| `suffix`       | Element rendered after the input value.                                                                            | `JSX.Element`                                                          | `-`        |
| `allowClear`   | Shows a clear button when the input has a value. Object form customizes the icon or disables the clear affordance. | `boolean \| { clearIcon?: JSX.Element; disabled?: boolean }`           | `false`    |
| `showCount`    | Shows the current character count. Object form customizes count rendering.                                         | `boolean \| { formatter?: (info: CountFormatterInfo) => JSX.Element }` | `false`    |
| `count`        | Character count configuration for custom max, count strategy, display and exceed formatting.                       | `CountConfig`                                                          | `-`        |
| `classNames`   | Semantic class names for wrapper, input, prefix, suffix, clear and count nodes.                                    | `InputSemanticClassNames`                                              | `-`        |
| `styles`       | Semantic inline styles for wrapper, input, prefix, suffix, clear and count nodes.                                  | `InputSemanticStyles`                                                  | `-`        |
| `value`        | Controlled input value.                                                                                            | `string \| number`                                                     | `-`        |
| `defaultValue` | Initial uncontrolled input value.                                                                                  | `string \| number`                                                     | `-`        |
| `onChange`     | Callback when the input value changes.                                                                             | `(event: Event) => void`                                               | `-`        |
| `onPressEnter` | Callback when Enter is pressed.                                                                                    | `(event: KeyboardEvent) => void`                                       | `-`        |
| `onClear`      | Callback when the clear button is clicked.                                                                         | `() => void`                                                           | `-`        |

### CountConfig

| Property          | Description                                                                                                                  | Type                                                     | Default |
| ----------------- | ---------------------------------------------------------------------------------------------------------------------------- | -------------------------------------------------------- | ------- |
| `max`             | Maximum character count for warning state. Unlike native maxLength, this marks exceed state instead of truncating by itself. | `number`                                                 | `-`     |
| `strategy`        | Custom count strategy, useful for emoji or domain-specific length rules.                                                     | `(value: string) => number`                              | `-`     |
| `show`            | Whether to show count, or a render function for custom count content.                                                        | `boolean \| ((info: CountFormatterInfo) => JSX.Element)` | `-`     |
| `exceedFormatter` | Custom value formatter when count exceeds max.                                                                               | `(value: string, config: { max: number }) => string`     | `-`     |

### Input.TextArea

| Property       | Description                                                                | Type                                                                   | Default    |
| -------------- | -------------------------------------------------------------------------- | ---------------------------------------------------------------------- | ---------- |
| `autoSize`     | Autosize textarea rows. Object form supports minRows and maxRows.          | `boolean \| { minRows?: number; maxRows?: number }`                    | `false`    |
| `allowClear`   | Shows a clear button when the textarea has a value.                        | `boolean \| { clearIcon?: JSX.Element; disabled?: boolean }`           | `false`    |
| `showCount`    | Shows the current character count. Object form customizes count rendering. | `boolean \| { formatter?: (info: CountFormatterInfo) => JSX.Element }` | `false`    |
| `count`        | Character count configuration shared with Input.                           | `CountConfig`                                                          | `-`        |
| `variant`      | Visual variant of the textarea.                                            | `'outlined' \| 'borderless' \| 'filled' \| 'underlined'`               | `outlined` |
| `status`       | Validation visual status.                                                  | `'error' \| 'warning'`                                                 | `-`        |
| `classNames`   | Semantic class names for wrapper, textarea, clear and count nodes.         | `TextAreaSemanticClassNames`                                           | `-`        |
| `styles`       | Semantic inline styles for wrapper, textarea, clear and count nodes.       | `TextAreaSemanticStyles`                                               | `-`        |
| `value`        | Controlled textarea value.                                                 | `string \| number`                                                     | `-`        |
| `defaultValue` | Initial uncontrolled textarea value.                                       | `string \| number`                                                     | `-`        |
| `onPressEnter` | Callback when Enter is pressed.                                            | `(event: KeyboardEvent) => void`                                       | `-`        |
| `onClear`      | Callback when the clear button is clicked.                                 | `() => void`                                                           | `-`        |

### Input.Search

Supports all Input props plus the following props.

| Property      | Description                                                                                               | Type                                                           | Default |
| ------------- | --------------------------------------------------------------------------------------------------------- | -------------------------------------------------------------- | ------- |
| `enterButton` | Render a search button after the input. True renders the default button text; JSX customizes the content. | `boolean \| JSX.Element`                                       | `false` |
| `loading`     | Shows loading state on the search icon or enter button.                                                   | `boolean`                                                      | `false` |
| `searchIcon`  | Custom search icon when enterButton is not used.                                                          | `JSX.Element`                                                  | `-`     |
| `onSearch`    | Callback when clicking search, pressing Enter, or clearing the input.                                     | `(value, event, info: { source: "input" \| "clear" }) => void` | `-`     |

### Input.Password

Supports most Input props plus password visibility controls.

| Property           | Description                                                              | Type                                                                             | Default |
| ------------------ | ------------------------------------------------------------------------ | -------------------------------------------------------------------------------- | ------- |
| `iconRender`       | Custom password visibility toggle icon.                                  | `(visible: boolean) => JSX.Element`                                              | `-`     |
| `visibilityToggle` | Whether to show the toggle icon, or controlled visibility configuration. | `boolean \| { visible?: boolean; onVisibleChange?: (visible: boolean) => void }` | `true`  |

### Input.OTP

| Property       | Description                                                     | Type                                                     | Default    |
| -------------- | --------------------------------------------------------------- | -------------------------------------------------------- | ---------- |
| `length`       | Number of OTP input boxes.                                      | `number`                                                 | `6`        |
| `mask`         | Masks displayed characters without changing the original value. | `boolean \| string`                                      | `false`    |
| `formatter`    | Formats each displayed character.                               | `(value: string) => string`                              | `-`        |
| `separator`    | Renders a separator after the input at the specified index.     | `JSX.Element \| ((index: number) => JSX.Element)`        | `-`        |
| `autoComplete` | Autocomplete attribute for OTP autofill, such as one-time-code. | `string`                                                 | `-`        |
| `value`        | Controlled OTP value.                                           | `string`                                                 | `-`        |
| `defaultValue` | Initial uncontrolled OTP value.                                 | `string`                                                 | `-`        |
| `size`         | Size of each OTP input.                                         | `'small' \| 'middle' \| 'large'`                         | `middle`   |
| `status`       | Validation visual status.                                       | `'error' \| 'warning'`                                   | `-`        |
| `variant`      | Visual variant of OTP inputs.                                   | `'outlined' \| 'borderless' \| 'filled' \| 'underlined'` | `outlined` |
| `classNames`   | Semantic class names for wrapper, input and separator nodes.    | `OTPSemanticClassNames`                                  | `-`        |
| `styles`       | Semantic inline styles for wrapper, input and separator nodes.  | `OTPSemanticStyles`                                      | `-`        |
| `onInput`      | Callback on every OTP input value update.                       | `(value: string[]) => void`                              | `-`        |
| `onChange`     | Callback when all OTP fields are filled.                        | `(value: string) => void`                                | `-`        |
