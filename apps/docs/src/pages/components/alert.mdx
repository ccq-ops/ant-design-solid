# Alert

### Basic

```tsx
import { Alert, Space } from '@ant-design-solid/core'

const Demo1 = function () {
  return (
    <Space direction="vertical" class="w-full">
      <Alert message="Success" type="success" showIcon />
      <Alert message="Info" type="info" showIcon />
      <Alert message="Warning" type="warning" showIcon />
      <Alert message="Error" type="error" showIcon />
    </Space>
  )
}

export default Demo1
```

### Closable

```tsx
import { Alert, Space } from '@ant-design-solid/core'

const Demo2 = function () {
  return <Alert closable message="Closable" description="More detail" />
}

export default Demo2
```

### Banner and custom icon

```tsx
import { Alert, Space } from '@ant-design-solid/core'

const Demo3 = function () {
  return <Alert banner title="Warning banner" icon={<span>!</span>} />
}

export default Demo3
```

### Closable config

```tsx
import { Alert, Space } from '@ant-design-solid/core'

const Demo4 = function () {
  return <Alert title="Configurable close" closable={{ closeIcon: <span>Dismiss</span> }} />
}

export default Demo4
```

## API

### Alert

| Property      | Description                                                            | Type                                          | Default  |
| ------------- | ---------------------------------------------------------------------- | --------------------------------------------- | -------- |
| `type`        | Alert semantic type.                                                   | `'success' \| 'info' \| 'warning' \| 'error'` | `'info'` |
| `title`       | Title content. `message` is also supported for Ant Design-style usage. | `JSX.Element`                                 | `-`      |
| `message`     | Primary alert message content.                                         | `JSX.Element`                                 | `-`      |
| `description` | Additional alert details.                                              | `JSX.Element`                                 | `-`      |
| `closable`    | Shows a close button or configures close behavior.                     | `boolean \| AlertClosableConfig`              | `false`  |
| `showIcon`    | Shows the default type icon.                                           | `boolean`                                     | `false`  |
| `icon`        | Custom icon content.                                                   | `JSX.Element`                                 | `-`      |
| `action`      | Action content rendered on the right.                                  | `JSX.Element`                                 | `-`      |
| `banner`      | Uses banner styling.                                                   | `boolean`                                     | `false`  |
| `afterClose`  | Called after the alert closes.                                         | `() => void`                                  | `-`      |
| `onClose`     | Called when the close button is clicked.                               | `(event: MouseEvent) => void`                 | `-`      |

### Alert closable config

| Property     | Description                               | Type                          | Default |
| ------------ | ----------------------------------------- | ----------------------------- | ------- |
| `closeIcon`  | Custom close icon content.                | `JSX.Element`                 | `-`     |
| `afterClose` | Called after this alert closes.           | `() => void`                  | `-`     |
| `onClose`    | Called when this close button is clicked. | `(event: MouseEvent) => void` | `-`     |
