# Tree

### Basic

```tsx
import { Tree } from '@ant-design-solid/core'

const treeData = [
  {
    title: 'Parent 1',
    key: 'parent-1',
    children: [
      { title: 'Child 1', key: 'child-1' },
      { title: 'Child 2', key: 'child-2', disabled: true },
    ],
  },
  {
    title: 'Parent 2',
    key: 'parent-2',
    children: [{ title: 'Child 3', key: 'child-3' }],
  },
]

const Demo1 = function () {
  return <Tree defaultExpandedKeys={['parent-1']} treeData={treeData} />
}

export default Demo1
```

## API

### Tree

| Property              | Description                             | Type                                                      | Default |
| --------------------- | --------------------------------------- | --------------------------------------------------------- | ------- |
| `treeData`            | Tree node data.                         | `TreeDataNode[]`                                          | `-`     |
| `expandedKeys`        | Controlled expanded node keys.          | `TreeKey[]`                                               | `-`     |
| `defaultExpandedKeys` | Initial expanded node keys.             | `TreeKey[]`                                               | `-`     |
| `selectedKeys`        | Controlled selected node keys.          | `TreeKey[]`                                               | `-`     |
| `defaultSelectedKeys` | Initial selected node keys.             | `TreeKey[]`                                               | `-`     |
| `checkedKeys`         | Controlled checked node keys.           | `TreeKey[]`                                               | `-`     |
| `defaultCheckedKeys`  | Initial checked node keys.              | `TreeKey[]`                                               | `-`     |
| `selectable`          | Whether nodes can be selected.          | `boolean`                                                 | `true`  |
| `checkable`           | Shows checkboxes for checkable nodes.   | `boolean`                                                 | `false` |
| `disabled`            | Disables the entire tree.               | `boolean`                                                 | `false` |
| `showLine`            | Shows line styling.                     | `boolean`                                                 | `false` |
| `blockNode`           | Makes node content fill the row.        | `boolean`                                                 | `false` |
| `prefixCls`           | Custom CSS class prefix.                | `string`                                                  | `-`     |
| `onExpand`            | Called when node expansion changes.     | `(expandedKeys: TreeKey[], info: TreeExpandInfo) => void` | `-`     |
| `onSelect`            | Called when node selection changes.     | `(selectedKeys: TreeKey[], info: TreeSelectInfo) => void` | `-`     |
| `onCheck`             | Called when node checked state changes. | `(checkedKeys: TreeKey[], info: TreeCheckInfo) => void`   | `-`     |

### TreeDataNode

| Property          | Description                                   | Type             | Default |
| ----------------- | --------------------------------------------- | ---------------- | ------- |
| `title`           | Node title content.                           | `JSX.Element`    | `-`     |
| `key`             | Unique node key.                              | `TreeKey`        | `-`     |
| `disabled`        | Disables the node.                            | `boolean`        | `false` |
| `disableCheckbox` | Disables this node checkbox.                  | `boolean`        | `false` |
| `selectable`      | Overrides whether this node can be selected.  | `boolean`        | `-`     |
| `checkable`       | Overrides whether this node shows a checkbox. | `boolean`        | `-`     |
| `children`        | Child tree nodes.                             | `TreeDataNode[]` | `-`     |
