# Checkbox

Checkbox lets users select one or more independent options.

### Basic

```tsx
import { createSignal } from 'solid-js'
import { Checkbox, Form, Space } from '@ant-design-solid/core'

const Demo1 = function () {
  return (
    <Space>
      <Checkbox defaultChecked>Agree</Checkbox>
      <Checkbox>Remember me</Checkbox>
      <Checkbox indeterminate>Partially selected</Checkbox>
    </Space>
  )
}

export default Demo1
```

### Controlled

```tsx
import { createSignal } from 'solid-js'
import { Checkbox, Form, Space } from '@ant-design-solid/core'

const Demo3 = function () {
  const [checked, setChecked] = createSignal(true)

  return (
    <Space>
      <Checkbox checked={checked()} onChange={(event) => setChecked(event.currentTarget.checked)}>
        Controlled
      </Checkbox>
      <span>{checked() ? 'Checked' : 'Unchecked'}</span>
    </Space>
  )
}

export default Demo3
```

### Group

```tsx
import { createSignal } from 'solid-js'
import { Checkbox, Form, Space } from '@ant-design-solid/core'

const checkboxOptions = [
  { label: 'Apple', value: 'apple' },
  { label: 'Orange', value: 'orange' },
  { label: 'Pear', value: 'pear', disabled: true },
]

const Demo4 = function () {
  const [fruits, setFruits] = createSignal(['apple'])

  return (
    <Space direction="vertical">
      <Checkbox.Group options={checkboxOptions} defaultValue={['apple']} />
      <Checkbox.Group
        options={checkboxOptions}
        value={fruits()}
        onChange={(next) => setFruits(next)}
      />
      <span>Selected: {fruits().join(', ') || 'none'}</span>
    </Space>
  )
}

export default Demo4
```

### In Form

```tsx
import { createSignal } from 'solid-js'
import { Checkbox, Form, Space } from '@ant-design-solid/core'

const checkboxOptions = [
  { label: 'Apple', value: 'apple' },
  { label: 'Orange', value: 'orange' },
  { label: 'Pear', value: 'pear', disabled: true },
]

const Demo5 = function () {
  return (
    <Form initialValues={{ agree: true, fruits: ['orange'] }}>
      <Space direction="vertical">
        <Form.Item name="agree" valuePropName="checked">
          <Checkbox>Agree to terms</Checkbox>
        </Form.Item>
        <Form.Item name="fruits">
          <Checkbox.Group options={checkboxOptions} />
        </Form.Item>
      </Space>
    </Form>
  )
}

export default Demo5
```

## API

### Checkbox

| Property         | Description                                         | Type                                        | Default |
| ---------------- | --------------------------------------------------- | ------------------------------------------- | ------- |
| `checked`        | Controlled checked state.                           | `boolean`                                   | `-`     |
| `defaultChecked` | Initial checked state for uncontrolled usage.       | `boolean`                                   | `false` |
| `disabled`       | Disables the checkbox.                              | `boolean`                                   | `false` |
| `indeterminate`  | Displays the mixed selection state.                 | `boolean`                                   | `false` |
| `prefixCls`      | Custom CSS class prefix.                            | `string`                                    | `-`     |
| `children`       | Checkbox label content.                             | `JSX.Element`                               | `-`     |
| `onChange`       | Called when the native checkbox change event fires. | `JSX.EventHandler<HTMLInputElement, Event>` | `-`     |

### Checkbox.Group

| Property       | Description                                    | Type                                    | Default |
| -------------- | ---------------------------------------------- | --------------------------------------- | ------- |
| `value`        | Controlled checked values.                     | `(string \| number \| boolean)[]`       | `-`     |
| `defaultValue` | Initial checked values for uncontrolled usage. | `(string \| number \| boolean)[]`       | `-`     |
| `options`      | Options rendered as checkboxes.                | `OptionInput[]`                         | `-`     |
| `disabled`     | Disables all checkboxes in the group.          | `boolean`                               | `false` |
| `prefixCls`    | Custom checkbox CSS class prefix.              | `string`                                | `-`     |
| `children`     | Additional custom checkbox content.            | `JSX.Element`                           | `-`     |
| `onChange`     | Called with the next checked value array.      | `(checkedValue: OptionValue[]) => void` | `-`     |
