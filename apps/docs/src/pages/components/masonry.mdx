# Masonry

Display variable-height content in balanced columns.

### Basic

```tsx
import { createSignal } from 'solid-js'
import { Button, Card, Masonry, Space } from '@ant-design-solid/core'

const baseItems = [
  { key: 1, title: 'Mountain', height: 120, color: '#e6f4ff' },
  { key: 2, title: 'Forest', height: 180, color: '#f6ffed' },
  { key: 3, title: 'River', height: 96, color: '#fff7e6' },
  { key: 4, title: 'City', height: 150, color: '#f9f0ff' },
  { key: 5, title: 'Desert', height: 110, color: '#fff1f0' },
  { key: 6, title: 'Ocean', height: 210, color: '#e6fffb' },
]

function DemoCard(props) {
  return (
    <Card title={props.title} style={{ background: props.color }}>
      <div style={{ height: `${props.height}px`, color: 'var(--docs-text-secondary)' }}>
        Variable-height content creates a waterfall layout.
      </div>
    </Card>
  )
}

const Demo1 = function () {
  return (
    <Masonry
      items={baseItems}
      itemRender={(item) => <DemoCard title={item.title} height={item.height} color={item.color} />}
    />
  )
}

export default Demo1
```

### Dynamic items

```tsx
import { createSignal } from 'solid-js'
import { Button, Card, Masonry, Space } from '@ant-design-solid/core'

function DemoCard(props) {
  return (
    <Card title={props.title} style={{ background: props.color }}>
      <div style={{ height: `${props.height}px`, color: 'var(--docs-text-secondary)' }}>
        Variable-height content creates a waterfall layout.
      </div>
    </Card>
  )
}

const Demo4 = function () {
  return (
    <Space direction="vertical" style={{ width: '100%' }}>
      <Button
        onClick={() =>
          setItems((previous) => [
            ...previous,
            {
              key: Date.now(),
              title: `New ${previous.length + 1}`,
              height: 80 + ((previous.length * 37) % 140),
              color: '#f0f5ff',
            },
          ])
        }
      >
        Add item
      </Button>
      <Masonry
        fresh
        columns={{ xs: 1, md: 3 }}
        items={items()}
        itemRender={(item) => (
          <DemoCard title={item.title} height={item.height} color={item.color} />
        )}
      />
    </Space>
  )
}

export default Demo4
```

## API

| Property         | Description                                                                   | Type                                                                       | Default |
| ---------------- | ----------------------------------------------------------------------------- | -------------------------------------------------------------------------- | ------- |
| `prefixCls`      | Custom CSS class prefix.                                                      | `string`                                                                   | `-`     |
| `class`          | Additional root class.                                                        | `string`                                                                   | `-`     |
| `classList`      | Conditional root classes.                                                     | `Record<string, boolean \| undefined>`                                     | `-`     |
| `style`          | Inline root style.                                                            | `JSX.CSSProperties \| string`                                              | `-`     |
| `columns`        | Column count or responsive breakpoint map.                                    | `number \| Partial<Record<MasonryBreakpoint, number>>`                     | `4`     |
| `gutter`         | Gap size or responsive breakpoint map.                                        | `number \| string \| Partial<Record<MasonryBreakpoint, number \| string>>` | `16`    |
| `items`          | Data items to render.                                                         | `T[]`                                                                      | `-`     |
| `itemRender`     | Render function for each data item.                                           | `(item: T, index: number) => JSX.Element`                                  | `-`     |
| `children`       | Manual masonry children used when items are not provided.                     | `JSX.Element`                                                              | `-`     |
| `fresh`          | Refreshes measured layout when item heights update.                           | `boolean`                                                                  | `false` |
| `classNames`     | Semantic class names for masonry parts.                                       | `{ item?: string }`                                                        | `-`     |
| `styles`         | Semantic inline styles for masonry parts.                                     | `{ item?: JSX.CSSProperties \| string }`                                   | `-`     |
| `onLayoutChange` | Called with columns, column heights, and item placement after layout changes. | `(info: MasonryLayoutInfo<T>) => void`                                     | `-`     |
