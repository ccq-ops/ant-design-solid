# Space

Space adds consistent gaps between inline or stacked elements.

### Basic

```tsx
import { Button, Card, Space } from '@ant-design-solid/core'

const Demo1 = function () {
  return (
    <Space>
      <Button>One</Button>
      <Button>Two</Button>
      <Button>Three</Button>
    </Space>
  )
}

export default Demo1
```

### Direction

```tsx
import { Button, Card, Space } from '@ant-design-solid/core'

const Demo2 = function () {
  return (
    <Space direction="vertical">
      <Button>Top</Button>
      <Button>Middle</Button>
      <Button>Bottom</Button>
    </Space>
  )
}

export default Demo2
```

### Size

```tsx
import { Button, Card, Space } from '@ant-design-solid/core'

const Demo3 = function () {
  return (
    <Space direction="vertical">
      <Space size="large">
        <Button>Large</Button>
        <Button>Gap</Button>
      </Space>
      <Space size={[8, 24]} wrap>
        <Button>8 x 24</Button>
        <Button>Tuple</Button>
        <Button>Gap</Button>
      </Space>
    </Space>
  )
}

export default Demo3
```

### Wrap

```tsx
import { Button, Card, Space } from '@ant-design-solid/core'

const Demo4 = function () {
  return (
    <Space wrap class="max-w-80">
      <Button>Alpha</Button>
      <Button>Beta</Button>
      <Button>Gamma</Button>
      <Button>Delta</Button>
      <Button>Epsilon</Button>
    </Space>
  )
}

export default Demo4
```

### Alignment

```tsx
import { Button, Card, Space } from '@ant-design-solid/core'

const Demo5 = function () {
  return (
    <Space align="start">
      <Card size="small" title="Card">
        Taller content
        <br />
        spans two lines.
      </Card>
      <Button>Aligned start</Button>
    </Space>
  )
}

export default Demo5
```

### Split

```tsx
import { Button, Card, Space } from '@ant-design-solid/core'

const Demo6 = function () {
  return (
    <Space split={<span class="text-gray-300">/</span>}>
      <a href="#home">Home</a>
      <a href="#docs">Docs</a>
      <a href="#components">Components</a>
    </Space>
  )
}

export default Demo6
```

## API

| Property    | Description                                                            | Type                                                           | Default        |
| ----------- | ---------------------------------------------------------------------- | -------------------------------------------------------------- | -------------- |
| `size`      | Gap size between items. Tuple values set horizontal and vertical gaps. | `'small' \| 'middle' \| 'large' \| number \| [number, number]` | `'middle'`     |
| `direction` | Stack items horizontally or vertically.                                | `'horizontal' \| 'vertical'`                                   | `'horizontal'` |
| `align`     | Cross-axis alignment of items.                                         | `'start' \| 'end' \| 'center' \| 'baseline'`                   | `-`            |
| `wrap`      | Allows horizontal items to wrap onto multiple lines.                   | `boolean`                                                      | `false`        |
| `split`     | Separator rendered between adjacent items.                             | `JSX.Element`                                                  | `-`            |
