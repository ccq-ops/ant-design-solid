# QRCode

### Basic

```tsx
import { QRCode, Space } from '@ant-design-solid/core'

const Demo1 = function () {
  return <QRCode value="https://solidjs.com" />
}

export default Demo1
```

### Custom colors

```tsx
import { QRCode, Space } from '@ant-design-solid/core'

const Demo2 = function () {
  return <QRCode value="ant-design-solid" color="#1677ff" bgColor="#f0f5ff" />
}

export default Demo2
```

### Icon

```tsx
import { QRCode, Space } from '@ant-design-solid/core'

const Demo3 = function () {
  return <QRCode value="solid" icon="/vite.svg" />
}

export default Demo3
```

### Status

```tsx
import { QRCode, Space } from '@ant-design-solid/core'

const Demo4 = function () {
  return (
    <Space>
      <QRCode value="loading" status="loading" size={128} />
      <QRCode value="expired" status="expired" size={128} />
    </Space>
  )
}

export default Demo4
```

### Borderless

```tsx
import { QRCode, Space } from '@ant-design-solid/core'

const Demo5 = function () {
  return <QRCode value="borderless" bordered={false} />
}

export default Demo5
```

## API

| Property       | Description                                | Type                                              | Default                  |
| -------------- | ------------------------------------------ | ------------------------------------------------- | ------------------------ |
| `value`        | Content encoded into the visual QR matrix. | `string`                                          | `-`                      |
| `size`         | QR code size in pixels.                    | `number`                                          | `160`                    |
| `color`        | Module color.                              | `string`                                          | `token.colorText`        |
| `bgColor`      | Background color.                          | `string`                                          | `token.colorBgContainer` |
| `bordered`     | Whether to render a border.                | `boolean`                                         | `true`                   |
| `icon`         | Image URL rendered in the center.          | `string`                                          | `-`                      |
| `iconSize`     | Center icon size in pixels.                | `number`                                          | `size / 4`               |
| `status`       | QR code status overlay.                    | `'active' \| 'expired' \| 'loading'`              | `'active'`               |
| `statusRender` | Custom status overlay renderer.            | `(info: { status: QRCodeStatus }) => JSX.Element` | `-`                      |
| `prefixCls`    | Custom CSS class prefix.                   | `string`                                          | `-`                      |
