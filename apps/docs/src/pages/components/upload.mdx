# Upload

Select files, optionally intercept uploads, and render a basic upload list.

### Basic

```tsx
import { createSignal } from 'solid-js'
import { Button, Space, Upload } from '@ant-design-solid/core'

const Demo1 = function () {
  return (
    <Upload>
      <Button>Upload file</Button>
    </Upload>
  )
}

export default Demo1
```

### beforeUpload false

```tsx
import { createSignal } from 'solid-js'
import { Button, Space, Upload } from '@ant-design-solid/core'

const Demo3 = function () {
  return (
    <Space direction="vertical">
      <Upload
        fileList={manualFiles()}
        beforeUpload={() => false}
        onChange={({ fileList }) => setManualFiles(fileList)}
      >
        <Button>Select without uploading</Button>
      </Upload>
      <span>
        Manual files:{' '}
        {manualFiles()
          .map((file) => file.name)
          .join(', ') || 'none'}
      </span>
    </Space>
  )
}

export default Demo3
```

### Custom request

```tsx
import { createSignal } from 'solid-js'
import { Button, Space, Upload } from '@ant-design-solid/core'

function delayedRequest({ onProgress, onSuccess }) {
  onProgress(35)
  window.setTimeout(() => {
    onProgress(100)
    onSuccess({ ok: true })
  }, 600)
}

const Demo4 = function () {
  return (
    <Upload customRequest={delayedRequest}>
      <Button>Upload with progress</Button>
    </Upload>
  )
}

export default Demo4
```

## API

### Upload

| Property          | Description                                                | Type                                                | Default |
| ----------------- | ---------------------------------------------------------- | --------------------------------------------------- | ------- |
| `fileList`        | Controlled upload file list.                               | `UploadFile[]`                                      | `-`     |
| `defaultFileList` | Initial upload file list for uncontrolled usage.           | `UploadFile[]`                                      | `-`     |
| `accept`          | Native file input accept attribute.                        | `string`                                            | `-`     |
| `multiple`        | Allows selecting multiple files.                           | `boolean`                                           | `false` |
| `disabled`        | Disables file selection and removal.                       | `boolean`                                           | `false` |
| `maxCount`        | Maximum number of files kept in the list.                  | `number`                                            | `-`     |
| `showUploadList`  | Whether to render the upload list.                         | `boolean`                                           | `true`  |
| `beforeUpload`    | Runs before upload. Return false to add without uploading. | `(file: File) => boolean \| Promise<boolean>`       | `-`     |
| `customRequest`   | Custom upload request implementation.                      | `(options: UploadRequestOptions) => void`           | `-`     |
| `onChange`        | Called whenever the file list changes.                     | `(info: UploadChangeInfo) => void`                  | `-`     |
| `onRemove`        | Called before removing a file. Return false to keep it.    | `(file: UploadFile) => boolean \| Promise<boolean>` | `-`     |
| `children`        | Upload trigger content.                                    | `JSX.Element`                                       | `-`     |
| `prefixCls`       | Custom CSS class prefix.                                   | `string`                                            | `-`     |

### UploadFile

| Property        | Description                                 | Type                                                       | Default |
| --------------- | ------------------------------------------- | ---------------------------------------------------------- | ------- |
| `uid`           | Unique file id.                             | `string`                                                   | `-`     |
| `name`          | File name.                                  | `string`                                                   | `-`     |
| `status`        | Upload status.                              | `'ready' \| 'uploading' \| 'done' \| 'error' \| 'removed'` | `-`     |
| `percent`       | Upload progress percent.                    | `number`                                                   | `-`     |
| `originFileObj` | Original native File object.                | `File`                                                     | `-`     |
| `url`           | Download or preview URL rendered as a link. | `string`                                                   | `-`     |
| `response`      | Upload response value.                      | `unknown`                                                  | `-`     |
| `error`         | Upload error value.                         | `unknown`                                                  | `-`     |

### UploadRequestOptions

| Property     | Description                     | Type                           | Default |
| ------------ | ------------------------------- | ------------------------------ | ------- |
| `file`       | Native File being uploaded.     | `File`                         | `-`     |
| `onProgress` | Report upload progress percent. | `(percent: number) => void`    | `-`     |
| `onSuccess`  | Mark upload as successful.      | `(response?: unknown) => void` | `-`     |
| `onError`    | Mark upload as failed.          | `(error: unknown) => void`     | `-`     |
