# Form

Form stores field values, validates rules, and coordinates common input controls.

### Basic form

```tsx
import { For, createSignal } from 'solid-js'
import {
  Button,
  Checkbox,
  Form,
  Input,
  Space,
  Switch,
  message,
  useForm,
} from '@ant-design-solid/core'

const Demo1 = function () {
  return (
    <Form onFinish={(values) => message.success(`Submitted ${values.username}`)}>
      <Space direction="vertical" class="w-80">
        <Form.Item name="username" rules={[{ required: true, message: 'Username is required' }]}>
          <Input placeholder="Username" />
        </Form.Item>
        <Button type="primary" htmlType="submit">
          Submit
        </Button>
      </Space>
    </Form>
  )
}

export default Demo1
```

### Validate only

```tsx
import { For, createSignal } from 'solid-js'

  Button,
  Checkbox,
  Form,
  Input,
  Space,
  Switch,
  message,
  useForm,

          void validateOnlyForm

            )
        }}
      >
        Validate only
      </Button>
    </Space>
  )
```

### Form instance

```tsx
import { For, createSignal } from 'solid-js'
import {
  Button,
  Checkbox,
  Form,
  Input,
  Space,
  Switch,
  message,
  useForm,
} from '@ant-design-solid/core'

const Demo9 = function () {
  const [instanceForm] = useForm()

  const [summary, setSummary] = createSignal('Submit the form to see values.')

  return (
    <Space direction="vertical" class="w-90">
      <Form
        form={instanceForm}
        initialValues={{ username: 'initial' }}
        onValuesChange={(_, values) => setSummary(JSON.stringify(values))}
      >
        <Form.Item label="Username" name="username">
          <Input />
        </Form.Item>
      </Form>
      <Space>
        <Button onClick={() => instanceForm.setFieldsValue({ username: 'updated' })}>Fill</Button>
        <Button onClick={() => instanceForm.resetFields()}>Reset</Button>
        <Button onClick={() => setSummary(JSON.stringify(instanceForm.getFieldsValue()))}>
          Read
        </Button>
      </Space>
      <span>{summary()}</span>
    </Space>
  )
}

export default Demo9
```

### Advanced dynamic list

```tsx
import { For, createSignal } from 'solid-js'

  Button,
  Checkbox,

  Input,
  Space,
  Switch,
  message,
  useForm,
} from '@ant-design-solid/core'

  return (

        name="passengers"

      >
        {(fields, operation, meta) => (
          <Space direction="vertical" class="w-100">
            <For each={fields()}>
              {(field) => (
                <Space>
                  <Form.Item
                    label={'Passenger ' + (field.name + 1)}

                  >
                    <Input placeholder="Passenger name" />

                  <Button
                    disabled={field.name === 0}
                    onClick={() => operation.move(field.name, field.name - 1)}
                  >
                    Up
                  </Button>
                  <Button onClick={() => operation.move(field.name, field.name + 1)}>Down</Button>
                  <Button danger onClick={() => operation.remove(field.name)}>
                    Remove
                  </Button>
                </Space>
              )}
            </For>
            <Space>
              <Button onClick={() => operation.add({ name: '' })}>Add passenger</Button>
              <Button onClick={() => operation.add({ name: 'Inserted' }, 0)}>Add to top</Button>
              <Button type="primary" htmlType="submit">
                Submit passengers
              </Button>
            </Space>
            <Form.ErrorList errors={meta.errors} />
          </Space>
        )}
      </Form.List>
    </Form>
  )
```

## API

### Form

| Property                | Description                                                                     | Type                                                           | Default        |
| ----------------------- | ------------------------------------------------------------------------------- | -------------------------------------------------------------- | -------------- |
| `form`                  | External form instance created by useForm or createFormInstance.                | `FormInstance`                                                 | `-`            |
| `initialValues`         | Initial field values applied when the form is mounted.                          | `FormValues`                                                   | `-`            |
| `layout`                | Form layout mode.                                                               | `'horizontal' \| 'vertical' \| 'inline'`                       | `'horizontal'` |
| `requiredMark`          | Required mark display style for Form.Item labels.                               | `boolean \| 'optional'`                                        | `true`         |
| `validateTrigger`       | Default event name or names that trigger validation for child Form.Item fields. | `string \| string[]`                                           | `'onChange'`   |
| `labelCol / wrapperCol` | Common label and control layout metadata passed to form items.                  | `unknown`                                                      | `-`            |
| `onFinish`              | Called with all values after validation succeeds on submit.                     | `(values: FormValues) => void`                                 | `-`            |
| `onFinishFailed`        | Called with values and error fields after validation fails.                     | `(errorInfo: ValidateErrorInfo) => void`                       | `-`            |
| `onValuesChange`        | Called whenever a registered field updates its value.                           | `(changedValues: FormValues, allValues: FormValues) => void`   | `-`            |
| `onFieldsChange`        | Called when registered field values or field state change.                      | `(changedFields: FieldData[], allFields: FieldData[]) => void` | `-`            |

### Form.Item

| Property                       | Description                                                                                   | Type                                                | Default                           |
| ------------------------------ | --------------------------------------------------------------------------------------------- | --------------------------------------------------- | --------------------------------- |
| `label`                        | Label displayed before the field control.                                                     | `JSX.Element`                                       | `-`                               |
| `name`                         | Field name used to read and write values in the form store.                                   | `NamePath`                                          | `-`                               |
| `rules`                        | Validation rules for this field.                                                              | `Rule[]`                                            | `-`                               |
| `required`                     | Adds a required rule and displays the required mark.                                          | `boolean`                                           | `false`                           |
| `help`                         | Custom help text shown below the control.                                                     | `JSX.Element`                                       | `-`                               |
| `validateStatus`               | Manually sets validation visual state.                                                        | `'success' \| 'warning' \| 'error' \| 'validating'` | `-`                               |
| `valuePropName`                | Prop name used by child controls for their value.                                             | `string`                                            | `'value'`                         |
| `trigger`                      | Child event name that updates the field value.                                                | `string`                                            | `'onChange'`                      |
| `validateTrigger`              | Event name or names that trigger validation.                                                  | `string \| string[]`                                | `Form validateTrigger or trigger` |
| `shouldUpdate`                 | For unnamed render areas, re-render when form values change or when a predicate returns true. | `boolean \| ((prevValues, nextValues) => boolean)`  | `-`                               |
| `labelCol / wrapperCol`        | Item-level label and control layout metadata.                                                 | `unknown`                                           | `-`                               |
| `labelAlign / colon / tooltip` | Override label alignment, colon display, or render extra label help for this item.            | `'left' \| 'right' / boolean / JSX.Element`         | `-`                               |
| `getValueFromEvent`            | Customize how a field value is extracted from control events.                                 | `(...args: unknown[]) => unknown`                   | `-`                               |
| `getValueProps`                | Customize props derived from the current field value.                                         | `(value: unknown) => Record<string, unknown>`       | `-`                               |
| `normalize`                    | Normalize a value before storing it in the form.                                              | `(value, prevValue, allValues) => unknown`          | `-`                               |
