# Layout

Layout provides the outer structure for pages with header, content, footer, and sider.

### Basic

```tsx
import { Layout, Menu } from '@ant-design-solid/core'
import { createSignal } from 'solid-js'

const panelStyle = {
  padding: '24px',
  background: 'var(--docs-surface-solid)',
  'min-height': '120px',
}

const Demo1 = function () {
  return (
    <Layout>
      <Layout.Header>Header</Layout.Header>
      <Layout.Content style={panelStyle}>Content</Layout.Content>
      <Layout.Footer>Footer</Layout.Footer>
    </Layout>
  )
}

export default Demo1
```

### With sider

```tsx
import { Layout, Menu } from '@ant-design-solid/core'
import { createSignal } from 'solid-js'

const navItems = [
  { key: 'dashboard', label: 'Dashboard' },
  { key: 'settings', label: 'Settings' },
  { key: 'profile', label: 'Profile' },
]

const panelStyle = {
  padding: '24px',
  background: 'var(--docs-surface-solid)',
  'min-height': '120px',
}

const Demo2 = function () {
  return (
    <Layout hasSider style={{ 'min-height': '240px' }}>
      <Layout.Sider>
        <Menu mode="inline" defaultSelectedKeys={['dashboard']} items={navItems} />
      </Layout.Sider>
      <Layout>
        <Layout.Header>Header</Layout.Header>
        <Layout.Content style={panelStyle}>Content</Layout.Content>
        <Layout.Footer>Footer</Layout.Footer>
      </Layout>
    </Layout>
  )
}

export default Demo2
```

### Collapsed sider

```tsx
import { Layout, Menu } from '@ant-design-solid/core'
import { createSignal } from 'solid-js'

const panelStyle = {
  padding: '24px',
  background: 'var(--docs-surface-solid)',
  'min-height': '120px',
}

const Demo3 = function () {
  const [collapsed, setCollapsed] = createSignal(false)

  return (
<button type="button" onClick={() => setCollapsed((value) => !value)}>
          Toggle collapsed
        </button>
        <Layout hasSider style={{ 'min-height': '220px', 'margin-top': '12px' }}>
          <Layout.Sider collapsed={collapsed()} collapsedWidth={64} width={220} theme="light">
            <Menu
              mode="inline"
              inlineCollapsed={collapsed()}
              defaultSelectedKeys={['dashboard']}
              items={[
                { key: 'dashboard', label: 'Dashboard', icon: '🏠' },
                { key: 'settings', label: 'Settings', icon: '⚙️' },
                { key: 'profile', label: 'Profile', icon: '👤' },
              ]}
            />
          </Layout.Sider>
          <Layout.Content style={panelStyle}>Content</Layout.Content>
        </Layout>
  )
}

export default Demo3
```

## API

### Layout

| Property   | Description                         | Type          | Default |
| ---------- | ----------------------------------- | ------------- | ------- |
| `hasSider` | Applies sider-aware layout styling. | `boolean`     | `false` |
| `children` | Layout regions and content.         | `JSX.Element` | `-`     |

### Layout.Header / Layout.Content / Layout.Footer

| Property   | Description     | Type          | Default |
| ---------- | --------------- | ------------- | ------- |
| `children` | Region content. | `JSX.Element` | `-`     |

### Layout.Sider

| Property         | Description                                         | Type                | Default  |
| ---------------- | --------------------------------------------------- | ------------------- | -------- |
| `width`          | Expanded sider width.                               | `number \| string`  | `200`    |
| `collapsedWidth` | Collapsed sider width.                              | `number \| string`  | `80`     |
| `collapsed`      | Whether the sider uses collapsed width and styling. | `boolean`           | `false`  |
| `theme`          | Sider color theme.                                  | `'light' \| 'dark'` | `'dark'` |
| `children`       | Sider content, often navigation.                    | `JSX.Element`       | `-`      |
