# Image

Render images with fallback, placeholder, and preview.

### Basic

```tsx
import { Image, Space } from '@ant-design-solid/core'

const src = 'https://dummyimage.com/240x160/e6f4ff/1677ff.png&text=Image'

const Demo1 = function () {
  return <Image width={240} src={src} alt="Example" />
}

export default Demo1
```

### Placeholder

```tsx
import { Image, Space } from '@ant-design-solid/core'

const src = 'https://dummyimage.com/240x160/e6f4ff/1677ff.png&text=Image'

const Demo2 = function () {
  return <Image width={240} src={src} alt="Placeholder" placeholder="Loading image" />
}

export default Demo2
```

### Fallback

```tsx
import { Image, Space } from '@ant-design-solid/core'

const src = 'https://dummyimage.com/240x160/e6f4ff/1677ff.png&text=Image'

const Demo3 = function () {
  return (
    <Image
      width={240}
      src="/missing.png"
      fallback="https://dummyimage.com/240x160/f5f5f5/999.png&text=Fallback"
      alt="Fallback"
    />
  )
}

export default Demo3
```

### Preview

```tsx
import { Image, Space } from '@ant-design-solid/core'

const src = 'https://dummyimage.com/240x160/e6f4ff/1677ff.png&text=Image'

const Demo4 = function () {
  return (
    <Space>
      <Image width={160} src={src} alt="Preview" />
      <Image preview={false} width={160} src={src} alt="No preview" />
    </Space>
  )
}

export default Demo4
```

## API

| Property            | Description                                                                               | Type                                         | Default |
| ------------------- | ----------------------------------------------------------------------------------------- | -------------------------------------------- | ------- |
| `src`               | Image source URL.                                                                         | `string`                                     | `-`     |
| `alt`               | Image alternate text.                                                                     | `string`                                     | `-`     |
| `width`             | Image width.                                                                              | `number \| string`                           | `-`     |
| `height`            | Image height.                                                                             | `number \| string`                           | `-`     |
| `fallback`          | Fallback image URL used after load failure.                                               | `string`                                     | `-`     |
| `placeholder`       | Placeholder shown before image load completes. Boolean true renders default loading text. | `JSX.Element \| boolean`                     | `-`     |
| `preview`           | Whether clicking opens the preview overlay.                                               | `boolean`                                    | `true`  |
| `prefixCls`         | Custom CSS class prefix.                                                                  | `string`                                     | `-`     |
| `zIndex`            | Custom preview overlay z-index.                                                           | `number`                                     | `-`     |
| `getPopupContainer` | Returns the container used for preview portal rendering.                                  | `(triggerNode?: HTMLElement) => HTMLElement` | `-`     |
| `onLoad`            | Called when the image loads.                                                              | `JSX.EventHandler<HTMLImageElement, Event>`  | `-`     |
| `onError`           | Called when the image fails to load.                                                      | `JSX.EventHandler<HTMLImageElement, Event>`  | `-`     |
