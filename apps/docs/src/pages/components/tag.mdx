# Tag

### Basic

```tsx
import { Tag, Space } from '@ant-design-solid/core'

const Demo1 = function () {
  return (
    <Space wrap>
      <Tag>Default</Tag>
      <Tag color="success">Success</Tag>
      <Tag color="warning">Warning</Tag>
      <Tag color="error">Error</Tag>
      <Tag color="#722ed1">Custom</Tag>
    </Space>
  )
}

export default Demo1
```

### Preset colors

```tsx
import { Tag, Space } from '@ant-design-solid/core'

const Demo2 = function () {
  return (
    <Space wrap>
      <Tag color="success">Success</Tag>
      <Tag color="processing">Processing</Tag>
      <Tag color="warning">Warning</Tag>
      <Tag color="error">Error</Tag>
    </Space>
  )
}

export default Demo2
```

### Custom colors

```tsx
import { Tag, Space } from '@ant-design-solid/core'

const Demo3 = function () {
  return (
    <Space wrap>
      <Tag color="#2f54eb">Custom blue</Tag>
      <Tag color="#13c2c2">Cyan</Tag>
      <Tag color="#722ed1">Purple</Tag>
    </Space>
  )
}

export default Demo3
```

### Closable

```tsx
import { Tag, Space } from '@ant-design-solid/core'

const Demo4 = function () {
  return (
    <Tag closable onClose={() => console.log('closed')}>
      Closable
    </Tag>
  )
}

export default Demo4
```

### Borderless

```tsx
import { Tag, Space } from '@ant-design-solid/core'

const Demo5 = function () {
  return <Tag bordered={false}>Borderless</Tag>
}

export default Demo5
```

## API

| Property   | Description                                      | Type                                                          | Default |
| ---------- | ------------------------------------------------ | ------------------------------------------------------------- | ------- |
| `color`    | Preset semantic color or custom CSS color.       | `'success' \| 'warning' \| 'error' \| 'processing' \| string` | `-`     |
| `closable` | Shows a close button.                            | `boolean`                                                     | `false` |
| `onClose`  | Callback fired when the close button is clicked. | `(event: MouseEvent) => void`                                 | `-`     |
| `bordered` | Whether to render the tag border.                | `boolean`                                                     | `true`  |
| `children` | Tag content.                                     | `JSX.Element`                                                 | `-`     |
