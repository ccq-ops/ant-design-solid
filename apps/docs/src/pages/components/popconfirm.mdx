# Popconfirm

Popconfirm asks users to confirm a potentially risky action.

### Basic

```tsx
import { createSignal } from 'solid-js'
import { Button, Popconfirm, Space, message } from '@ant-design-solid/core'

const Demo1 = function () {
  return (
    <Popconfirm
      title="Delete?"
      description="This action cannot be undone."
      onConfirm={() => {
        message.success('Deleted')
      }}
    >
      <Button danger>Delete</Button>
    </Popconfirm>
  )
}

export default Demo1
```

### Custom text

### Controlled

```tsx
import { createSignal } from 'solid-js'
import { Button, Popconfirm, Space, message } from '@ant-design-solid/core'

const Demo4 = function () {
  return (
    <Space>
      <Popconfirm
        open={open()}
        onOpenChange={setOpen}
        title="Controlled popup"
        onConfirm={() => {
          setOpen(false)
        }}
        onCancel={() => setOpen(false)}
      >
        <Button>Controlled</Button>
      </Popconfirm>
      <Button onClick={() => setOpen(true)}>Open from outside</Button>
    </Space>
  )
}

export default Demo4
```

### Async confirm

```tsx
import { createSignal } from 'solid-js'
import { Button, Popconfirm, Space, message } from '@ant-design-solid/core'

const Demo5 = function () {
  return (
    <Popconfirm
      title="Save changes?"
      description="The popup closes after the promise resolves."
      onConfirm={() =>
        new Promise((resolve) => {
          window.setTimeout(() => {
            message.success('Saved')
            resolve()
          }, 600)
        })
      }
    >
      <Button>Async save</Button>
    </Popconfirm>
  )
}

export default Demo5
```

## API

| Property            | Description                                                       | Type                                         | Default    |
| ------------------- | ----------------------------------------------------------------- | -------------------------------------------- | ---------- |
| `title`             | Main confirmation title.                                          | `JSX.Element`                                | `-`        |
| `description`       | Additional explanatory content.                                   | `JSX.Element`                                | `-`        |
| `open`              | Controlled popup open state.                                      | `boolean`                                    | `-`        |
| `defaultOpen`       | Initial popup open state.                                         | `boolean`                                    | `false`    |
| `disabled`          | Disables opening the popup.                                       | `boolean`                                    | `false`    |
| `placement`         | Popup placement relative to the trigger.                          | `'top' \| 'bottom' \| 'left' \| 'right'`     | `'top'`    |
| `okText`            | OK button text.                                                   | `JSX.Element`                                | `'OK'`     |
| `cancelText`        | Cancel button text.                                               | `JSX.Element`                                | `'Cancel'` |
| `onConfirm`         | Called when OK is clicked. Rejected promises keep the popup open. | `() => void \| Promise<void>`                | `-`        |
| `onCancel`          | Called when Cancel is clicked.                                    | `() => void`                                 | `-`        |
| `onOpenChange`      | Called whenever the popup requests an open-state change.          | `(open: boolean) => void`                    | `-`        |
| `zIndex`            | Overrides popup z-index.                                          | `number`                                     | `-`        |
| `getPopupContainer` | Returns the element used to mount the popup portal.               | `(triggerNode?: HTMLElement) => HTMLElement` | `-`        |
| `children`          | Trigger element.                                                  | `JSX.Element`                                | `-`        |
