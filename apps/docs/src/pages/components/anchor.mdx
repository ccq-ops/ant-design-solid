# Anchor

Anchor renders page-section navigation and highlights the current section while scrolling.

### Basic

```tsx
import { Anchor, Card, Space } from '@ant-design-solid/core'

const items = [
  { href: '#anchor-basic', title: 'Basic' },
  {
    href: '#anchor-examples',
    title: 'Examples',
    children: [{ href: '#anchor-api', title: 'API' }],
  },
]

const Demo1 = function () {
  return (
    <Space align="start" style={{ width: '100%' }}>
      <Anchor affix={false} items={items} />
      <div style={{ flex: 1 }}>
        <Card id="anchor-basic" title="Basic">
          Basic section content.
        </Card>
        <div style={{ height: '24px' }} />
        <Card id="anchor-examples" title="Examples">
          Example section content.
        </Card>
        <div style={{ height: '24px' }} />
        <Card id="anchor-api" title="API">
          API section content.
        </Card>
      </div>
    </Space>
  )
}

export default Demo1
```

### Affixed

```tsx
import { Anchor, Card, Space } from '@ant-design-solid/core'

const items = [
  { href: '#anchor-basic', title: 'Basic' },
  {
    href: '#anchor-examples',
    title: 'Examples',
    children: [{ href: '#anchor-api', title: 'API' }],
  },
]

const Demo2 = function () {
  return <Anchor offsetTop={80} items={items} />
}

export default Demo2
```

## API

### Anchor

| Property           | Description                                                 | Type                                               | Default      |
| ------------------ | ----------------------------------------------------------- | -------------------------------------------------- | ------------ |
| `items`            | Anchor links rendered as navigation items.                  | `AnchorItem[]`                                     | `-`          |
| `affix`            | Wraps the anchor in Affix behavior.                         | `boolean`                                          | `true`       |
| `offsetTop`        | Top offset used by the affixed anchor.                      | `number`                                           | `-`          |
| `targetOffset`     | Scroll target offset used when navigating to anchors.       | `number`                                           | `-`          |
| `getContainer`     | Returns the scroll container observed by the anchor.        | `() => Window \| HTMLElement \| undefined \| null` | `-`          |
| `bounds`           | Pixel tolerance when determining the current active anchor. | `number`                                           | `-`          |
| `getCurrentAnchor` | Overrides active anchor selection.                          | `(activeLink: string) => string`                   | `-`          |
| `direction`        | Anchor layout direction.                                    | `'vertical' \| 'horizontal'`                       | `'vertical'` |
| `replace`          | Uses history replace instead of push for navigation.        | `boolean`                                          | `-`          |
| `onClick`          | Called when a link is clicked.                              | `(event: MouseEvent, link: AnchorItem) => void`    | `-`          |
| `onChange`         | Called when the active link changes.                        | `(currentActiveLink: string) => void`              | `-`          |

### AnchorItem

| Property       | Description                               | Type               | Default |
| -------------- | ----------------------------------------- | ------------------ | ------- |
| `key`          | Optional stable item key.                 | `string \| number` | `-`     |
| `href`         | Anchor href target.                       | `string`           | `-`     |
| `target`       | Link target attribute.                    | `string`           | `-`     |
| `title`        | Link label content.                       | `JSX.Element`      | `-`     |
| `replace`      | Overrides replace behavior for this item. | `boolean`          | `-`     |
| `targetOffset` | Overrides target offset for this item.    | `number`           | `-`     |
| `children`     | Nested anchor items.                      | `AnchorItem[]`     | `-`     |
