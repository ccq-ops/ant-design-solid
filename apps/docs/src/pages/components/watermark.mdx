# Watermark

### Basic

```tsx
import { Watermark } from '@ant-design-solid/core'

const Demo1 = function () {
  return (
    <Watermark content="Ant Design Solid">
      <div class="h-[180px] docs-surface-solid p-6">
        A lightweight watermark overlay for protected content.
      </div>
    </Watermark>
  )
}

export default Demo1
```

### Multiple lines

```tsx
import { Watermark } from '@ant-design-solid/core'

const Demo2 = function () {
  return (
    <Watermark content={['Ant Design Solid', 'Confidential']}>
      <div class="h-[180px] docs-surface-solid p-6">
        Multiple text lines are rendered into the repeated watermark tile.
      </div>
    </Watermark>
  )
}

export default Demo2
```

### Custom style

```tsx
import { Watermark } from '@ant-design-solid/core'

const Demo3 = function () {
  return (
    <Watermark
      content="Draft"
      rotate={-12}
      gap={[64, 64]}
      font={{ color: 'rgba(22, 119, 255, 0.18)', fontSize: 20 }}
    >
      <div class="h-[180px] docs-surface-solid p-6">Customize rotation, gap and text style.</div>
    </Watermark>
  )
}

export default Demo3
```

## API

| Property    | Description                                         | Type                                                                                                            | Default      |
| ----------- | --------------------------------------------------- | --------------------------------------------------------------------------------------------------------------- | ------------ |
| `width`     | Watermark tile width.                               | `number`                                                                                                        | `120`        |
| `height`    | Watermark tile height.                              | `number`                                                                                                        | `64`         |
| `rotate`    | Rotation angle in degrees.                          | `number`                                                                                                        | `-22`        |
| `zIndex`    | Watermark overlay z-index.                          | `number`                                                                                                        | `9`          |
| `image`     | Image URL used instead of text content.             | `string`                                                                                                        | `-`          |
| `content`   | Watermark text content or multiple text lines.      | `string \| string[]`                                                                                            | `-`          |
| `font`      | Text font configuration.                            | `{ color?: string; fontSize?: number; fontWeight?: number \| string; fontFamily?: string; fontStyle?: string }` | `-`          |
| `gap`       | Horizontal and vertical gap between repeated tiles. | `[number, number]`                                                                                              | `[100, 100]` |
| `offset`    | Background position offset.                         | `[number, number]`                                                                                              | `[0, 0]`     |
| `prefixCls` | Custom CSS class prefix.                            | `string`                                                                                                        | `-`          |
| `className` | Additional root class alias.                        | `string`                                                                                                        | `-`          |
| `children`  | Content covered by the watermark.                   | `JSX.Element`                                                                                                   | `-`          |
